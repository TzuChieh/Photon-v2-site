224 inline void beginFrame()
225 {
226 PH_ASSERT(isParentThread());
227 PH_ASSERT(hasWorkerStarted());
228 PH_ASSERT(!isStopRequested());
229
230 // Wait until current frame is available for adding works
231 m_frames.beginProduce();
232 Frame& currentFrame = m_frames.getBufferForProducer();
233
234 //--------------------------------------------------------------------//
235 // vvvvv Works can now be added after the begin produce call vvvvv //
236 //--------------------------------------------------------------------//
237
238#if PH_DEBUG
239 m_isBetweenFrameBeginAndEnd = true;
240#endif
241
242 // Must be empty (already cleared by worker)
243 PH_ASSERT_EQ(currentFrame.numAnyThreadWorks.load(std::memory_order_relaxed), 0);
244
245 onBeginFrame();
246 }

252 inline void endFrame()
253 {
254 PH_ASSERT(isParentThread());
255 PH_ASSERT(hasWorkerStarted());
256
257 onEndFrame();
258
259#if PH_DEBUG
260 m_isBetweenFrameBeginAndEnd = false;
261#endif
262
263 // Mark that current frame is now filled and is ready for processing
264 m_frames.endProduce();
265
266 //--------------------------------------------------------------------------//
267 // ^^^^^ Works can no longer be added after the end produce call ^^^^^ //
268 //--------------------------------------------------------------------------//
269
270 if(isStopRequested())
271 {
272 // Ensure memory effects of `m_frames` will be visible on worker
273 m_hasFinalFrameEnded.test_and_set(std::memory_order_release);
274 m_hasFinalFrameEnded.notify_one();
275 }
276
277 ++m_frameNumber;

286 template<typename Func>
287 inline void addWork(Func&& workFunc)
288 {
289 if constexpr(Work::template TIsStorableFunctor<Func>::value)
290 {
291 addWork(Work(std::forward<Func>(workFunc)));
292 }
293 else
294 {
295 // If `Func` cannot be stored in the internal buffer of `TFunction`, then
296 // we allocate its space in the arena and call it through a wrapper
297
298 auto* const funcPtr = storeFuncInMemoryArena(std::forward<Func>(workFunc));
299 PH_ASSERT(funcPtr);
300
301 addWork(Work(
302 [funcPtr](Args... args) -> R
303 {
304 return (*funcPtr)(std::forward<Args>(args)...);
305 }));
306 }
307 }

315 inline void addWork(Work work)
316 {
317 PH_ASSERT(work.isValid());
318 PH_ASSERT(m_isBetweenFrameBeginAndEnd);
319
320 // Unsafe buffer reference is okay--producer threads should synchronize with frame begin/end
321 Frame& currentFrame = m_frames.unsafeGetBufferReference(m_frames.unsafeGetProduceHead());
322 if(isParentThread())
323 {
324 currentFrame.parentThreadWorkQueue.push_back(std::move(work));
325 }
326 else
327 {
328 PH_ASSERT(!isWorkerThread());
329
330 currentFrame.anyThreadWorkQueue.enqueue(std::move(work));
331
332#if PH_DEBUG
333 currentFrame.numAnyThreadWorks.fetch_add(1, std::memory_order_relaxed);
334#endif
335 }
336 }

404 inline BufferedFrameInfo getFrameInfo(const bool shouldIncludeDetails = false) const
405 {
406 // For all producer threads, it is only safe to access between being/end frame.
407 PH_ASSERT(!isWorkerThread());
408 PH_ASSERT(m_isBetweenFrameBeginAndEnd);
409
410 // Unsafe buffer reference is okay--producer threads should synchronize with frame begin/end
411 const Frame& currentFrame = m_frames.unsafeGetBufferReference(m_frames.unsafeGetProduceHead());
412
413 BufferedFrameInfo info;
414 info.frameNumber = getFrameNumber();
415 info.frameCycleIndex = static_cast<uint32>(m_frames.getProduceHead());
416 info.numParentWorks = currentFrame.parentThreadWorkQueue.size();
417 info.sizeofWorkerThread = sizeof(*this);
418
419 if(shouldIncludeDetails)
420 {
421 info.detail.numEstimatedAnyThreadWorks = currentFrame.anyThreadWorkQueue.estimatedSize();
422 info.detail.numEstimatedTotalWorks = info.numParentWorks + info.detail.numEstimatedAnyThreadWorks;
423 info.detail.estimatedLocalBytesForWorks = info.detail.numEstimatedTotalWorks * sizeof(Work);
424 info.detail.extraBytesAllocatedForWorks = currentFrame.workQueueMemory->numAllocatedBytes();
425
426 info.hasDetail = true;
427 }
428
429 return info;

436 inline void asyncProcessFrames()
437 {
438 PH_ASSERT(isWorkerThread());
439
440 onAsyncWorkerStart();
441
442 do
443 {
444 // Wait until being notified there is new frame that can be processed
445 m_frames.beginConsume();
446 Frame& currentFrame = m_frames.getBufferForConsumer();
447
448 //------------------------------------------------------------------------//
449 // vvvvv Works can now be processed after the begin consume call vvvvv //
450 //------------------------------------------------------------------------//
451
452 processSingleFrame(currentFrame);
453
454 //------------------------------------------------------------------------------//
455 // ^^^^^ Works can no longer be processed after the end consume call ^^^^^ //
456 //------------------------------------------------------------------------------//
457
458 // Mark that current frame is now processed and is ready for producer again
459 m_frames.endConsume();
460
461 } while(!isStopRequested());
462
463 // Ensure memory effects of `m_frames` from parent will be visible. After this, contents in
464 // `m_frames` are for worker thread to use exclusively.
465 m_hasFinalFrameEnded.wait(false, std::memory_order_acquire);
466
467 const std::size_t numRemainingFrames = m_frames.getHeadDistanceDirectly();
468 for(std::size_t i = 0; i < numRemainingFrames; ++i)
469 {
470 const auto consumerHead = m_frames.nextProducerConsumerHead(m_frames.getConsumeHead(), i);
471 Frame& remainingFrame = m_frames.unsafeGetBufferReference(consumerHead);
472
473 // Process all remaining works
474 if(m_shouldFlushBufferBeforeStop)
475 {
476 processSingleFrame(remainingFrame);
477 }
478 // Perform cleanup directly
479 else
480 {
481 remainingFrame.parentThreadWorkQueue.clear();
482 remainingFrame.anyThreadWorkQueue = TAtomicQuasiQueue<Work>{};
483#if PH_DEBUG
484 remainingFrame.numAnyThreadWorks.store(0, std::memory_order_relaxed);
485#endif
486 remainingFrame.workQueueMemory.unsafeGetReference().clear();
487 }
488 }
489
490 onAsyncWorkerStop();
491 }

493 inline void processSingleFrame(Frame& frame)
494 {
495 PH_ASSERT(isWorkerThread());
496
497 // Process works added from parent thread
498 {
499 for(const Work& workFromParentThread : frame.parentThreadWorkQueue)
500 {
501 onAsyncProcessWork(workFromParentThread);
502 }
503 frame.parentThreadWorkQueue.clear();
504 }
505
506 // Process works added from any thread
507 {
508 Work workFromAnyThread;
509 std::size_t numDequeuedAnyThreadWorks = 0;
510 while(frame.anyThreadWorkQueue.tryDequeue(&workFromAnyThread))
511 {
512 ++numDequeuedAnyThreadWorks;
513
514 onAsyncProcessWork(workFromAnyThread);
515
516 if(numDequeuedAnyThreadWorks == maxAnyThreadWorksPerFrame + 1)
517 {
518 PH_LOG(TFrameWorkerThread, Warning,
519 "too many concurrently added works ({}), current safe limit is {}",
520 numDequeuedAnyThreadWorks + frame.anyThreadWorkQueue.estimatedSize(), maxAnyThreadWorksPerFrame);
521 }
522
523#if PH_DEBUG
524 frame.numAnyThreadWorks.fetch_sub(1, std::memory_order_relaxed);
525#endif
526 }
527 }
528
529 // Since there is no consumer contention, there should be no work once `tryDequeue` fails
530 PH_ASSERT_EQ(frame.numAnyThreadWorks.load(std::memory_order_relaxed), 0);
531
532 // It is worker's job to cleanup this frame for producer (for performance reasons;
533 // also, some code is only meant to be called on the worker thread, this includes
534 // non-trivial destructors).
535 // No need to lock the arena here as `isSealedForProcessing` already established
536 // proper ordering for the memory effects to be visible.
537 frame.workQueueMemory.unsafeGetReference().clear();
538 }

570 template<typename Func>
571 inline std::remove_reference_t<Func>* storeFuncInMemoryArena(Func&& workFunc)
572 {
573 PH_ASSERT(!isWorkerThread());
574 PH_ASSERT(m_isBetweenFrameBeginAndEnd);
575
576 // Unsafe buffer reference is okay--producer threads should synchronize with frame begin/end
577 TSynchronized<MemoryArena>& currentArena =
578 m_frames.unsafeGetBufferReference(m_frames.unsafeGetProduceHead()).workQueueMemory;
579
580 // Since it is possible that multiple threads are concurrently storing oversized functor,
581 // we simply lock the arena on allocation to avoid contention
582 return currentArena->make<std::remove_reference_t<Func>>(std::forward<Func>(workFunc));
583 }
