51inline bool PLatLongEnvSphere::mayOverlapVolume(const math::AABB3D& volume) const
52{
53 // Rather than transforming `volume` to local space, we transform the sphere to world space
54 // instead. Under static rigid transform the sphere shape is rotational invariant, we can
55 // inversely translate `volume` and test against the sphere at the origin
56 auto effectiveVolume = volume.getTranslated(m_worldOrigin.negate());
57 return PBasicSphere::mayOverlapVolume(effectiveVolume);
58}
59
60inline math::AABB3D PLatLongEnvSphere::calcAABB() const
61{
62 // Under static rigid transform, only translation will matter to the AABB of a sphere
63 return PBasicSphere::calcAABB().translate(m_worldOrigin);
64}

66inline real PLatLongEnvSphere::calcExtendedArea() const
