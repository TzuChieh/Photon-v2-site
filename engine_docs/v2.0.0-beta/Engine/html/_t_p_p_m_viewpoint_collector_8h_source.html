95template<CViewpoint Viewpoint, CPhoton Photon>
96inline bool TPPMViewpointCollector<Viewpoint, Photon>::impl_onReceiverSampleStart(
97 const math::Vector2D& rasterCoord,
98 const math::Vector2S& sampleIndex,
99 const math::Spectrum& pathThroughput)
100{
101 if(pathThroughput.isZero())
102 {
103 // TODO: should we add a 0-contribution viewpoint?
104 return false;
105 }
106
107 if constexpr(Viewpoint::template has<EViewpointData::RasterCoord>())
108 {
109 m_viewpoint.template set<EViewpointData::RasterCoord>(rasterCoord);
110 }
111 if constexpr(Viewpoint::template has<EViewpointData::Radius>())
112 {
113 m_viewpoint.template set<EViewpointData::Radius>(m_initialKernelRadius);
114 }
115 if constexpr(Viewpoint::template has<EViewpointData::NumPhotons>())
116 {
117 m_viewpoint.template set<EViewpointData::NumPhotons>(0.0_r);
118 }
119 if constexpr(Viewpoint::template has<EViewpointData::Tau>())
120 {
121 m_viewpoint.template set<EViewpointData::Tau>(math::Spectrum(0));
122 }
123 if constexpr(Viewpoint::template has<EViewpointData::ViewRadiance>())
124 {
125 m_viewpoint.template set<EViewpointData::ViewRadiance>(math::Spectrum(0));
126 }
127
128 m_viewRadiance = math::Spectrum(0);
129 m_numBranchedPathViewpoints = 0;
130
131 return true;
132}

134template<CViewpoint Viewpoint, CPhoton Photon>
135inline auto TPPMViewpointCollector<Viewpoint, Photon>::impl_onPathHitSurface(
136 const std::size_t pathLength,
137 const SurfaceHit& surfaceHit,
138 const math::Spectrum& pathThroughput) -> ViewPathTracingPolicy
139{
140 const SurfaceOptics* optics = surfaceHit.getSurfaceOptics();
141 if(!optics)
142 {
143 return ViewPathTracingPolicy().kill();
144 }
145
146 PH_ASSERT_LE(pathLength, m_maxViewpointDepth);
147
148 if constexpr(Viewpoint::template has<EViewpointData::ViewRadiance>())
149 {
150 const auto unaccountedEnergy = estimate_certainly_lost_energy(
151 pathLength,
152 surfaceHit,
153 pathThroughput,
154 m_photonMapInfo,
155 m_scene);
156 m_viewRadiance += unaccountedEnergy;
157 }
158
159 // Below we explicitly list each possible delta phenomenon to emphasize one point: we are
160 // not using RR, the number of added viewpoints can grow exponentially, and the base is number
161 // of phenomena we list (exponent = view path depth).
162
163 // Add viewpoint if there is any non-delta elemental or max viewpoint depth reached.
164 // NOTE: Also merges on glossy! This is just a reference implementation and this can simplify
165 // the logic. Can also merge on delta elemental when max viewpoint depth reached (in this case
166 // some energy is lost due to the view point depth limit).
167 if(pathLength == m_maxViewpointDepth || optics->getAllPhenomena().hasNone({
168 ESurfacePhenomenon::DeltaReflection, ESurfacePhenomenon::DeltaTransmission}))
169 {
170 if constexpr(Viewpoint::template has<EViewpointData::ViewRadiance>())
171 {
172 const auto unaccountedEnergy = estimate_lost_energy_for_merging(
173 pathLength,
174 surfaceHit,
175 pathThroughput,
176 m_photonMapInfo,
177 m_scene);
178 m_viewRadiance += unaccountedEnergy;
179 }
180
181 // If on delta surface when max view depth is reached, the viewpoint is for the view
182 // radiance only. Could be more efficient if we use a separate film or render pass for that.
183 addViewpoint(
184 surfaceHit,
185 surfaceHit.getIncidentRay().getDir().mul(-1),
186 pathThroughput);
187
188 return ViewPathTracingPolicy().kill();
189 }
190 else
191 {
192 PH_ASSERT(optics->getAllPhenomena().hasAny({
193 ESurfacePhenomenon::DeltaReflection, ESurfacePhenomenon::DeltaTransmission}));
194
195 if constexpr(Viewpoint::template has<EViewpointData::ViewRadiance>())
196 {
197 const auto unaccountedEnergy = estimate_lost_energy_for_extending(
198 pathLength,
199 surfaceHit,
200 pathThroughput,
201 m_photonMapInfo,
202 m_scene);
203 m_viewRadiance += unaccountedEnergy;
204 }
205
206 return ViewPathTracingPolicy().
207 traceBranchedPathFor(SurfacePhenomena({
208 ESurfacePhenomenon::DeltaReflection, ESurfacePhenomenon::DeltaTransmission})).
209 useRussianRoulette(false);
210 }
211}

213template<CViewpoint Viewpoint, CPhoton Photon>
214inline void TPPMViewpointCollector<Viewpoint, Photon>::impl_onReceiverSampleEnd()
215{
216 if(m_numBranchedPathViewpoints > 0)
217 {
218 // For each viewpoint gathered on this single receiver sample
219 for(std::size_t i = m_viewpoints.size() - m_numBranchedPathViewpoints; i < m_viewpoints.size(); ++i)
220 {
221 auto& viewpoint = m_viewpoints[i];
222
223 // Normalize current receiver sample's path throughput (since no RR is used
224 // and each view path carries an independent component of the total energy)
225 if constexpr(Viewpoint::template has<EViewpointData::ViewThroughput>())
226 {
227 math::Spectrum pathThroughput = viewpoint.template get<EViewpointData::ViewThroughput>();
228 pathThroughput.mulLocal(static_cast<real>(m_numBranchedPathViewpoints));
229 viewpoint.template set<EViewpointData::ViewThroughput>(pathThroughput);
230 }
231
232 // `m_viewRadiance` is already properly weighted and accumulated
233 if constexpr(Viewpoint::template has<EViewpointData::ViewRadiance>())
234 {
235 viewpoint.template set<EViewpointData::ViewRadiance>(m_viewRadiance);
236 }
237 }
238 }
239 else
240 {

246 //addViewpoint(
247 // SurfaceHit(),
248 // Vector3R(),
249 // Spectrum());

265 const SurfaceHit& surfaceHit,
266 const math::Vector3R& viewDir,
267 const math::Spectrum& pathThroughput)
268{
