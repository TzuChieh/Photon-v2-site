146template<CPhoton Photon, typename PhotonMap>
147inline auto TVPMRadianceEvaluator<Photon, PhotonMap>
148::impl_onPathHitSurface(
149 const std::size_t pathLength,
150 const SurfaceHit& surfaceHit,
151 const math::Spectrum& pathThroughput)
152-> ViewPathTracingPolicy
153{
154 const SurfaceOptics* optics = surfaceHit.getSurfaceOptics();
155 if(!optics)
156 {
157 return ViewPathTracingPolicy().kill();
158 }
159
160 const auto unaccountedEnergy = estimate_certainly_lost_energy(
161 pathLength,
162 surfaceHit,
163 pathThroughput,
164 m_photonMap->getInfo(),
165 m_scene,
166 m_minFullPathLength,
167 m_maxFullPathLength);
168 m_sampledRadiance += unaccountedEnergy;
169
170 constexpr auto smoothEnoughPhenomena = {
171 DIFFUSE_SURFACE_PHENOMENA,
172 ESurfacePhenomenon::NearDiffuseReflection,
173 ESurfacePhenomenon::NearDiffuseTransmission};
174
175 const auto phenomena = optics->getAllPhenomena();
176 const bool isSufficientlyDiffuse = pathLength >= m_glossyMergeBeginLength
177 ? phenomena.hasAny(smoothEnoughPhenomena)
178 : phenomena.hasExactly(DIFFUSE_SURFACE_PHENOMENA);
179
180 if(m_photonMap->canContribute(pathLength, m_minFullPathLength, m_maxFullPathLength) &&
181 phenomena.hasNone(DELTA_SURFACE_PHENOMENA) &&
182 isSufficientlyDiffuse)
183 {
184 const BsdfQueryContext bsdfContext(
185 ALL_SURFACE_ELEMENTALS, lta::ETransport::Importance, lta::ESidednessPolicy::Strict);
186
187 // For path length = N, we can construct light transport path lengths with photon map,
188 // all at once, for the range [N_min, N_max] =
189 // [`N + m_photonMap->minPathLength`, `N + m_photonMap->maxPathLength`].
190 m_sampledRadiance += estimateRadianceWithPhotonMap(
191 surfaceHit, bsdfContext, pathThroughput);
192
193 const auto unaccountedEnergy = estimate_lost_energy_for_merging(
194 pathLength,
195 surfaceHit,
196 pathThroughput,
197 m_photonMap->getInfo(),
198 m_scene,
199 m_minFullPathLength,
200 m_maxFullPathLength);
201 m_sampledRadiance += unaccountedEnergy;
202
203 return ViewPathTracingPolicy().kill();
204 }
205 else
206 {
207 const auto unaccountedEnergy = estimate_lost_energy_for_extending(
208 pathLength,
209 surfaceHit,
210 pathThroughput,
211 m_photonMap->getInfo(),
212 m_scene,
213 m_minFullPathLength,
214 m_maxFullPathLength);
215 m_sampledRadiance += unaccountedEnergy;
216
217 if(pathLength < m_stochasticSampleBeginLength)
218 {
219 return ViewPathTracingPolicy().
220 traceBranchedPathFor(SurfacePhenomena(ALL_SURFACE_PHENOMENA)).
221 useRussianRoulette(false);
222 }
223 else
224 {
225 return ViewPathTracingPolicy().
226 traceSinglePathFor(ALL_SURFACE_ELEMENTALS).
227 useRussianRoulette(true);
228 }
229 }
230}

249template<CPhoton Photon, typename PhotonMap>
250inline math::Spectrum TVPMRadianceEvaluator<Photon, PhotonMap>
251::estimateRadianceWithPhotonMap(
252 const SurfaceHit& X,
253 const BsdfQueryContext& bsdfContext,
254 const math::Spectrum& viewPathThroughput)
255{
256 m_photonCache.clear();
257 m_photonMap->find(X.getPos(), m_kernelRadius, m_photonCache);
258
259 const lta::SurfaceTracer surfaceTracer{m_scene};
260
261 const math::Vector3R L = X.getIncidentRay().getDir().mul(-1);
262 const math::Vector3R Ns = X.getShadingNormal();
263 const math::Vector3R Ng = X.getGeometryNormal();
264
265 BsdfEvalQuery bsdfEval(bsdfContext);
266 math::Spectrum radiance(0);
267 for(const auto& photon : m_photonCache)
268 {
269 const math::Vector3R V = photon.get<EPhotonData::FromDir>();
270 if(!accept_photon_by_surface_topology(photon, Ng, Ns, L, V, bsdfContext.sidedness))
271 {
272 continue;
273 }
274
275 bsdfEval.inputs.set(X, L, V);
276 if(!surfaceTracer.doBsdfEvaluation(bsdfEval))
277 {
278 continue;
279 }
280
281 math::Spectrum throughput(viewPathThroughput);
282 throughput.mulLocal(bsdfEval.outputs.getBsdf());
283 throughput.mulLocal(lta::tamed_importance_BSDF_Ns_corrector(Ns, Ng, V));
284
285 radiance.addLocal(throughput * photon.template get<EPhotonData::ThroughputRadiance>());
286 }
287 radiance.mulLocal(m_kernelDensityNormalizer);
288
289 return radiance;
290}

299template<CPhoton Photon, typename PhotonMap>
300inline void TVPMRadianceEvaluator<Photon, PhotonMap>
301::setKernelRadius(const real radius)
302{
303 PH_ASSERT_GT(radius, 0.0_r);
304
305 m_kernelRadius = radius;
306
307 const real kernelArea = radius * radius * math::constant::pi<real>;
308 m_kernelDensityNormalizer = 1.0_r / (kernelArea * static_cast<real>(m_photonMap->numPaths));
309}
