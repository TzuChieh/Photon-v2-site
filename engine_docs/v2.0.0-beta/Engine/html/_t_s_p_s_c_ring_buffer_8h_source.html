60 inline void beginProduce()
61 {
62#if PH_DEBUG
63 markProducerThread();
64#endif
65
66 PH_ASSERT(isProducerThread());
67
68 // Wait until current item is available for producer
69 Item& currentItem = getCurrentProducerItem();
70 currentItem.isSealedForConsume.wait(true, std::memory_order_acquire);
71
72 // `item` can now be modified by producer
73
74 currentItem.isBetweenProduceBeginAndEnd = true;
75 }
76
80 inline void endProduce()
81 {
82 PH_ASSERT(isProducerThread());
83
84 // Increment before flag release--we want this addition to be visible on consumer
85 m_headDistance.fetch_add(1, std::memory_order_relaxed);
86
87 Item& currentItem = getCurrentProducerItem();
88 currentItem.isBetweenProduceBeginAndEnd = false;
89
90 // `item` can no longer be modified by producer
91
92 // Mark that current item is now set and is ready for being consumed
93 const bool hasAlreadySealed = currentItem.isSealedForConsume.test_and_set(std::memory_order_release);
94 PH_ASSERT(!hasAlreadySealed);
95
96 // Notify the potentially waiting consumer so it can start consuming this item
97 currentItem.isSealedForConsume.notify_one();
98
99 advanceProduceHead();
100 }
101
105 inline void beginConsume()
106 {
107#if PH_DEBUG
108 markConsumerThread();
109#endif
110
111 PH_ASSERT(isConsumerThread());
112
113 // Wait until being notified there is new item that can be consumed
114 Item& currentItem = getCurrentConsumerItem();
115 currentItem.isSealedForConsume.wait(false, std::memory_order_acquire);
116
117 // `item` can now be modified by consumer
118
119 currentItem.isBetweenConsumeBeginAndEnd = true;
120 }
121
125 inline void endConsume()
126 {
127 PH_ASSERT(isConsumerThread());
128
129 // Decrement after flag acquire--will never underflow since every consumption happens after a
130 // production with memory effects made visible
131 m_headDistance.fetch_sub(1, std::memory_order_relaxed);
132
133 Item& currentItem = getCurrentConsumerItem();
134 currentItem.isBetweenConsumeBeginAndEnd = false;
135
136 // `item` can no longer be modified by consumer
137
138 // Mark that current item is now consumed and is ready for use by producer again
139 currentItem.isSealedForConsume.clear(std::memory_order_release);
140
141 // Notify the potentially waiting producer so it can start producing
142 currentItem.isSealedForConsume.notify_one();
143
144 advanceConsumeHead();
145 }

186 inline std::size_t getHeadDistanceDirectly() const
187 {
188 // Pad produce head to the next cycle first to ensure positive result, then mod it.
189 //
190 // `return (m_produceHead + N - m_consumeHead) % N;`
191 //
192 // Above is the straightforward way of calculating distance, however the result is ambiguous on
193 // determining whether the buffer is empty or full--returning 0 on both conditions.
194
195 return lossless_integer_cast<std::size_t>(
196 m_headDistance.load(std::memory_order_relaxed));
197 }
