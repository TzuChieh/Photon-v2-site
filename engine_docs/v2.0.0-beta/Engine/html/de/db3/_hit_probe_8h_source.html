142inline void HitProbe::pushCache(const T& data)
143{
144 static_assert(std::is_trivially_copyable_v<T>,
145 "target type is not cacheable");
146 static_assert(sizeof(T) <= sizeof(m_cache),
147 "not enough cache to store target type, consider increasing config.PH_HIT_PROBE_CACHE_BYTES");
148
149 PH_ASSERT_MSG(m_cacheHead + sizeof(T) <= sizeof(m_cache),
150 "ran out of cache, consider increasing config.PH_HIT_PROBE_CACHE_BYTES \n"
151 "m_cacheHead = " + std::to_string(m_cacheHead) + "\n" +
152 "sizeof(T) = " + std::to_string(sizeof(T)) + "\n" +
153 "sizeof(m_cache) = " + std::to_string(sizeof(m_cache)));
154
155 std::memcpy(m_cache + m_cacheHead, &data, sizeof(T));
156 m_cacheHead += sizeof(T);
157}

159template<typename T>
160inline T HitProbe::popCache()
161{
162 static_assert(std::is_trivially_copyable_v<T>);
163 static_assert(sizeof(T) <= sizeof(m_cache));
164
165 PH_ASSERT_IN_RANGE_INCLUSIVE(m_cacheHead, sizeof(T), sizeof(m_cache));
166 m_cacheHead -= sizeof(T);
167
168 T data;
169 std::memcpy(&data, m_cache + m_cacheHead, sizeof(T));
170 return data;
