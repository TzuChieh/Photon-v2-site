145template<std::size_t N>
146inline bool TMatrix2<T>::solve(
147 const std::array<std::array<T, 2>, N>& bs,
148 std::array<std::array<T, 2>, N>* const out_xs) const
149{
150 static_assert(!std::numeric_limits<T>::is_integer);
151
152 PH_ASSERT(out_xs);
153
154 const T det = determinant();
155 if(std::abs(det) <= std::numeric_limits<T>::epsilon())
156 {
157 return false;
158 }
159
160 const T rcpDeterminant = static_cast<T>(1) / det;
161
162 for(std::size_t i = 0; i < N; ++i)
163 {
164 (*out_xs)[i][0] = (m[1][1] * bs[i][0] - m[0][1] * bs[i][1]) * rcpDeterminant;
165 (*out_xs)[i][1] = (m[0][0] * bs[i][1] - m[1][0] * bs[i][0]) * rcpDeterminant;
166
167 if(!std::isfinite((*out_xs)[i][0]) || !std::isfinite((*out_xs)[i][1]))
168 {
169 return false;
170 }
171 }
172
173 return true;
