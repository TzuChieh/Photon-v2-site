119template<typename T, std::size_t N>
120inline void TFrame<T, N>::sample(
121 TFrame& sampled, const math::TMathFunction2D<float64>& kernel, const uint32 kernelRadiusPx) const
122{
123 if(isEmpty() || sampled.isEmpty() ||
124 kernelRadiusPx == 0)
125 {
126 sampled.fill(0);
127 return;
128 }
129
130 for(uint32 y = 0; y < sampled.heightPx(); ++y)
131 {
132 for(uint32 x = 0; x < sampled.widthPx(); ++x)
133 {
134 const math::Vector2D samplePosPx(
135 (x + 0.5) / sampled.widthPx() * widthPx(),
136 (y + 0.5) / sampled.heightPx() * heightPx());
137
138 // compute filter bounds
139 math::Vector2D filterMin(samplePosPx.sub(kernelRadiusPx));
140 math::Vector2D filterMax(samplePosPx.add(kernelRadiusPx));
141
142 // make filter bounds not to exceed frame bounds
143 filterMin = filterMin.max(math::Vector2D(0, 0));
144 filterMax = filterMax.min(math::Vector2D(widthPx(), heightPx()));
145
146 PH_ASSERT_LE(filterMin.x(), filterMax.x());
147 PH_ASSERT_LE(filterMin.y(), filterMax.y());
148
149 // compute pixel index bounds
150 math::TVector2<int64> x0y0(filterMin.sub(0.5).ceil());
151 math::TVector2<int64> x1y1(filterMax.sub(0.5).floor());
152
153 PH_ASSERT(x0y0.x() >= 0 && x0y0.y() >= 0 &&
154 x1y1.x() < widthPx() && x1y1.y() < heightPx());
155
156 TPixelType<float64> pixelSum = makeMonochromaticPixel<float64>(0);
157 float64 weightSum = 0.0;
158 for(int64 ky = x0y0.y(); ky <= x1y1.y(); ++ky)
159 {
160 for(int64 kx = x0y0.x(); kx <= x1y1.x(); ++kx)
161 {
162 const float64 kernelX = (kx + 0.5) - samplePosPx.x();
163 const float64 kernelY = (ky + 0.5) - samplePosPx.y();
164
165 PixelType pixel;
166 getPixel(static_cast<uint32>(kx), static_cast<uint32>(ky), &pixel);
167 const float64 weight = kernel.evaluate(kernelX, kernelY);
168
169 for(std::size_t i = 0; i < N; ++i)
170 {
171 pixelSum[i] += static_cast<float64>(pixel[i]);
172 }
173 weightSum += weight;
174 }//
175 } // end for each pixel in kernel support
176
177 PixelType sampledPixel;
178 if(weightSum > 0.0)
179 {
180 const float64 reciWeightSum = 1.0 / weightSum;
181 for(std::size_t i = 0; i < N; ++i)
182 {
183 float64 sampledValue = pixelSum[i] * reciWeightSum;
184 sampledValue = math::clamp(sampledValue,
185 static_cast<float64>(std::numeric_limits<T>::lowest()),
186 static_cast<float64>(std::numeric_limits<T>::max()));
187 sampledPixel[i] = static_cast<T>(sampledValue);
188 }
189 }
190 else
191 {
192 sampledPixel = makeMonochromaticPixel(T(0));
193 }
194 sampled.setPixel(x, y, sampledPixel);
195 }//
196 } // end for each pixel in sampled frame
197}

200inline void TFrame<T, N>::flipHorizontally()
201{
202 const uint32 halfWidthPx = m_widthPx / 2;
203
204 for(uint32 y = 0; y < m_heightPx; ++y)
205 {
206 for(uint32 x = 0; x < halfWidthPx; ++x)
207 {
208 const std::size_t leftPixelBegin = calcPixelDataBaseIndex(x, y);
209 const std::size_t rightPixelBegin = calcPixelDataBaseIndex(m_widthPx - 1 - x, y);
210
211 for(std::size_t i = 0; i < N; ++i)
212 {
213 std::swap(m_pixelData[leftPixelBegin + i], m_pixelData[rightPixelBegin + i]);
214 }
215 }
216 }

266template<typename PerPixelOperation>
267inline void TFrame<T, N>::forEachPixel(const math::TAABB2D<uint32>& region, PerPixelOperation op)
268{
269 // OPT
270
271 PixelType pixel;
272 for(uint32 y = region.getMinVertex().y(); y < region.getMaxVertex().y(); ++y)
273 {
274 for(uint32 x = region.getMinVertex().x(); x < region.getMaxVertex().x(); ++x)
275 {
276 getPixel(x, y, &pixel);
277
278 if constexpr(frame_detail::CIsGetPixelOp<PerPixelOperation, T, N>)
279 {
280 op(pixel);
281 }
282 else if constexpr(frame_detail::CIsGetPixelWithCoordsOp<PerPixelOperation, T, N>)
283 {
284 op(x, y, pixel);
285 }
286 else if constexpr(frame_detail::CIsSetPixelOp<PerPixelOperation, T, N>)
287 {
288 setPixel(x, y, op());
289 }
290 else if constexpr(frame_detail::CIsSetPixelWithCoordsOp<PerPixelOperation, T, N>)
291 {
292 setPixel(x, y, op(x, y));
293 }
294 else if constexpr(frame_detail::CIsGetAndSetPixelOp<PerPixelOperation, T, N>)
295 {
296 setPixel(x, y, op(pixel));
297 }
298 else if constexpr(frame_detail::CIsGetAndSetPixelWithCoordsOp<PerPixelOperation, T, N>)
299 {
300 setPixel(x, y, op(x, y, pixel));
301 }
302 else
303 {
304 // Unsupported per pixel operation
305 PH_ASSERT_UNREACHABLE_SECTION();
306 return;
307 }
308 }
309 }
