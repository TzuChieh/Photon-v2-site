104inline bool SidednessAgreement::isSidednessAgreed(
105 const math::Vector3R& Ng,
106 const math::Vector3R& Ns,
107 const math::Vector3R& vec) const
108{
109 switch(m_policy)
110 {
111 case ESidednessPolicy::DoNotCare:
112 case ESidednessPolicy::TrustGeometry:
113 case ESidednessPolicy::TrustShading:
114 {
115 // No agreement issue with single input vector with these policies
116 return true;
117 }
118
119 case ESidednessPolicy::Strict:
120 {
121 return Ng.dot(vec) * Ns.dot(vec) > 0.0_r;
122 }
123
124 default:
125 PH_ASSERT_UNREACHABLE_SECTION();
126 return false;
127 }
128}
