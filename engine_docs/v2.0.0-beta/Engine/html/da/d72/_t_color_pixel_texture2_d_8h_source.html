108template<math::EColorSpace COLOR_SPACE>
109inline void TColorPixelTexture2D<COLOR_SPACE>::sample(
110 const SampleLocation& sampleLocation,
111 math::Spectrum* const out_value) const
112{
113 PH_ASSERT(out_value);
114
115 // FIXME: vec cast
116 const pixel_buffer::TPixel<float64> sampledPixel = samplePixelBuffer(math::Vector2D(sampleLocation.uv()));
117
118 // Get tristimulus color according to pixel buffer layout. Alpha is ignored for this texture.
119 // TODO: we can also add a premultiplied alpha mode
120
121 PH_ASSERT_LE(pixel_texture::num_pixel_elements(m_colorLayout), getPixelBuffer()->numPixelElements());
122
123 math::TVector3<float64> color(0);
124 switch(m_colorLayout)
125 {
126 case pixel_texture::EPixelLayout::R:
127 color.r() = sampledPixel[0];
128 break;
129
130 case pixel_texture::EPixelLayout::G:
131 color.g() = sampledPixel[0];
132 break;
133
134 case pixel_texture::EPixelLayout::B:
135 color.b() = sampledPixel[0];
136 break;
137
138 // FIXME: monochromatic may not mean same value on all components -> HSV?
139 case pixel_texture::EPixelLayout::Monochromatic:
140 color.set(sampledPixel[0]);
141 break;
142
143 case pixel_texture::EPixelLayout::RG:
144 color.r() = sampledPixel[0];
145 color.g() = sampledPixel[1];
146 break;
147
148 case pixel_texture::EPixelLayout::RGB:
149 case pixel_texture::EPixelLayout::RGBA:
150 color.r() = sampledPixel[0];
151 color.g() = sampledPixel[1];
152 color.b() = sampledPixel[2];
153 break;
154
155 case pixel_texture::EPixelLayout::BGR:
156 case pixel_texture::EPixelLayout::BGRA:
157 color.r() = sampledPixel[2];
158 color.g() = sampledPixel[1];
159 color.b() = sampledPixel[0];
160 break;
161
162 case pixel_texture::EPixelLayout::ARGB:
163 color.r() = sampledPixel[1];
164 color.g() = sampledPixel[2];
165 color.b() = sampledPixel[3];
166 break;
167
168 case pixel_texture::EPixelLayout::ABGR:
169 color.r() = sampledPixel[3];
170 color.g() = sampledPixel[2];
171 color.b() = sampledPixel[1];
172 break;
173
174 default:
175 // The layout does not contain color
176 PH_ASSERT_UNREACHABLE_SECTION();
177 break;
178 }
179
180 const math::TristimulusValues castedColor = {
181 static_cast<math::ColorValue>(color.r()),
182 static_cast<math::ColorValue>(color.g()),
183 static_cast<math::ColorValue>(color.b())};
184
185 // Transform color values to the engine color space
186 if constexpr(COLOR_SPACE != math::EColorSpace::Unspecified)
187 {
188 out_value->setTransformed<COLOR_SPACE>(castedColor, sampleLocation.expectedUsage());
189 }
190 // This is raw color values
191 else
192 {
193 // No transform needed when direct conversion is possible
194 if constexpr(math::Spectrum::NUM_VALUES == 3)
195 {
196 out_value->setColorValues(castedColor);
197 }
198 // Best guess: raw data is in linear sRGB (as a fallback)
199 else
200 {
201 out_value->setLinearSRGB(castedColor, sampleLocation.expectedUsage());
202 }
203 }
204}
