145template<CViewpoint Viewpoint, CPhoton Photon>
146inline auto TSPPMRadianceEvaluator<Viewpoint, Photon>::impl_onPathHitSurface(
147 const std::size_t pathLength,
148 const SurfaceHit& surfaceHit,
149 const math::Spectrum& pathThroughput) -> ViewPathTracingPolicy
150{
151 const SurfaceOptics* optics = surfaceHit.getSurfaceOptics();
152 if(!optics)
153 {
154 return ViewPathTracingPolicy().kill();
155 }
156
157 if constexpr(Viewpoint::template has<EViewpointData::ViewRadiance>())
158 {
159 const auto unaccountedEnergy = estimate_certainly_lost_energy(
160 pathLength,
161 surfaceHit,
162 pathThroughput,
163 m_photonMap->getInfo(),
164 m_scene);
165 addViewRadiance(*m_viewpoint, unaccountedEnergy);
166 }
167
168 constexpr auto smoothEnoughPhenomena = {
169 DIFFUSE_SURFACE_PHENOMENA,
170 ESurfacePhenomenon::NearDiffuseReflection,
171 ESurfacePhenomenon::NearDiffuseTransmission};
172
173 if(optics->getAllPhenomena().hasNone(DELTA_SURFACE_PHENOMENA) &&
174 optics->getAllPhenomena().hasAny(smoothEnoughPhenomena))
175 {
176 // For path length = N, we can construct light transport path lengths with photon map,
177 // all at once, for the range [N_min, N_max] =
178 // [`N + m_photonMap->minPathLength`, `N + m_photonMap->maxPathLength`].
179
180 if constexpr(Viewpoint::template has<EViewpointData::SurfaceHit>())
181 {
182 m_viewpoint->template set<EViewpointData::SurfaceHit>(surfaceHit);
183 }
184 if constexpr(Viewpoint::template has<EViewpointData::ViewThroughput>())
185 {
186 m_viewpoint->template set<EViewpointData::ViewThroughput>(pathThroughput);
187 }
188 if constexpr(Viewpoint::template has<EViewpointData::ViewDir>())
189 {
190 m_viewpoint->template set<EViewpointData::ViewDir>(surfaceHit.getIncidentRay().getDir().mul(-1));
191 }
192
193 m_foundTargetHitPoint = true;
194
195 if constexpr(Viewpoint::template has<EViewpointData::ViewRadiance>())
196 {
197 const auto unaccountedEnergy = estimate_lost_energy_for_merging(
198 pathLength,
199 surfaceHit,
200 pathThroughput,
201 m_photonMap->getInfo(),
202 m_scene);
203 addViewRadiance(*m_viewpoint, unaccountedEnergy);
204 }
205
206 return ViewPathTracingPolicy().kill();
207 }
208 else
209 {
210 if constexpr(Viewpoint::template has<EViewpointData::ViewRadiance>())
211 {
212 const auto unaccountedEnergy = estimate_lost_energy_for_extending(
213 pathLength,
214 surfaceHit,
215 pathThroughput,
216 m_photonMap->getInfo(),
217 m_scene);
218 addViewRadiance(*m_viewpoint, unaccountedEnergy);
219 }
220
221 return ViewPathTracingPolicy().
222 traceSinglePathFor(ALL_SURFACE_ELEMENTALS).
223 useRussianRoulette(true);
224 }
225}

227template<CViewpoint Viewpoint, CPhoton Photon>
228inline void TSPPMRadianceEvaluator<Viewpoint, Photon>::impl_onReceiverSampleEnd()
229{
230 if(!m_foundTargetHitPoint)
231 {
232 return;
233 }
234
235 constexpr auto transport = lta::ETransport::Importance;
236 constexpr auto sidednessPolicy = lta::ESidednessPolicy::Strict;
237
238 const lta::SurfaceTracer surfaceTracer{m_scene};
239
240 const SurfaceHit& surfaceHit = m_viewpoint->template get<EViewpointData::SurfaceHit>();
241 const math::Vector3R L = m_viewpoint->template get<EViewpointData::ViewDir>();
242 const math::Vector3R Ng = surfaceHit.getGeometryNormal();
243 const math::Vector3R Ns = surfaceHit.getShadingNormal();
244 const real R = m_viewpoint->template get<EViewpointData::Radius>();
245
246 m_photonCache.clear();
247 m_photonMap->find(surfaceHit.getPos(), R, m_photonCache);
248
249 // FIXME: as a parameter
250 const real alpha = 2.0_r / 3.0_r;
251
252 const real N = m_viewpoint->template get<EViewpointData::NumPhotons>();
253 const real M = static_cast<real>(m_photonCache.size());
254 const real newN = N + alpha * M;
255 const real newR = (N + M) != 0.0_r ? R * std::sqrt(newN / (N + M)) : R;
256
257 const BsdfQueryContext bsdfContext(ALL_SURFACE_ELEMENTALS, transport, sidednessPolicy);
258
259 math::Spectrum tauM(0);
260 BsdfEvalQuery bsdfEval(bsdfContext);
261 for(const auto& photon : m_photonCache)
262 {
263 const math::Vector3R V = photon.template get<EPhotonData::FromDir>();
264 if(!accept_photon_by_surface_topology(photon, Ng, Ns, L, V, bsdfContext.sidedness))
265 {
266 continue;
267 }
268
269 bsdfEval.inputs.set(surfaceHit, L, V);
270 if(!surfaceTracer.doBsdfEvaluation(bsdfEval))
271 {
272 continue;
273 }
274
275 math::Spectrum tau = photon.template get<EPhotonData::ThroughputRadiance>();
276 tau.mulLocal(bsdfEval.outputs.getBsdf());
277 tau.mulLocal(lta::tamed_importance_BSDF_Ns_corrector(Ns, Ng, V));
278
279 tauM.addLocal(tau);
280 }
281 tauM.mulLocal(m_viewpoint->template get<EViewpointData::ViewThroughput>());
282
283 const math::Spectrum tauN = m_viewpoint->template get<EViewpointData::Tau>();
284 const math::Spectrum newTau = (N + M) != 0.0_r ? (tauN + tauM) * (newN / (N + M)) : math::Spectrum(0);
285
286 m_viewpoint->template set<EViewpointData::Radius>(newR);
287 m_viewpoint->template set<EViewpointData::NumPhotons>(newN);
288 m_viewpoint->template set<EViewpointData::Tau>(newTau);
289}

291template<CViewpoint Viewpoint, CPhoton Photon>
292inline void TSPPMRadianceEvaluator<Viewpoint, Photon>::impl_onSampleBatchFinished()
293{
294 PH_ASSERT_GT(m_statisticsRes.product(), 0);
295
296 const math::TAABB2D<float64> statisticsRegion(m_statisticsRegion);
297 const auto rcpStatisticsRes = math::Vector2D(m_statisticsRes).rcp();
298
299 // Evaluate radiance using current iteration's data. Each SPPM statistics gathered is for
300 // estimating average radiance for a small sub-pixel region. We treat each estimated value
301 // as if it is centered on the sub-pixel region it represents.
302 for(int64 y = 0; y < m_statisticsRes.y(); ++y)
303 {
304 for(int64 x = 0; x < m_statisticsRes.x(); ++x)
305 {
306 const auto& viewpoint = m_viewpoints[getViewpointIndex(x, y)];
307 const auto& statisticsRasterCoord = statisticsRegion.xy01ToSurface(
308 (math::Vector2D(math::TVector2<int64>{x, y}) + 0.5) * rcpStatisticsRes);
309
310 // For most accurate result, we should average the viewpoints' radiance by ourselves and
311 // use `setPixel()` to store the value directly. Using `addSample()` will only be correct
312 // if a box filter is used. Nevertheless, the worst result is a slightly overblurred image.
313 // If higher number of viewpoints/regions per pixel were used, it will still approach a
314 // per-sample filtered PT result in the limit (infinite viewpoints per pixel) except on
315 // edges as we are not storing out-of-raster-bound statistics.
316 m_film->addSample(
317 statisticsRasterCoord.x(),
318 statisticsRasterCoord.y(),
319 estimateRadiance(viewpoint));
320 }
321 }

324template<CViewpoint Viewpoint, CPhoton Photon>
325inline math::Spectrum TSPPMRadianceEvaluator<Viewpoint, Photon>::estimateRadiance(
326 const Viewpoint& viewpoint) const
327{
328 const real radius = viewpoint.template get<EViewpointData::Radius>();
329 const real kernelArea = radius * radius * math::constant::pi<real>;
330 const real radianceMultiplier = m_rcpTotalPhotonPaths / kernelArea;
331 const auto tau = viewpoint.template get<EViewpointData::Tau>();
332 const auto viewRadiance = viewpoint.template get<EViewpointData::ViewRadiance>();
333
334 math::Spectrum radiance(tau * radianceMultiplier);
335 radiance += viewRadiance * m_rcpNumViewRadianceSamples;
336 return radiance;
337}

339template<CViewpoint Viewpoint, CPhoton Photon>
340inline std::size_t TSPPMRadianceEvaluator<Viewpoint, Photon>::getViewpointIndex(
341 const int64 sampleX, const int64 sampleY) const
342{
343 PH_ASSERT_IN_RANGE(sampleX, 0, m_statisticsRes.x());
344 PH_ASSERT_IN_RANGE(sampleY, 0, m_statisticsRes.y());
345
346 const std::size_t viewpointIdx = sampleY * m_statisticsRes.x() + sampleX;
347 PH_ASSERT_LT(viewpointIdx, m_viewpoints.size());
348
349 return viewpointIdx;
350}
