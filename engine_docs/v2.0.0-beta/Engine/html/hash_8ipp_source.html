123template<typename T>
124inline uint32 murmur3_32(const T* const data, const std::size_t dataSize, const uint32 seed)
125{
126 /*
127 References:
128 [1] Wiki: https://en.wikipedia.org/wiki/MurmurHash (`murmur3_32()`)
129 [2] aappleby's smhasher: https://github.com/aappleby/smhasher/ (`MurmurHash3_x86_32()`)
130 */
131
132 static_assert(CHAR_BIT == 8);
133 static_assert(std::is_trivially_copyable_v<T>,
134 "`T` should be trivially copyable to be able to interpret it as bytes.");
135
136 constexpr uint32 c1 = 0xCC9E2D51UL;
137 constexpr uint32 c2 = 0x1B873593UL;
138 constexpr int r1 = 15;
139 constexpr int r2 = 13;
140 constexpr uint32 m = 5;
141 constexpr uint32 n = 0xE6546B64UL;
142
143 auto const bytes = reinterpret_cast<const uint8*>(data);
144 const std::size_t numBytes = dataSize * sizeof(T);
145 const std::size_t numBlocks = numBytes / 4;
146
147 uint32 h1 = seed;
148
149 // Body
150
151 // Read in blocks of 4 bytes
152 for(uint32 bi = 0; bi < numBlocks; ++bi)
153 {
154 const auto byteIndex = bi * 4;
155 PH_ASSERT_LT(byteIndex, numBytes);
156
157 uint32 block32;
158 std::copy_n(bytes + byteIndex, 4, reinterpret_cast<uint8*>(&block32));
159
160 // To remove a source of differing results across endiannesses, perform byte swap on
161 // big-endian CPUs (a swap here has no effects on hash properties though)
162 if constexpr(std::endian::native == std::endian::big)
163 {
164 block32 = std::byteswap(block32);
165 }
166
167 uint32 k1 = block32;
168
169 k1 *= c1;
170 k1 = std::rotl(k1, r1);
171 k1 *= c2;
172
173 h1 ^= k1;
174 h1 = std::rotl(h1, r2);
175 h1 = h1 * m + n;
176 }
177
178 // Tail
179
180 auto const tailBytes = bytes + numBlocks * 4;
181
182 uint32 k1 = 0;
183
184 switch(numBytes & 3)
185 {
186 case 3: k1 ^= tailBytes[2] << 16UL;
187 [[fallthrough]];
188 case 2: k1 ^= tailBytes[1] << 8UL;
189 [[fallthrough]];
190 case 1: k1 ^= tailBytes[0];
191 k1 *= c1; k1 = std::rotl(k1, r1); k1 *= c2; h1 ^= k1;
192 };
193
194 // Finalization
195
196 h1 ^= static_cast<uint32>(dataSize);
197 h1 = murmur3_bit_mix_32(h1);
198 return h1;
199}

201inline uint32 permuted_index(uint32 i, uint32 l, uint32 p)
202{
203 // This is the implementation from Kensler's paper: "Correlated Multi-Jittered Sampling".
204 // See https://afnan.io/posts/2019-04-05-explaining-the-hashed-permutation/ for
205 // a nice introduction.
206 // Note that PBRT-v4 also uses the same implementation, see their `PermutationElement()`.
207
208 unsigned w = l - 1;
209 w |= w >> 1;
210 w |= w >> 2;
211 w |= w >> 4;
212 w |= w >> 8;
213 w |= w >> 16;
214
215 do
216 {
217 i ^= p;
218 i *= 0xe170893d;
219 i ^= p >> 16;
220 i ^= (i & w) >> 4;
221 i ^= p >> 8;
222 i *= 0x0929eb3f;
223 i ^= p >> 23;
224 i ^= (i & w) >> 1;
225 i *= 1 | p >> 27;
226 i *= 0x6935fa69;
227 i ^= (i & w) >> 11;
228 i *= 0x74dcb303;
229 i ^= (i & w) >> 2;
230 i *= 0x9e501cc3;
231 i ^= (i & w) >> 2;
232 i *= 0xc860a3df;
233 i &= w;
234 i ^= i >> 5;
235 } while (i >= l);
236
237 return (i + p) % l;
238}
