55 /*
56 For inner nodes: Splitting position <splitPos> along the axis of
57 splitting is stored.
58
59 For leaf nodes: An index value for accessing item is stored in <index>.
60 */
61 union
62 {
63 real u0_splitPos;
64 Index u0_index;
65 };
66
67 /*
68 Assuming Index type has N bits, we divide it into two parts:
69 [N - 2 bits][2 bits]. The [2 bits] part <flags> has the following meaning
70
71 0b00: splitting axis is X // indicates this node is inner
72 0b01: splitting axis is Y //
73 0b10: splitting axis is Z //
74 0b11: this node is leaf
75
76 For inner nodes, <positiveChildIndex> is stored in the upper [N - 2 bits].
77 For leaf nodes, <numItems> is stored in the upper [N - 2 bits] instead.
78 */
79 union
80 {
81 Index u1_flags;
82 Index u1_numItems;
83 Index u1_positiveChildIndex;
84 };
85};

89template<typename Index, bool USE_SINGLE_ITEM_OPT>
90inline auto TIndexedKdtreeNode<Index, USE_SINGLE_ITEM_OPT>
91::makeInner(
92 const real splitPos,
93 const math::constant::AxisIndexType splitAxisIndex,
94 const std::size_t rightChildIndex) -> TIndexedKdtreeNode
95{
96 PH_ASSERT(
97 (!std::isnan(splitPos) && !std::isinf(splitPos)) &&
98 (0 <= splitAxisIndex && splitAxisIndex <= 2) &&
99 (rightChildIndex <= MAX_U1_NUMBER));
100
101 TIndexedKdtreeNode node;
102
103 node.u0_splitPos = splitPos;
104
105 const Index shiftedIndex = static_cast<Index>(rightChildIndex << 2);
106 node.u1_flags = static_cast<Index>(splitAxisIndex);
107 node.u1_positiveChildIndex |= shiftedIndex;
108
109 return node;
110}

112template<typename Index, bool USE_SINGLE_ITEM_OPT>
113inline auto TIndexedKdtreeNode<Index, USE_SINGLE_ITEM_OPT>
114::makeLeaf(
115 const Index index,
116 const std::size_t numItems) -> TIndexedKdtreeNode
117{
118 PH_ASSERT_LE(numItems, MAX_U1_NUMBER);
119
120 TIndexedKdtreeNode node;
121
122 const Index shiftedNumItems = static_cast<Index>(numItems << 2);
123 node.u1_flags = 0b11;
124 node.u1_numItems |= shiftedNumItems;
125
126 node.u0_index = index;
127
128 return node;
129}

131template<typename Index, bool USE_SINGLE_ITEM_OPT>
132inline auto TIndexedKdtreeNode<Index, USE_SINGLE_ITEM_OPT>
133::makeLeaf(
134 const TSpanView<Index> itemIndices,
135 std::vector<Index>& indexBuffer) -> TIndexedKdtreeNode
136{
137 PH_ASSERT(itemIndices.data());
138 PH_ASSERT_LE(itemIndices.size(), MAX_U1_NUMBER);
139
140 if(!(USE_SINGLE_ITEM_OPT && itemIndices.size() == 1))
141 {
142 // For leaf nodes we directly store index offset value in <u0>. If Index is signed type,
143 // value conversion from negative Index back to std::size_t can mess up the stored bits.
144 // So here we make sure that we will not overflow Index.
145 // OPT: try to find an efficient way to make use of the sign bit for storing index
146 PH_ASSERT(indexBuffer.size() <= static_cast<std::size_t>(std::numeric_limits<Index>::max()));
147 const Index indexBufferOffset = static_cast<Index>(indexBuffer.size());
148
149 indexBuffer.insert(indexBuffer.end(), itemIndices.begin(), itemIndices.end());
150
151 return makeLeaf(indexBufferOffset, itemIndices.size());
152 }
153 else
154 {
155 return makeLeaf(itemIndices[0], itemIndices.size());
156 }
157}
