66inline ESdlTypeCategory string_to_category(const std::string_view categoryStr)
67{

72 {category_to_string(ESdlTypeCategory::Ref_Motion), ESdlTypeCategory::Ref_Motion},
73 {category_to_string(ESdlTypeCategory::Ref_Image), ESdlTypeCategory::Ref_Image},
74 {category_to_string(ESdlTypeCategory::Ref_Actor), ESdlTypeCategory::Ref_Actor},
75 {category_to_string(ESdlTypeCategory::Ref_FrameProcessor), ESdlTypeCategory::Ref_FrameProcessor},
76 {category_to_string(ESdlTypeCategory::Ref_Observer), ESdlTypeCategory::Ref_Observer},
77 {category_to_string(ESdlTypeCategory::Ref_SampleSource), ESdlTypeCategory::Ref_SampleSource},
78 {category_to_string(ESdlTypeCategory::Ref_Visualizer), ESdlTypeCategory::Ref_Visualizer},
79 {category_to_string(ESdlTypeCategory::Ref_Option), ESdlTypeCategory::Ref_Option},
80 {category_to_string(ESdlTypeCategory::Ref_Object), ESdlTypeCategory::Ref_Object}
81
82 // Any other strings including the string for `ESdlTypeCategory::Unspecified`
83 // is not explicitly mapped here, as by default they all mapped to `ESdlTypeCategory::Unspecified`
84 };
85
86 const auto& iter = map.find(categoryStr);
87 if(iter == map.end())
88 {
89 // Map to "unspecified" by default
90 return ESdlTypeCategory::Unspecified;
91 }
92
93 return iter->second;
94}
