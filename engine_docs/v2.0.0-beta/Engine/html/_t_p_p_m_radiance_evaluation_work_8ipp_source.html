53template<CPhoton Photon, CViewpoint Viewpoint>
54inline void TPPMRadianceEvaluationWork<Photon, Viewpoint>
55::doWork()
56{
57 PH_PROFILE_SCOPE();
58
59 sanitizeVariables();
60
61 constexpr auto transport = lta::ETransport::Importance;
62 constexpr auto sidednessPolicy = lta::ESidednessPolicy::Strict;
63
64 const BsdfQueryContext bsdfContext(ALL_SURFACE_ELEMENTALS, transport, sidednessPolicy);
65 const lta::SurfaceTracer surfaceTracer{m_scene};
66
67 // For each viewpoint, evaluate radiance using collected data. If the viewpoint is for
68 // view radiance only (and is on a delta optics), the evaluation would still work fine
69 // (just the radius cannot shrink).
70
71 std::vector<FullPhoton> photonCache;
72 for(Viewpoint& viewpoint : m_viewpoints)
73 {
74 const SurfaceHit& surfaceHit = viewpoint.template get<EViewpointData::SurfaceHit>();
75 const math::Vector3R L = viewpoint.template get<EViewpointData::ViewDir>();
76 const math::Vector3R Ng = surfaceHit.getGeometryNormal();
77 const math::Vector3R Ns = surfaceHit.getShadingNormal();
78 const real R = viewpoint.template get<EViewpointData::Radius>();
79
80 photonCache.clear();
81 m_photonMap->find(surfaceHit.getPos(), R, photonCache);
82
83 const real N = viewpoint.template get<EViewpointData::NumPhotons>();
84 const real M = static_cast<real>(photonCache.size());
85 const real newN = N + m_alpha * M;
86 const real newR = (N + M) != 0.0_r ? R * std::sqrt(newN / (N + M)) : R;
87
88 math::Spectrum tauM(0);
89 BsdfEvalQuery bsdfEval(bsdfContext);
90 for(const auto& photon : photonCache)
91 {
92 const math::Vector3R V = photon.get<EPhotonData::FromDir>();
93 if(!accept_photon_by_surface_topology(photon, Ng, Ns, L, V, bsdfContext.sidedness))
94 {
95 continue;
96 }
97
98 bsdfEval.inputs.set(surfaceHit, L, V);
99 if(!surfaceTracer.doBsdfEvaluation(bsdfEval))
100 {
101 continue;
102 }
103
104 math::Spectrum tau = photon.get<EPhotonData::ThroughputRadiance>();
105 tau.mulLocal(bsdfEval.outputs.getBsdf());
106 tau.mulLocal(lta::tamed_importance_BSDF_Ns_corrector(Ns, Ng, V));
107
108 tauM.addLocal(tau);
109 }
110 const math::Spectrum tauN = viewpoint.template get<EViewpointData::Tau>();
111 const math::Spectrum newTau = (N + M) != 0.0_r ? (tauN + tauM) * (newN / (N + M)) : math::Spectrum(0);
112
113 viewpoint.template set<EViewpointData::Radius>(newR);
114 viewpoint.template set<EViewpointData::NumPhotons>(newN);
115 viewpoint.template set<EViewpointData::Tau>(newTau);
116
117 // Evaluate radiance using current iteration's data
118
119 const real kernelArea = newR * newR * math::constant::pi<real>;
120 const real radianceMultiplier = m_rcpTotalPhotonPaths / kernelArea;
121
122 math::Spectrum radiance(viewpoint.template get<EViewpointData::Tau>() * radianceMultiplier);
123 radiance.mulLocal(viewpoint.template get<EViewpointData::ViewThroughput>());
124 radiance.addLocal(viewpoint.template get<EViewpointData::ViewRadiance>());
125
126 const math::Vector2D rasterCoord = viewpoint.template get<EViewpointData::RasterCoord>();
127 m_film->addSample(rasterCoord.x(), rasterCoord.y(), radiance);
128 }
129}
