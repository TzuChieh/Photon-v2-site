148template<typename T>
149inline T log2_floor(const T value)
150{
151 if constexpr(std::is_floating_point_v<T>)
152 {
153 PH_ASSERT_GT(value, T(0));
154
155 return static_cast<T>(std::floor(std::log2(value)));
156 }
157 else
158 {
159 constexpr T NUM_BITS = sizeof(T) * CHAR_BIT;
160
161 static_assert(std::is_integral_v<T>);
162
163 PH_ASSERT_GT(value, T(0));
164
165#if PH_COMPILER_IS_MSVC
166
167 unsigned long first1BitFromLeftIndex = std::numeric_limits<unsigned long>::max();
168 if constexpr(sizeof(T) <= sizeof(unsigned long))
169 {
170 const auto isIndexSet = _BitScanReverse(
171 &first1BitFromLeftIndex,
172 static_cast<unsigned long>(value));
173 PH_ASSERT_GT(isIndexSet, 0);
174 }
175 else
176 {
177 static_assert(sizeof(T) <= sizeof(unsigned __int64),
178 "size of T is too large");
179
180 const auto isIndexSet = _BitScanReverse64(
181 &first1BitFromLeftIndex,
182 static_cast<unsigned __int64>(value));
183 PH_ASSERT_GT(isIndexSet, 0);
184 }
185 PH_ASSERT_IN_RANGE_INCLUSIVE(first1BitFromLeftIndex, T(0), NUM_BITS - 1);
186
187 return static_cast<T>(first1BitFromLeftIndex);
188
189#elif PH_COMPILER_IS_CLANG || PH_COMPILER_IS_GCC
190
191 if constexpr(sizeof(T) <= sizeof(unsigned int))
192 {
193 const int numLeftZeros = __builtin_clz(static_cast<unsigned int>(value));
194 return static_cast<T>(sizeof(unsigned int) * CHAR_BIT - 1 - numLeftZeros);
195 }
196 else if constexpr(sizeof(T) <= sizeof(unsigned long))
197 {
198 const int numLeftZeros = __builtin_clzl(static_cast<unsigned long>(value));
199 return static_cast<T>(sizeof(unsigned long) * CHAR_BIT - 1 - numLeftZeros);
200 }
201 else
202 {
203 static_assert(sizeof(T) <= sizeof(unsigned long long),
204 "size of T is too large");
205
206 const int numLeftZeros = __builtin_clzll(static_cast<unsigned long long>(value));
207 return static_cast<T>(sizeof(unsigned long long) * CHAR_BIT - 1 - numLeftZeros);
208 }
209
210#endif
211 }

229template<typename T, std::enable_if_t<std::is_integral_v<T>, int> = 0>
230inline T wrap(T value, const T lowerBound, const T upperBound)
231{
232 PH_ASSERT_GE(upperBound, lowerBound);
233
234 const T rangeSize = upperBound - lowerBound + static_cast<T>(1);
235
236 if(value < lowerBound)
237 {
238 value += rangeSize * ((lowerBound - value) / rangeSize + static_cast<T>(1));
239
240 // Possibly fail if <value> overflow
241 PH_ASSERT_GE(value, lowerBound);
242 }
243
244 return lowerBound + (value - lowerBound) % rangeSize;
245}

362inline float fast_rcp_sqrt(float x)
363{
364 // TODO:
365 // 1. implement double version in the reference paper and templatize iteration step
366 // 2. is more iteration steps better than doing it in double
367 // 3. establish a standard, e.g., fast_<X>() is guaranteed to have max. rel. error < 1%
368
369 PH_ASSERT_GT(x, 0.0f);
370
371 const float halvedInput = 0.5f * x;
372
373 // Gives initial guess for later refinements
374 auto bits = bitwise_cast<std::uint32_t>(x);
375 bits = 0x5F375A86 - (bits >> 1);
376 x = bitwise_cast<float>(bits);
377
378 // Newton's method, each iteration increases accuracy.
379
380 // Iteration 1, max. relative error < 0.175125%
381 x = x * (1.5f - halvedInput * x * x);
382
383 // Iteration 2, disabled since <x> is already good enough
384 //x = x * (1.5f - halvedInput * x * x);
385
386 return x;
387}

402template<typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
403inline T reverse_bits(const T value)
404{
405 using namespace detail;
406
407 static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>,
408 "Unsupported type detected");
409
410 constexpr std::size_t NUM_BITS = sizeof(T) * CHAR_BIT;
411
412 // Note that arbitrary number of bits can be supported by careful handling of bitwise operations
413 static_assert(NUM_BITS % 8 == 0,
414 "Non-multiple-of-8 integer bits are not supported");
415
416 if constexpr(NUM_BITS == 8)
417 {
418 return T(table::detail::BITS8_REVERSE[value]);
419 }
420 else if constexpr(NUM_BITS == 16)
421 {
422 return (T(table::detail::BITS8_REVERSE[value & 0xFF]) << 8) |
423 (T(table::detail::BITS8_REVERSE[(value >> 8) & 0xFF]));
424 }
425 else if constexpr(NUM_BITS == 32)
426 {
427 return (T(table::detail::BITS8_REVERSE[value & 0xFF]) << 24) |
428 (T(table::detail::BITS8_REVERSE[(value >> 8) & 0xFF]) << 16) |
429 (T(table::detail::BITS8_REVERSE[(value >> 16) & 0xFF]) << 8) |
430 (T(table::detail::BITS8_REVERSE[(value >> 24) & 0xFF]));
431 }
432 else
433 {
434 static_assert(NUM_BITS == 64);
435
436 return (T(table::detail::BITS8_REVERSE[value & 0xFF]) << 56) |
437 (T(table::detail::BITS8_REVERSE[(value >> 8) & 0xFF]) << 48) |
438 (T(table::detail::BITS8_REVERSE[(value >> 16) & 0xFF]) << 40) |
439 (T(table::detail::BITS8_REVERSE[(value >> 24) & 0xFF]) << 32) |
440 (T(table::detail::BITS8_REVERSE[(value >> 32) & 0xFF]) << 24) |
441 (T(table::detail::BITS8_REVERSE[(value >> 40) & 0xFF]) << 16) |
442 (T(table::detail::BITS8_REVERSE[(value >> 48) & 0xFF]) << 8) |
443 (T(table::detail::BITS8_REVERSE[(value >> 56) & 0xFF]));
444 }
445}

451template<std::unsigned_integral UIntType, std::integral RangeType>
452inline UIntType set_bits_in_range(const UIntType bits, const RangeType beginBitIdx, const RangeType endBitIdx)
453{
454 // Inclusive as empty range is allowed, e.g., `beginBitIdx` == `endBitIdx`
455 PH_ASSERT_IN_RANGE_INCLUSIVE(beginBitIdx, 0, endBitIdx);
456 PH_ASSERT_IN_RANGE_INCLUSIVE(endBitIdx, beginBitIdx, sizeof_in_bits<UIntType>());
457
458 // Mask for the bits in range. Constructed by producing required number of 1's then shift
459 // by <beginBitIdx>.
460 // Note that if there is integer promotion to signed types, it should be fine--promoted type
461 // must be wider than the original unsigned type, the bit shifts will never reach sign bit.
462 //
463 const auto bitMask = static_cast<UIntType>(((UIntType(1) << (endBitIdx - beginBitIdx)) - 1) << beginBitIdx);
464
465 return bits | bitMask;
466}

501template<typename T, T MIN, T MAX, std::size_t N>
502inline std::array<T, N> evenly_spaced_array()
503{
504 static_assert(MAX > MIN);
505 static_assert(N >= 2);
506
507 constexpr T TOTAL_SIZE = MAX - MIN;
508
509 std::array<T, N> arr;
510 arr[0] = MIN;
511 for(std::size_t i = 1; i < N - 1; ++i)
512 {
513 // Calculate current fraction i/(N-1), and multiply it with the total size
514 // (we multiply total size with i fist, otherwise integral types will always result in 0)
515 arr[i] = MIN + static_cast<T>(i * TOTAL_SIZE / (N - 1));
516
517 PH_ASSERT_IN_RANGE_INCLUSIVE(arr[i], MIN, MAX);
518 }
519 arr[N - 1] = MAX;
520
521 return arr;
522}

557inline uint16 fp32_to_fp16_bits(const float32 value)
558{
559 static_assert(std::numeric_limits<float32>::is_iec559);
560
561 if(std::abs(value) < 0.0000610352f)
562 {
563 return std::signbit(value) ? 0x8000 : 0x0000;
564 }
565
566 uint16 fp16Bits;
567 uint32 fp32Bits;
568
569 fp32Bits = std::bit_cast<uint32>(value);
570
571 // truncate the mantissa and doing (exp - 127 + 15)
572 fp16Bits = static_cast<uint16>((((fp32Bits & 0x7FFFFFFF) >> 13) - (0x38000000 >> 13)));
573 fp16Bits |= ((fp32Bits & 0x80000000) >> 16);// sign
574
575 return fp16Bits;
576}

580inline float32 fp16_bits_to_fp32(const uint16 fp16Bits)
581{
582 static_assert(std::numeric_limits<float32>::is_iec559);
583
584 // 0 if all bits other than the sign bit are 0
585 if((fp16Bits & 0x7FFF) == 0)
586 {
587 return (fp16Bits & 0x8000) == 0 ? 0.0f : -0.0f;
588 }
589
590 uint32 fp32Bits = ((fp16Bits & 0x8000) << 16);
591 fp32Bits |= (((fp16Bits & 0x7FFF) << 13) + 0x38000000);
592
593 return std::bit_cast<float32>(fp32Bits);
594}

646inline void uint64_mul(const uint64 lhs, const uint64 rhs, uint64& out_high64, uint64& out_low64)
647{
648#if defined(__SIZEOF_INT128__)
649 const auto result128 = __uint128_t(lhs) + __uint128_t(rhs);
650 out_high64 = static_cast<uint64>(result128 >> 64);
651 out_low64 = static_cast<uint64>(result128);
652#elif PH_COMPILER_IS_MSVC
653 out_low64 = _umul128(lhs, rhs, &out_high64);
654#else
655 /*
656 Divide each 64-bit number into two 32-bit parts, then multiplying (or adding) any 32-bit
657 pairs cannot overflow if done using 64-bit arithmetic. If we multiply them using base 2^32
658 arithmetic, a total of 4 products must be shifted and added to obtain the final result.
659 Here is a simple illustration of the algorithm:
660
661 lhs = [ a ][ b ]
662 rhs = [ c ][ d ]
663
664 Treat this like performing a regular multiplication on paper. Note that the multiplication
665 is done in base 2^32:
666
667 [ a ][ b ]
668 x [ c ][ d ]
669 -------------------
670 [a*d][b*d]
671 + [a*c][b*c]
672 -------------------
673
674 The overflowing part of each partial product is the carry (32-bit carry). This is a good reference:
675 https://stackoverflow.com/questions/26852435/reasonably-portable-way-to-get-top-64-bits-from-64x64-bit-multiply
676 */
677
678 const uint64 a = lhs >> 32, b = lhs & 0xFFFFFFFF;
679 const uint64 c = rhs >> 32, d = rhs & 0xFFFFFFFF;
680
681 const uint64 ac = a * c;
682 const uint64 bc = b * c;
683 const uint64 ad = a * d;
684 const uint64 bd = b * d;
685
686 // This is the middle part of the above illustration, which is a sum of three 32-bit numbers
687 // (so it is 34-bit at most): two products b*c and a*d, and the carry from b*d
688 const uint64 mid34 = (bd >> 32) + (bc & 0xFFFFFFFF) + (ad & 0xFFFFFFFF);
689
690 out_high64 = ac + (bc >> 32) + (ad >> 32) + (mid34 >> 32);
691 out_low64 = (mid34 << 32) | (bd & 0xFFFFFFFF);
692#endif
693}
