41template<typename Index>
42inline bool TPIndexedKdTreeTriangleMesh<Index>::isIntersecting(const Ray& ray, HitProbe& probe) const
43{
44 real closestHitT = std::numeric_limits<real>::max();
45 ClosestHitProbeResult closestHit;
46
47 const bool hasHit = m_kdTree.nearestTraversal(
48 ray.getSegment(),
49 [ray, &closestHitT, &closestHit](
50 const Triangle& triangle,
51 const math::TLineSegment<real>& segment,
52 const std::size_t triangleIdx)
53 -> std::optional<real>
54 {
55 real hitT;
56 math::Vector3R hitBary;
57 if(!triangle.isIntersecting(segment, &hitT, &hitBary))
58 {
59 return {};
60 }
61
62 if(hitT < closestHitT)
63 {
64 closestHitT = hitT;
65 closestHit.bary = hitBary;
66 closestHit.faceIdx = lossless_cast<Index>(triangleIdx);
67 }
68
69 return hitT;
70 });
71
72 if(hasHit)
73 {
74 probe.pushBaseHit(this, closestHitT);
75 probe.pushCache(closestHit);
76
77 return true;
78 }
79 else
80 {
81 return false;
82 }

114inline void TPIndexedKdTreeTriangleMesh<Index>::calcHitDetail(

116 HitProbe& probe,
117 HitDetail* const out_detail) const
118{
119 PH_ASSERT(out_detail);
120 PH_ASSERT(m_triangleBuffer);
121
122 const auto closestHit = probe.popCache<ClosestHitProbeResult>();
123 PH_ASSERT_MSG(!closestHit.bary.isZero() && closestHit.bary.isFinite(),
124 closestHit.bary.toString());
125
126 probe.popHit();
127
128 const auto& positions = m_triangleBuffer->getPositions(closestHit.faceIdx);
129 const auto& texCoords = m_triangleBuffer->getTexCoords(closestHit.faceIdx);
130 const auto& normals = m_triangleBuffer->getNormals(closestHit.faceIdx);
131
132 const Triangle triangle(positions);
133 const math::Vector3R position = triangle.barycentricToSurface(closestHit.bary);
134 const math::Vector3R faceNormal = triangle.getFaceNormal();
135
136 const bool hasTexCoords = m_triangleBuffer->hasTexCoord();
137 const bool hasShadingNormal = m_triangleBuffer->hasNormal();
138
139 const auto shadingNormal = hasShadingNormal
140 ? Triangle::interpolate(normals, closestHit.bary).normalize()
141 : faceNormal;
142
143 const auto uvw = hasTexCoords
144 ? Triangle::interpolate(texCoords, closestHit.bary)
145 : math::Vector3R(0);
146
147 PH_ASSERT_MSG(position.isFinite() && shadingNormal.isFinite() && uvw.isFinite(), "\n"
148 "position = " + position.toString() + "\n"
149 "shading-normal = " + shadingNormal.toString() + "\n"
150 "uvw = " + uvw.toString() + "\n");
151
152 // TODO: respect primitive channel
153 // (if it's default channel, use vertex uvw; otherwise, use mapper)
154
155 math::Vector3R dPdU(0.0_r), dPdV(0.0_r);
156 math::Vector3R dNdU(0.0_r), dNdV(0.0_r);
157 if(hasTexCoords)
158 {
159 const std::array<math::Vector2R, 3> uvs = {
160 math::Vector2R(texCoords[0].x(), texCoords[0].y()),
161 math::Vector2R(texCoords[1].x(), texCoords[1].y()),
162 math::Vector2R(texCoords[2].x(), texCoords[2].y())};
163
164 if(!Triangle::calcSurfaceParamDerivatives(
165 positions, uvs, &dPdU, &dPdV))
166 {
167 dPdU.set(0); dPdV.set(0);
168 }
169
170 if(hasShadingNormal && !Triangle::calcSurfaceParamDerivatives(
171 normals, uvs, &dNdU, &dNdV))
172 {
173 dNdU.set(0); dNdV.set(0);
174 }
175 }
176
177 PH_ASSERT_MSG(dPdU.isFinite() && dPdV.isFinite() &&
178 dNdU.isFinite() && dNdV.isFinite(), "\n"
179 "dPdU = " + dPdU.toString() + ", dPdV = " + dPdV.toString() + "\n"
180 "dNdU = " + dNdU.toString() + ", dNdV = " + dNdV.toString() + "\n");
181
182 out_detail->setHitIntrinsics(
183 this,
184 uvw,
185 probe.getHitRayT(),
186 lossless_cast<uint64>(closestHit.faceIdx),
187 FaceTopology({EFaceTopology::Planar, EFaceTopology::Triangular}));
188 out_detail->getHitInfo(ECoordSys::Local).setAttributes(
189 position,
190 faceNormal,
191 shadingNormal);
192 out_detail->getHitInfo(ECoordSys::Local).setDerivatives(
193 dPdU, dPdV, dNdU, dNdV);
194 out_detail->getHitInfo(ECoordSys::World) = out_detail->getHitInfo(ECoordSys::Local);
195
196 constexpr auto meanFactor = 5e-8_r;
197 out_detail->setDistanceErrorFactors(meanFactor, meanFactor * 5e2_r);
198}
