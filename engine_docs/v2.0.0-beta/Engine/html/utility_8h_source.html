105{

107 static_assert(std::is_trivially_copyable_v<Target>);

245template<std::integral DstType, std::integral SrcType>
246inline DstType lossless_integer_cast(const SrcType src)
247{
248 using SrcLimits = std::numeric_limits<SrcType>;
249 using DstLimits = std::numeric_limits<DstType>;
250
251 // Note that the use of `std::cmp_<X>` functions are important as the comparisons
252 // may be signed <-> unsigned comparisons, which may cause signed limits to overflow
253
254 // TODO: we may need to cast src to some integer first to support char and bool types (they are not supported by cmp functions)
255
256 constexpr bool mayHavePositiveOverflow = std::cmp_greater(SrcLimits::max(), DstLimits::max());
257 constexpr bool mayHaveNegativeOverflow = std::cmp_less(SrcLimits::lowest(), DstLimits::lowest());
258
259 if constexpr(mayHavePositiveOverflow)
260 {
261 if(std::cmp_greater(src, DstLimits::max()))
262 {
263 throw_formatted<OverflowException>("cast results in positive overflow: {} exceeds the limit {}",
264 src, DstLimits::max());
265 }
266 }
267
268 if constexpr(mayHaveNegativeOverflow)
269 {
270 if(std::cmp_less(src, DstLimits::lowest()))
271 {
272 throw_formatted<OverflowException>("cast results in negative overflow: {} exceeds the limit {}",
273 src, DstLimits::lowest());
274 }
275 }
276
277 // All possible integer overflow scenarios are checked so it is safe to cast now
278 return static_cast<DstType>(src);
279}

281template<std::floating_point DstType, std::floating_point SrcType>
282inline DstType lossless_float_cast(const SrcType src)
283{
284 // Nothing to do if both types are the same
285 if constexpr(std::is_same_v<SrcType, DstType>)
286 {
287 return src;
288 }
289 // If we are converting to a wider floating-point type, generally it will be lossless
290 else if constexpr(sizeof(DstType) > sizeof(SrcType))
291 {
292 // We need both types to be IEEE-754
293 static_assert(std::numeric_limits<SrcType>::is_iec559);
294 static_assert(std::numeric_limits<DstType>::is_iec559);
295
296 return static_cast<DstType>(src);
297 }
298 // Otherwise, cast to `DstType` then back to `SrcType` and see if there is any difference
299 else
300 {
301 const auto dst = static_cast<DstType>(src);
302 const auto dstBackToSrc = static_cast<SrcType>(dst);
303 if(src != dstBackToSrc)
304 {
305 throw_formatted<NumericException>("cast results in numeric precision loss: {} -> {}",
306 src, dstBackToSrc);
307 }
308
309 return dst;
310 }
311}

318template<CNumber DstType, CNumber SrcType>
319inline DstType lossless_cast(const SrcType src)
320{
321 // Integer -> Integer
322 if constexpr(std::is_integral_v<SrcType> && std::is_integral_v<DstType>)
323 {
324 return lossless_integer_cast<DstType>(src);
325 }
326 // Integer -> Floating-point
327 else if constexpr(std::is_integral_v<SrcType> && std::is_floating_point_v<DstType>)
328 {
329 // TODO
330 PH_ASSERT_UNREACHABLE_SECTION();
331 return 0;
332 }
333 // Floating-point -> Integer
334 else if constexpr(std::is_floating_point_v<SrcType> && std::is_integral_v<DstType>)

336 // TODO
337 PH_ASSERT_UNREACHABLE_SECTION();

339 }
340 // Floating-point -> Floating-point
341 else
342 {
343 static_assert(std::is_floating_point_v<SrcType> && std::is_floating_point_v<DstType>);
