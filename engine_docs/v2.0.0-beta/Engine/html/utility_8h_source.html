126template<typename Target, typename Source>
127inline Target bitwise_cast(const Source& source)
128{
129 static_assert(std::is_trivially_copyable_v<Source>);
130 static_assert(std::is_trivially_copyable_v<Target>);
131
132 static_assert(sizeof(Source) == sizeof(Target),
133 "Source and Target must have the same size");
134

136 {
137 return source;
138 }

141#if __cpp_lib_bit_cast

143#else

148 }

271template<std::integral DstType, std::integral SrcType>
272inline DstType lossless_integer_cast(const SrcType src)
273{
274 using SrcLimits = std::numeric_limits<SrcType>;
275 using DstLimits = std::numeric_limits<DstType>;
276
277 // Note that the use of `std::cmp_<X>` functions are important as the comparisons
278 // may be signed <-> unsigned comparisons, which may cause signed limits to overflow
279
280 // TODO: we may need to cast src to some integer first to support char and bool types (they are not supported by cmp functions)
281
282 constexpr bool mayHavePositiveOverflow = std::cmp_greater(SrcLimits::max(), DstLimits::max());
283 constexpr bool mayHaveNegativeOverflow = std::cmp_less(SrcLimits::lowest(), DstLimits::lowest());
284
285 if constexpr(mayHavePositiveOverflow)
286 {
287 if(std::cmp_greater(src, DstLimits::max()))
288 {
289 throw_formatted<OverflowException>("cast results in positive overflow: {} exceeds the limit {}",
290 src, DstLimits::max());
291 }
292 }
293
294 if constexpr(mayHaveNegativeOverflow)
295 {
296 if(std::cmp_less(src, DstLimits::lowest()))
297 {
298 throw_formatted<OverflowException>("cast results in negative overflow: {} exceeds the limit {}",
299 src, DstLimits::lowest());
300 }
301 }
302
303 // All possible integer overflow scenarios are checked so it is safe to cast now
304 return static_cast<DstType>(src);
305}

307template<std::floating_point DstType, std::floating_point SrcType>
308inline DstType lossless_float_cast(const SrcType src)
309{
310 // Nothing to do if both types are the same
311 if constexpr(std::is_same_v<SrcType, DstType>)
312 {
313 return src;
314 }
315 // If we are converting to a wider floating-point type, generally it will be lossless
316 else if constexpr(sizeof(DstType) > sizeof(SrcType))
317 {
318 // We need both types to be IEEE-754
319 static_assert(std::numeric_limits<SrcType>::is_iec559);
320 static_assert(std::numeric_limits<DstType>::is_iec559);
321
322 return static_cast<DstType>(src);
323 }
324 // Otherwise, cast to `DstType` then back to `SrcType` and see if there is any difference
325 else
326 {
327 const auto dst = static_cast<DstType>(src);
328 const auto dstBackToSrc = static_cast<SrcType>(dst);
329 if(src != dstBackToSrc)
330 {
331 throw_formatted<NumericException>("cast results in numeric precision loss: {} -> {}",
332 src, dstBackToSrc);
333 }
334
335 return dst;
336 }
337}

344template<CNumber DstType, CNumber SrcType>
345inline DstType lossless_cast(const SrcType src)
346{
347 // Integer -> Integer
348 if constexpr(std::is_integral_v<SrcType> && std::is_integral_v<DstType>)
