41template<typename Item, typename Index>
42template<typename TesterFunc, bool IS_ROBUST>
43inline bool TLinearDepthFirstBinaryBvh<Item, Index>
44::nearestTraversal(const TLineSegment<real>& segment, TesterFunc&& intersectionTester) const
45{
46 static_assert(CItemSegmentIntersectionTester<TesterFunc, Item>);
47
48 if(isEmpty())
49 {
50 return false;
51 }
52
53 PH_ASSERT(m_nodes);
54 PH_ASSERT(m_items);
55
56 // Traversal states
57 TArrayStack<Index, TRAVERSAL_STACK_SIZE> todoNodes;
58 Index currentNodeIndex = 0;
59 TLineSegment<real> longestSegment(segment);
60 bool hasHit = false;
61
62 // Precompute common values
63 const bool isNegDir[3] = {
64 segment.getDir().x() < 0,
65 segment.getDir().y() < 0,
66 segment.getDir().z() < 0};
67 const auto rcpSegmentDir = segment.getDir().rcp();
68
69 // Traverse nodes
70 while(true)
71 {
72 PH_ASSERT_LT(currentNodeIndex, m_numNodes);
73 const NodeType& node = m_nodes[currentNodeIndex];
74
75 const auto [aabbMinT, aabbMaxT] = node.getAABB().isIntersectingVolume<IS_ROBUST>(
76 longestSegment, rcpSegmentDir);
77 if(aabbMinT <= aabbMaxT)
78 {
79 if(node.isLeaf())
80 {
81 for(std::size_t i = 0; i < node.numItems(); ++i)
82 {
83 const Item& item = m_items[node.getItemOffset() + i];
84
85 const auto optHitT = intersectionTester(item, longestSegment);
86 if(optHitT)
87 {
88 longestSegment.setMaxT(*optHitT);
89 hasHit = true;
90 }
91 }
92
93 if(todoNodes.isEmpty())
94 {
95 break;
96 }
97 else
98 {
99 currentNodeIndex = todoNodes.top();
100 todoNodes.pop();
101 }
102 }
103 else
104 {
105 if(isNegDir[node.getSplitAxis()])
106 {
107 todoNodes.push(currentNodeIndex + 1);
108 currentNodeIndex = node.getChildOffset();
109 }
110 else
111 {
112 todoNodes.push(node.getChildOffset());
113 currentNodeIndex = currentNodeIndex + 1;
114 }
115 }
116 }
117 else
118 {
119 if(todoNodes.isEmpty())
120 {
121 break;
122 }
123 else
124 {
125 currentNodeIndex = todoNodes.top();
126 todoNodes.pop();
127 }
128 }
129 }
130
131 return hasHit;
132}
