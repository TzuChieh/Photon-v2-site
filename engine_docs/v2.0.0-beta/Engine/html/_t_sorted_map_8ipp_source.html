33{

37 return std::nullopt;

45template<typename KeyType, typename ValueType, typename IsLess>
46inline bool TSortedMap<KeyType, ValueType, IsLess>::unmap(const KeyType& key)
47{
48 const auto optFirstKeyIndex = m_keys.removeValue(key);
49 if(optFirstKeyIndex.has_value())
50 {
51 m_values.erase(m_values.begin() + *optFirstKeyIndex);
52 }
53 return optFirstKeyIndex.has_value();

56template<typename KeyType, typename ValueType, typename IsLess>
57inline bool TSortedMap<KeyType, ValueType, IsLess>::unmapValues(const KeyType& key)
58{
59 std::size_t numRemovedValues = 0;
60 const auto optFirstKeyIndex = m_keys.removeValues(key, &numRemovedValues);
61 if(optFirstKeyIndex.has_value())
62 {
63 // Handles the cases where `m_values` is empty and/or `numRemovedValues` is 0
64 m_values.erase(
65 m_values.begin() + *optFirstKeyIndex,
66 m_values.begin() + *optFirstKeyIndex + numRemovedValues);
67 }
68 return optFirstKeyIndex.has_value();
69}

96template<typename KeyType, typename ValueType, typename IsLess>
97inline const ValueType* TSortedMap<KeyType, ValueType, IsLess>::getValue(const KeyType& key, const std::size_t ithValue) const
98{
99 const auto optFirstKeyIndex = m_keys.indexOfValue(key);
100 if(optFirstKeyIndex.has_value())
101 {
102 // We must make sure that the i-th index starting from the first key index stores a key value
103 // equals to `key`, which means we are still in the range of values that are mapped to `key`.
104 // Do not rely on testing the equality of values as `ValueType` may not be comparable, or
105 // values that compare equal were mapped to different keys.
106
107 const std::size_t ithKeyIndex = *optFirstKeyIndex + ithValue;
108 if(ithKeyIndex < m_keys.size() && m_keys.get(ithKeyIndex) == key)
109 {
110 PH_ASSERT_LT(ithKeyIndex, m_values.size());
111 return &m_values[ithKeyIndex];
112 }
113 else
114 {
115 return nullptr;
116 }
117 }
118 else
119 {
120 return nullptr;
121 }
122}
