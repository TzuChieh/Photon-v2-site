23template<typename T>
24inline TPwcDistribution1D<T>::TPwcDistribution1D(
25 const T min,
26 const T max,
27 const T* const weights,
28 const std::size_t numWeights) :
29
30 m_min(min), m_max(max),
31
32 m_delta(0),
33
34 m_firstNonZeroPdfColumn(0),
35
36 // One more entry since we are storing values on endpoints
37 m_cdf(numWeights + 1, 0)
38{
39 PH_ASSERT(max > min && weights && numWeights > 0);
40 m_delta = (max - min) / static_cast<T>(numWeights);
41
42 // Construct CDF by first integrating the weights
43 m_cdf[0] = 0;
44 for(std::size_t i = 1; i < m_cdf.size(); ++i)
45 {
46 const T wi = weights[i - 1];
47 PH_ASSERT(wi >= 0);
48
49 m_cdf[i] = m_cdf[i - 1] + wi * m_delta;
50 }
51
52 const T sum = m_cdf.back();
53 if(sum > 0)
54 {
55 // Normalize the CDF
56 for(std::size_t i = 1; i < m_cdf.size(); ++i)
57 {
58 // We do not multiply with reciprocal of sum here since we want to
59 // ensure that trailing entries with zero weight are normalized
60 // to one.
61 m_cdf[i] /= sum;
62 }
63 }
64 else
65 {
66 PH_ASSERT_EQ(sum, T(0));
67
68 // If the sum is zero, make a simple linear CDF.
69 for(std::size_t i = 1; i < m_cdf.size(); ++i)
70 {
71 m_cdf[i] = static_cast<T>(i) / static_cast<T>(numWeights);
72 }
73 }
74 PH_ASSERT_EQ(m_cdf.back(), T(1));
75
76 // Find first column with non-zero PDF
77 for(std::size_t i = 0; i < numColumns(); ++i)
78 {
79 if(pdfContinuous(i) > 0)
80 {
81 m_firstNonZeroPdfColumn = i;
82 break;
83 }
84 }
85}

95template<typename T>
96inline std::size_t TPwcDistribution1D<T>::sampleDiscrete(const T sample) const
97{
98 const auto& result = std::lower_bound(m_cdf.begin(), m_cdf.end(), sample);
99 PH_ASSERT_MSG(result != m_cdf.end(),
100 "sample = " + std::to_string(sample) + ", "
101 "last CDF value = " + (m_cdf.empty() ? "(empty CDF)" : std::to_string(m_cdf.back())));
102
103 return result != m_cdf.begin() ? result - m_cdf.begin() - 1 : m_firstNonZeroPdfColumn;
104}

170template<typename T>
171std::size_t TPwcDistribution1D<T>::continuousToDiscrete(const T sample) const
172{
173 PH_ASSERT_MSG(m_min <= sample && sample <= m_max,
174 "m_min = " + std::to_string(m_min) + ", "
175 "m_max = " + std::to_string(m_max) + ", "
176 "sample = " + std::to_string(sample));
177
178 const T continuousColumn = (sample - m_min) / m_delta;
179 return math::clamp(static_cast<std::size_t>(continuousColumn),
180 static_cast<std::size_t>(0), numColumns() - 1);

183template<typename T>
184inline T TPwcDistribution1D<T>::continuouslySampleValue(const T sample, const std::size_t straddledColumn) const
185{
186 PH_ASSERT(straddledColumn < numColumns());
187
188 const T cdfDelta = m_cdf[straddledColumn + 1] - m_cdf[straddledColumn];
189 T overshoot = sample - m_cdf[straddledColumn];
190 if(cdfDelta > 0)
191 {
192 overshoot /= cdfDelta;
193 }
194 PH_ASSERT(0 <= overshoot && overshoot <= 1);
195
196 // NOTE: <sampledValue> may have value straddling neighbor column's range
197 // due to numerical error. Currently this is considered acceptable since
198 // continuous sample does not require precise result.
199 const T sampledValue = m_delta * (overshoot + static_cast<T>(straddledColumn));
200
201 // TODO: check rare, sampled value should rarely exceed [min, max]
202 return math::clamp(sampledValue, m_min, m_max);
203}
