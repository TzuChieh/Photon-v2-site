52template<CPhoton Photon>
53inline void TPhotonPathTracingWork<Photon>::doWork()
54{
55 PH_PROFILE_SCOPE();
56
57 // FIXME: currently we exit immediately when photon buffer is full; we should trace a full path instead
58
59 Timer timer;
60 timer.start();
61
62 constexpr auto transport = lta::ETransport::Importance;
63 constexpr auto sidednessPolicy = lta::ESidednessPolicy::Strict;
64
65 const BsdfQueryContext bsdfContext(ALL_SURFACE_ELEMENTALS, transport, sidednessPolicy);
66 const lta::SurfaceTracer surfaceTracer{m_scene};
67 const lta::RussianRoulette rr{};
68
69 const auto raySampleHandle = m_sampleGenerator->declareStageND(2, m_photonBuffer.size());
70 m_sampleGenerator->prepareSampleBatch();// HACK: check if succeeded
71 auto raySamples = m_sampleGenerator->getSamplesND(raySampleHandle);
72
73 m_numPhotonPaths = 0;
74 std::size_t numStoredPhotons = 0;
75 std::size_t photonStatsCounter = 0;
76 while(numStoredPhotons < m_photonBuffer.size())
77 {
78 ++m_numPhotonPaths;
79
80 SampleFlow sampleFlow = raySamples.readSampleAsFlow();
81
82 // TODO: properly sample time
83 EnergyEmissionSampleQuery energyEmission;
84 energyEmission.inputs.set(Time{});
85
86 // Generate initial hit on the emitter and ray originated from the hit
87 SurfaceHit surfaceHit;
88 Ray tracingRay;
89 {
90 HitProbe probe;
91 m_scene->emitRay(energyEmission, sampleFlow, probe);
92 if(!energyEmission.outputs)
93 {
94 continue;
95 }
96
97 constexpr SurfaceHitReason reason(ESurfaceHitReason::SampledPosDir);
98 tracingRay = energyEmission.outputs.getEmittedRay();
99 surfaceHit = SurfaceHit(tracingRay, probe, reason);
100 }
101
102 PH_ASSERT_IN_RANGE(tracingRay.getDir().lengthSquared(), 0.9_r, 1.1_r);
103
104 // Here 0-bounce lighting is never accounted for
105 math::Spectrum throughputRadiance(energyEmission.outputs.getEmittedEnergy());
106 throughputRadiance.divLocal(energyEmission.outputs.getPdfA());
107 throughputRadiance.divLocal(energyEmission.outputs.getPdfW());
108 throughputRadiance.mulLocal(surfaceHit.getShadingNormal().absDot(tracingRay.getDir()));
109
110 // Start tracing single photon path with at least 1 bounce
111 uint32 photonPathLength = 0;
112 real rrScale = 1.0_r;
113 while(!throughputRadiance.isZero())
114 {
115 PH_ASSERT_LT(photonPathLength, m_maxPhotonPathLength);
116
117 if(!surfaceTracer.traceNextSurfaceFrom(
118 surfaceHit, tracingRay, bsdfContext.sidedness, &surfaceHit))
119 {
120 break;
121 }
122
123 ++photonPathLength;
124
125 // TODO: we can also skip storing this photon if the BSDF has little contribution
126 // (e.g., by measuring its integrated value)
127
128 const PrimitiveMetadata* metadata = surfaceHit.getDetail().getPrimitive()->getMetadata();
129 const SurfaceOptics* optics = metadata->getSurface().getOptics();
130
131 real rrSurvivalProb;
132 if(rr.surviveOnLuminance(throughputRadiance * rrScale, sampleFlow, &rrSurvivalProb))
133 {
134 throughputRadiance *= 1.0_r / rrSurvivalProb;
135
136 if(photonPathLength >= m_minPhotonPathLength)
137 {
138 m_photonBuffer[numStoredPhotons++] = makePhoton(
139 surfaceHit, throughputRadiance, tracingRay, photonPathLength);
140 ++photonStatsCounter;
141 }
142
143 if(numStoredPhotons == m_photonBuffer.size() ||
144 photonPathLength == m_maxPhotonPathLength)
145 {
146 break;
147 }
148 }// end if photon survived
149 else
150 {
151 break;
152 }
153
154 BsdfSampleQuery bsdfSample(bsdfContext);
155 Ray sampledRay;
156 bsdfSample.inputs.set(surfaceHit, tracingRay.getDir().mul(-1));
157 if(!surfaceTracer.doBsdfSample(bsdfSample, sampleFlow, &sampledRay))
158 {
159 break;
160 }
161
162 math::Vector3R V = tracingRay.getDir().mul(-1);
163 math::Vector3R L = bsdfSample.outputs.getL();
164 math::Vector3R Ng = surfaceHit.getGeometryNormal();
165 math::Vector3R Ns = surfaceHit.getShadingNormal();
166 throughputRadiance *= bsdfSample.outputs.getPdfAppliedBsdfCos();
167 throughputRadiance *= lta::tamed_importance_scatter_Ns_corrector(Ns, Ng, L, V);
168
169 // Prevent premature termination of the path due to solid angle compression/expansion
170 rrScale /= bsdfSample.outputs.getRelativeIor2();
171
172 tracingRay = sampledRay;
173 }// end single photon path
174
175 if(photonStatsCounter >= 16384)
176 {
177 timer.stop();
178 setElapsedMs(timer.getDeltaMs());
179
180 if(m_statistics)
181 {
182 m_statistics->addNumTracedPhotons(photonStatsCounter);
183 }
184
185 photonStatsCounter = 0;
186 }
187 }// end while photon buffer is not full
188
189 timer.stop();
190 setElapsedMs(timer.getDeltaMs());
191
192 if(m_statistics)
193 {
194 m_statistics->addNumTracedPhotons(photonStatsCounter);
195 }
196}

210template<CPhoton Photon>
211inline Photon TPhotonPathTracingWork<Photon>::makePhoton(
212 const SurfaceHit& surfaceHit,
213 const math::Spectrum& throughputRadiance,
214 const Ray& tracingRay,
215 const std::size_t pathLength)
216{
217 Photon photon;
218 if constexpr(Photon::template has<EPhotonData::Pos>())
219 {
220 photon.template set<EPhotonData::Pos>(surfaceHit.getPos());
221 }
222 if constexpr(Photon::template has<EPhotonData::ThroughputRadiance>())
223 {
224 photon.template set<EPhotonData::ThroughputRadiance>(throughputRadiance);
225 }
226 if constexpr(Photon::template has<EPhotonData::FromDir>())
227 {
228 photon.template set<EPhotonData::FromDir>(tracingRay.getDir().mul(-1));
229 }
230 if constexpr(Photon::template has<EPhotonData::GeometryNormal>())
231 {
232 photon.template set<EPhotonData::GeometryNormal>(surfaceHit.getGeometryNormal());
233 }
234 if constexpr(Photon::template has<EPhotonData::PathLength>())
235 {
236 photon.template set<EPhotonData::PathLength>(pathLength);
237 }
238
239 return photon;
240}
