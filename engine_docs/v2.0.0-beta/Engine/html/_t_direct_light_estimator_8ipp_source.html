38template<ESidednessPolicy POLICY>
39inline bool TDirectLightEstimator<POLICY>::bsdfSampleEmission(
40 BsdfSampleQuery& bsdfSample,
41 SampleFlow& sampleFlow,
42 math::Spectrum* const out_Le,
43 std::optional<SurfaceHit>* const out_X) const
44{
45 SurfaceHit nextX;
46 const bool foundNextX = SurfaceTracer{m_scene}.bsdfSampleNextSurface(bsdfSample, sampleFlow, &nextX);
47 if(!bsdfSample.outputs.isMeasurable())
48 {
49 return false;
50 }
51
52 math::Spectrum Le(0);
53 if(foundNextX && nextX.getSurfaceEmitter())
54 {
55 nextX.getSurfaceEmitter()->evalEmittedRadiance(nextX, &Le);
56 }
57
58 PH_ASSERT_IN_RANGE(bsdfSample.outputs.getL().lengthSquared(), 0.9_r, 1.1_r);
59
60 if(out_Le) { *out_Le = Le; }
61 if(out_X) { *out_X = foundNextX ? std::make_optional(nextX) : std::nullopt; }
62
63 return true;
64}

66template<ESidednessPolicy POLICY>
67inline bool TDirectLightEstimator<POLICY>::neeSampleEmission(
68 DirectEnergySampleQuery& directSample,
69 SampleFlow& sampleFlow,
70 SurfaceHit* const out_Xe) const
71{
72 PH_ASSERT(isNeeSamplable(directSample.inputs.getX()));
73
74 const SidednessAgreement sidedness{POLICY};
75 const SurfaceHit& X = directSample.inputs.getX();
76
77 HitProbe probe;
78 getScene().genDirectSample(directSample, sampleFlow, probe);
79 if(!directSample.outputs || !sidedness.isSidednessAgreed(X, directSample.getTargetToEmit()))
80 {
81 return false;
82 }
83
84 constexpr SurfaceHitReason reason{ESurfaceHitReason::SampledPos};
85 const SurfaceHit Xe(directSample.outputs.getObservationRay(), probe, reason);
86 const std::optional<Ray> visibilityRay = SurfaceHitRefinery{X}.tryEscape(Xe);
87 if(!visibilityRay || getScene().isOccluding(*visibilityRay))
88 {
89 return false;
90 }
91
92 PH_ASSERT_IN_RANGE(visibilityRay->getDirection().lengthSquared(), 0.9_r, 1.1_r);
93 PH_ASSERT(Xe.getSurfaceEmitter());
94
95 if(out_Xe) { *out_Xe = Xe; }
96
97 return true;
98}

100template<ESidednessPolicy POLICY>
101inline bool TDirectLightEstimator<POLICY>::bsdfSamplePathWithNee(
102 BsdfSampleQuery& bsdfSample,
103 SampleFlow& sampleFlow,
104 math::Spectrum* const out_Lo,
105 std::optional<SurfaceHit>* const out_X) const
106{
107 using MIS = TMis<EMisStyle::Power>;
108
109 const SurfaceHit& X = bsdfSample.inputs.getX();
110 const math::Vector3R V = X.getIncidentRay().getDirection().mul(-1);
111 const math::Vector3R N = X.getShadingNormal();
112 math::Spectrum sampledLo(0);
113
114 // BSDF sample
115 {
116 math::Spectrum bsdfLe;
117 std::optional<SurfaceHit> nextX;
118 if(bsdfSampleEmission(bsdfSample, sampleFlow, &bsdfLe, &nextX) &&
119 bsdfSample.outputs.isMeasurable() &&
120 nextX)
121 {
122 const SurfaceOptics* optics = X.getSurfaceOptics();
123 PH_ASSERT(optics);
124
125 const auto L = bsdfSample.outputs.getL();
126 const auto pdfAppliedBsdf = bsdfSample.outputs.getPdfAppliedBsdf();
127
128 // If NEE cannot sample the same light from `X` (due to delta BSDF, etc.), then we
129 // cannot use MIS weighting to remove NEE contribution as BSDF sampling may not
130 // always have an explicit PDF term.
131
132 // MIS
133 if(isNeeSamplable(X) && nextX->getSurfaceEmitter())
134 {
135 // No need to test occlusion again as `bsdfSampleEmission()` already done that
136 const real neePdfW = neeSamplePdfWUnoccluded(X, *nextX);
137
138 BsdfPdfQuery bsdfPdfQuery{bsdfSample.context};
139 bsdfPdfQuery.inputs.set(bsdfSample);
140 optics->calcBsdfSamplePdfW(bsdfPdfQuery);
141
142 // `isNeeSamplable()` is already checked, but `bsdfSamplePdfW` can still be 0 (e.g.,
143 // sidedness policy or by the distribution itself).
144 const real bsdfSamplePdfW = bsdfPdfQuery.outputs.getSampleDirPdfW();
145 if(bsdfSamplePdfW > 0)
146 {
147 const real misWeighting = MIS{}.weight(bsdfSamplePdfW, neePdfW);
148
149 math::Spectrum weight(pdfAppliedBsdf * N.absDot(L) * misWeighting);
150
151 // Avoid excessive, negative weight and possible NaNs
152 weight.clampLocal(0.0_r, 1e9_r);
153
154 sampledLo += bsdfLe * weight;
155 }
156 }
157 // BSDF sample only
158 else
159 {
160 sampledLo += bsdfLe * pdfAppliedBsdf * N.absDot(L);
161 }
162 }
163
164 // If BSDF sampling failed for whatever reason, we cannot simply return `false`
165 // as NEE could still sample a non-zero outgoing energy
166 if(out_X) { *out_X = nextX; }
167 }
168
169 // NEE
170 if(isNeeSamplable(X))
171 {
172 DirectEnergySampleQuery directSample;
173 directSample.inputs.set(bsdfSample.inputs.getX());
174 if(neeSampleEmission(directSample, sampleFlow) &&
175 directSample.outputs)
176 {
177 // Always do MIS. If NEE can sample a light from `X`, then BSDF light sample should have
178 // no problem doing the same. No need to consider delta light sources as Photon do not
179 // have them.
180
181 const SurfaceOptics* optics = X.getSurfaceOptics();
182 PH_ASSERT(optics);
183
184 BsdfEvalQuery bsdfEval{bsdfSample.context};
185 bsdfEval.inputs.set(X, directSample.getTargetToEmit().normalize(), V);
186 optics->calcBsdf(bsdfEval);
187
188 if(bsdfEval.outputs.isMeasurable())
189 {
190 BsdfPdfQuery bsdfPdfQuery{bsdfSample.context};
191 bsdfPdfQuery.inputs.set(bsdfEval.inputs);
192 optics->calcBsdfSamplePdfW(bsdfPdfQuery);
193
194 const auto L = bsdfEval.inputs.getL();
195 const real neePdfW = directSample.outputs.getPdfW();
196 const real bsdfSamplePdfW = bsdfPdfQuery.outputs.getSampleDirPdfW();
197 const real misWeighting = MIS{}.weight(neePdfW, bsdfSamplePdfW);
198
199 math::Spectrum weight(bsdfEval.outputs.getBsdf() * N.absDot(L) * misWeighting / neePdfW);
200
201 // Avoid excessive, negative weight and possible NaNs
202 weight.clampLocal(0.0_r, 1e9_r);
203
204 sampledLo += directSample.outputs.getEmittedEnergy() * weight;
205 }
206 }
207 }
208
209 if(out_Lo) { *out_Lo = sampledLo; }
210
211 return true;
212}

215inline real TDirectLightEstimator<POLICY>::neeSamplePdfWUnoccluded(
216 const SurfaceHit& X,
217 const SurfaceHit& Xe) const
218{
219 PH_ASSERT(isNeeSamplable(X));
220 PH_ASSERT(Xe.getSurfaceEmitter());
221
222 DirectEnergySamplePdfQuery pdfQuery;
223 pdfQuery.inputs.set(X, Xe);
224
