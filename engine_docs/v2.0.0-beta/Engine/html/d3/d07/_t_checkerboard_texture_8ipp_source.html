48 const SampleLocation& sampleLocation, OutputType* const out_value) const
49{
50 PH_ASSERT(m_oddTexture && m_oddTexture.get() != this);
51 PH_ASSERT(m_evenTexture && m_evenTexture.get() != this);
52
53 const math::Vector3R uvw = sampleLocation.uvw();
54 const int32 uNumber = static_cast<int32>(std::floor(uvw.x() / m_uTileSize));
55 const int32 vNumber = static_cast<int32>(std::floor(uvw.y() / m_vTileSize));
56
57 if(std::abs(uNumber % 2) != std::abs(vNumber % 2))

60 out_value);
