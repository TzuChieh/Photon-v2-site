253template<typename T>
254inline bool TAABB3D<T>::isIntersectingVolumeKajiyaKay(
255 const TLineSegment<T>& segment,
256 T* const out_nearHitT,
257 T* const out_farHitT) const
258{
259 PH_ASSERT(out_nearHitT);
260 PH_ASSERT(out_farHitT);
261
262 // The starting ray interval (tMin, tMax) will be incrementally intersect
263 // against each ray-slab hitting interval (t1, t2) and be updated with the
264 // resulting interval.
265 //
266 // Note that the following implementation is NaN-aware
267 // (tMin & tMax will never have NaNs)
268
269 PH_ASSERT(!std::isnan(segment.getMinT()) && !std::isnan(segment.getMaxT()));
270
271 T tMin = segment.getMinT();
272 T tMax = segment.getMaxT();
273
274 // find ray-slab hitting interval in x-axis then intersect with (tMin, tMax)
275
276 T rcpDir = T(1) / segment.getDir().x();
277 T t1 = (m_minVertex.x() - segment.getOrigin().x()) * rcpDir;
278 T t2 = (m_maxVertex.x() - segment.getOrigin().x()) * rcpDir;
279
280 if(t1 < t2)
281 {
282 tMin = t1 > tMin ? t1 : tMin;
283 tMax = t2 < tMax ? t2 : tMax;
284 }
285 else
286 {
287 tMin = t2 > tMin ? t2 : tMin;
288 tMax = t1 < tMax ? t1 : tMax;
289 }
290
291 if(tMin > tMax)
292 {
293 return false;
294 }
295
296 // find ray-slab hitting interval in y-axis then intersect with (tMin, tMax)
297
298 rcpDir = T(1) / segment.getDir().y();
299 t1 = (m_minVertex.y() - segment.getOrigin().y()) * rcpDir;
300 t2 = (m_maxVertex.y() - segment.getOrigin().y()) * rcpDir;
301
302 if(t1 < t2)
303 {
304 tMin = t1 > tMin ? t1 : tMin;
305 tMax = t2 < tMax ? t2 : tMax;
306 }
307 else
308 {
309 tMin = t2 > tMin ? t2 : tMin;
310 tMax = t1 < tMax ? t1 : tMax;
311 }
312
313 if(tMin > tMax)
314 {
315 return false;
316 }
317
318 // find ray-slab hitting interval in z-axis then intersect with (tMin, tMax)
319
320 rcpDir = T(1) / segment.getDir().z();
321 t1 = (m_minVertex.z() - segment.getOrigin().z()) * rcpDir;
322 t2 = (m_maxVertex.z() - segment.getOrigin().z()) * rcpDir;
323
324 if(t1 < t2)
325 {
326 tMin = t1 > tMin ? t1 : tMin;
327 tMax = t2 < tMax ? t2 : tMax;
328 }
329 else
330 {
331 tMin = t2 > tMin ? t2 : tMin;
332 tMax = t1 < tMax ? t1 : tMax;
333 }
334
335 if(tMin > tMax)
336 {
337 return false;
338 }
339
340 *out_nearHitT = tMin;
341 *out_farHitT = tMax;
342 return true;
343}
