124inline void HitInfo::computeBases()
125{
126 // FIXME: currently this is a hacky way to avoid crossing two parallel vectors
127 // (this condition can rarely happen)
128 // (which will result in 0-length vector and cause normalization to fail)
129
130 m_geometryBasis.setXAxis(m_geometryBasis.getYAxis().cross(m_dPdU));
131 if(m_geometryBasis.getXAxis().lengthSquared() > 0.0_r)
132 {
133 m_geometryBasis.renormalizeXAxis();
134 m_geometryBasis.setZAxis(m_geometryBasis.getXAxis().cross(m_geometryBasis.getYAxis()));
135 }
136 else
137 {
138 m_geometryBasis = math::Basis3R::makeFromUnitY(m_geometryBasis.getYAxis());
139 }
140
141 m_shadingBasis.setXAxis(m_shadingBasis.getYAxis().cross(m_dNdU));
142 if(m_shadingBasis.getXAxis().lengthSquared() > 0.0_r)
143 {
144 m_shadingBasis.renormalizeXAxis();
145 m_shadingBasis.setZAxis(m_shadingBasis.getXAxis().cross(m_shadingBasis.getYAxis()));
146 }
147 else
148 {
149 m_shadingBasis = math::Basis3R::makeFromUnitY(m_shadingBasis.getYAxis());
150 }
151
152 PH_ASSERT_MSG(m_geometryBasis.getYAxis().isFinite() && m_shadingBasis.getYAxis().isFinite(), "\n"
153 "geometry-y-axis = " + m_geometryBasis.getYAxis().toString() + "\n"
154 "shading-y-axis = " + m_shadingBasis.getYAxis().toString() + "\n");
155}
