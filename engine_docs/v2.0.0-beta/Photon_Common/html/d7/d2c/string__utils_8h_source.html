185inline std::string_view cut_head(const std::string_view srcStr, const std::string_view candidates)
186{
187 const auto nonCutPos = srcStr.find_first_not_of(candidates);
188
189 auto cutStr = srcStr;
190
191 // remove_prefix(): behavior is undefined for inputPos > size(), avoid that
192 // with the ternary operator
193 cutStr.remove_prefix(
194 nonCutPos != std::string_view::npos ? nonCutPos : srcStr.size());
195
196 return cutStr;
197}

208inline std::string_view cut_tail(const std::string_view srcStr, const std::string_view candidates)
209{
210 const auto nonCutPos = srcStr.find_last_not_of(candidates);
211
212 auto cutStr = srcStr;
213
214 // remove_suffix(): behavior is undefined for inputPos > size(), avoid that
215 // with the ternary operator;
216 // also, if <nonCutPos> is not npos, <srcStr> will not be empty
217 cutStr.remove_suffix(
218 nonCutPos != std::string_view::npos ? srcStr.size() - 1 - nonCutPos : srcStr.size());
219
220 return cutStr;
221}

277inline std::string_view next_token(
278 std::string_view srcStr,
279 std::string_view* const out_remainingStr = nullptr,
280 const std::string_view tokenSeparators = get_whitespaces<>())
281{
282 srcStr = cut_head(srcStr, tokenSeparators);
283
284 const auto separatorPos = srcStr.find_first_of(tokenSeparators);
285 if(separatorPos != std::string_view::npos)
286 {
287 const auto nextToken = srcStr.substr(0, separatorPos);
288 if(out_remainingStr)
289 {
290 // `separatorPos + 1` as we do not want to include the separator
291 *out_remainingStr = srcStr.substr(separatorPos + 1);
292 }
293
294 return nextToken;
295 }
296 else
297 {
298 return srcStr;

389inline void throw_from_std_errc_if_has_error(const std::errc errorCode)
390{
391 // According to several sources, 0, or zero-initialized std::errc,
392 // indicates no error.
393 //
394 // [1] see the example for std::from_chars
395 // https://en.cppreference.com/w/cpp/utility/from_chars
396 // [2] https://stackoverflow.com/a/63567008
397 //
398 constexpr std::errc NO_ERROR_VALUE = std::errc();
399
400 switch(errorCode)
401 {
402 case NO_ERROR_VALUE:
403 return;
404
405 case std::errc::invalid_argument:
406 throw InvalidArgumentException(
407 "input cannot be interpreted as a numeric value");
408
409 case std::errc::result_out_of_range:
410 throw OverflowException(
411 "result will overflow the arithmetic type");
412
413 case std::errc::value_too_large:
414 throw OutOfRangeException(
415 "result cannot fit in the output buffer");
416
417 default:
418 throw RuntimeException(
419 "unknown error: std::errc = " + std::to_string(
420 static_cast<std::underlying_type_t<std::errc>>(errorCode)));
421 }
422}

545template<std::integral T>
546inline std::size_t stringify_int_alphabetic(
547 T value,
548 char* const out_buffer,
549 const std::size_t bufferSize,
550 const int base)
551{
552 PH_ASSERT(out_buffer);
553 PH_ASSERT_IN_RANGE_INCLUSIVE(base, 2, 62);
554
555 std::size_t numCharsWritten = 0;
556
557 // Write sign
558 if constexpr(std::is_signed_v<T>)
559 {
560 if(value < 0)
561 {
562 if(bufferSize == 0)
563 {
564 throw OutOfRangeException(
565 "result cannot fit in the output buffer: 0 buffer size, cannot hold the negative sign");
566 }
567
568 out_buffer[0] = '-';
569 ++numCharsWritten;
570
571 value = -value;
572 }
573 }
574
575 // Use a temporary buffer, enough to hold base 2 output
576 std::array<unsigned char, sizeof(T) * CHAR_BIT> tmpBuffer;
577 auto tmpBufferEnd = tmpBuffer.end();
578
579 PH_ASSERT_GE(value, 0);
580 do
581 {
582 *(--tmpBufferEnd) = table::BASE62_DIGITS[value % base];
583 value /= base;
584 } while(value > 0);
585
586 auto numDigits = tmpBuffer.end() - tmpBufferEnd;
587 if(numCharsWritten + numDigits > bufferSize)
588 {
589 throw_formatted<OutOfRangeException>(
590 "result cannot fit in the output buffer: need={}, given={}",
591 numCharsWritten + numDigits, bufferSize);
592 }
593 else
594 {
595 std::copy(tmpBufferEnd, tmpBuffer.end(), out_buffer + numCharsWritten);
596 numCharsWritten += numDigits;
597 }
598
599 return numCharsWritten;
600}
