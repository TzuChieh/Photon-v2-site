77 static_cast<std::size_t>(m_numAlpha) *

79 m_table.resize(tableSize, 0.0f);

83inline int TableTIR::calcIndex(const int iCosWi, const int iAlpha, const int iRelIor) const
84{
85 // make sure the indices stay in the limits
86 PH_ASSERT(0 <= iCosWi && iCosWi < m_numCosWi);
87 PH_ASSERT(0 <= iAlpha && iAlpha < m_numAlpha);
88 PH_ASSERT(0 <= iRelIor && iRelIor < m_numRelIor);
89
90 return iRelIor + m_numRelIor * (iAlpha + m_numAlpha * iCosWi);
91}
