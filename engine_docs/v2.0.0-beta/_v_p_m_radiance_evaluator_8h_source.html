106inline auto VPMRadianceEvaluator::impl_onPathHitSurface(
107 const std::size_t pathLength,
108 const SurfaceHit& surfaceHit,
109 const math::Spectrum& pathThroughput) -> ViewPathTracingPolicy
110{
111 // TODO: remove hardcoded max path length
112 constexpr std::size_t MAX_PATH_LENGTH = 5;
113
114 PH_ASSERT_GE(pathLength, 1);
115
116 const BsdfQueryContext bsdfContext(ALL_ELEMENTALS, ETransport::IMPORTANCE, ESidednessPolicy::STRICT);
117 const PrimitiveMetadata* const metadata = surfaceHit.getDetail().getPrimitive()->getMetadata();
118 const SurfaceOptics* const surfaceOptics = metadata->getSurface().getOptics();
119
120 if(metadata->getSurface().getEmitter())
121 {
122 math::Spectrum viewPathRadiance;
123 metadata->getSurface().getEmitter()->evalEmittedRadiance(surfaceHit, &viewPathRadiance);
124 m_sampledRadiance.addLocal(pathThroughput * viewPathRadiance);
125 }
126
127 // FIXME: properly handle delta optics (mixed case)
128
129 if(pathLength < MAX_PATH_LENGTH && surfaceOptics->getAllPhenomena().hasAny({
130 ESurfacePhenomenon::DELTA_REFLECTION,
131 ESurfacePhenomenon::DELTA_TRANSMISSION}))
132 {
133 return ViewPathTracingPolicy().
134 traceBranchedPathFor(SurfacePhenomena({
135 ESurfacePhenomenon::DELTA_REFLECTION,
136 ESurfacePhenomenon::DELTA_TRANSMISSION})).
137 useRussianRoulette(false);
138 }
139
140 PH_ASSERT_LE(pathLength, MAX_PATH_LENGTH);
141
142 m_photonCache.clear();
143 m_photonMap->findWithinRange(surfaceHit.getPosition(), m_kernelRadius, m_photonCache);
144
145 const SurfaceTracer surfaceTracer(m_scene);
146
147 const math::Vector3R L = surfaceHit.getIncidentRay().getDirection().mul(-1);
148 const math::Vector3R Ns = surfaceHit.getShadingNormal();
149 const math::Vector3R Ng = surfaceHit.getGeometryNormal();
150
151 BsdfEvalQuery bsdfEval(bsdfContext);
152 math::Spectrum radiance(0);
153 for(const auto& photon : m_photonCache)
154 {
155 const math::Vector3R V = photon.get<EPhotonData::FROM_DIR>();
156
157 bsdfEval.inputs.set(surfaceHit, L, V);
158 if(!surfaceTracer.doBsdfEvaluation(bsdfEval))
159 {
160 continue;
161 }
162
163 math::Spectrum throughput(pathThroughput);
164 throughput.mulLocal(bsdfEval.outputs.bsdf);
165 throughput.mulLocal(lta::importance_BSDF_Ns_corrector(Ns, Ng, L, V));
166
167 radiance.addLocal(throughput * photon.get<EPhotonData::THROUGHPUT_RADIANCE>());
168 }
169
170 // OPT: cache
171 const real kernelArea = m_kernelRadius * m_kernelRadius * math::constant::pi<real>;
172 const real radianceMultiplier = 1.0_r / (kernelArea * static_cast<real>(m_numPhotonPaths));
173
174 radiance.mulLocal(radianceMultiplier);
175
176 m_sampledRadiance.addLocal(radiance);
177
178 return ViewPathTracingPolicy().kill();
179}
