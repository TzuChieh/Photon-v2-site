110template<typename Viewpoint, typename Photon>
111inline bool TSPPMRadianceEvaluator<Viewpoint, Photon>::impl_onReceiverSampleStart(
112 const math::Vector2D& rasterCoord,
113 const math::Spectrum& pathThroughput)
114{
115 // FIXME: sample res
116
117 const auto fRasterCoord = math::Vector2R(rasterCoord);
118
119 const math::Vector2S regionPosPx(math::Vector2R(
120 math::clamp(
121 fRasterCoord.x() - static_cast<real>(m_film->getEffectiveWindowPx().getMinVertex().x()),
122 0.0_r,
123 static_cast<real>(m_film->getEffectiveResPx().x() - 1)),
124 math::clamp(
125 fRasterCoord.y() - static_cast<real>(m_film->getEffectiveWindowPx().getMinVertex().y()),
126 0.0_r,
127 static_cast<real>(m_film->getEffectiveResPx().y() - 1))));
128
129 const std::size_t viewpointIdx =
130 regionPosPx.y() * static_cast<std::size_t>(m_film->getEffectiveResPx().x()) +
131 regionPosPx.x();
132
133 PH_ASSERT_LT(viewpointIdx, m_numViewpoints);
134 m_viewpoint = &(m_viewpoints[viewpointIdx]);
135
136 m_isViewpointFound = false;
137
138 return true;
139}

141template<typename Viewpoint, typename Photon>
142inline auto TSPPMRadianceEvaluator<Viewpoint, Photon>::impl_onPathHitSurface(
143 const std::size_t pathLength,
144 const SurfaceHit& surfaceHit,
145 const math::Spectrum& pathThroughput) -> ViewPathTracingPolicy
146{
147 const PrimitiveMetadata* metadata = surfaceHit.getDetail().getPrimitive()->getMetadata();
148 const SurfaceOptics* optics = metadata->getSurface().getOptics();
149
150 // TODO: MIS
151 if constexpr(Viewpoint::template has<EViewpointData::VIEW_RADIANCE>())
152 {
153 if(metadata->getSurface().getEmitter())
154 {
155 math::Spectrum viewRadiance;
156 metadata->getSurface().getEmitter()->evalEmittedRadiance(surfaceHit, &viewRadiance);
157 addViewRadiance(pathThroughput * viewRadiance);
158 }
159 }
160
161 // TODO: better handling of glossy optics
162 if(optics->getAllPhenomena().hasAny({ESurfacePhenomenon::DIFFUSE_REFLECTION}) ||
163 pathLength >= m_maxViewpointDepth)
164 {
165 if constexpr(Viewpoint::template has<EViewpointData::SURFACE_HIT>()) {
166 m_viewpoint->template set<EViewpointData::SURFACE_HIT>(surfaceHit);
167 }
168 if constexpr(Viewpoint::template has<EViewpointData::VIEW_THROUGHPUT>()) {
169 m_viewpoint->template set<EViewpointData::VIEW_THROUGHPUT>(pathThroughput);
170 }
171 if constexpr(Viewpoint::template has<EViewpointData::VIEW_DIR>()) {
172 m_viewpoint->template set<EViewpointData::VIEW_DIR>(surfaceHit.getIncidentRay().getDirection().mul(-1));
173 }
174
175 m_isViewpointFound = true;
176
177 return ViewPathTracingPolicy().kill();
178 }
179 else
180 {
181 return ViewPathTracingPolicy().
182 traceSinglePathFor(ALL_ELEMENTALS).
183 useRussianRoulette(true);
184 }
185}

187template<typename Viewpoint, typename Photon>
188inline void TSPPMRadianceEvaluator<Viewpoint, Photon>::impl_onReceiverSampleEnd()
189{
190 if(!m_isViewpointFound)
191 {
192 return;
193 }
194
195 const SurfaceTracer surfaceTracer(m_scene);
196
197 const SurfaceHit& surfaceHit = m_viewpoint->template get<EViewpointData::SURFACE_HIT>();
198 const math::Vector3R L = m_viewpoint->template get<EViewpointData::VIEW_DIR>();
199 const math::Vector3R Ng = surfaceHit.getGeometryNormal();
200 const math::Vector3R Ns = surfaceHit.getShadingNormal();
201 const real R = m_viewpoint->template get<EViewpointData::RADIUS>();
202
203 m_photonCache.clear();
204 m_photonMap->findWithinRange(surfaceHit.getPosition(), R, m_photonCache);
205
206 // FIXME: as a parameter
207 const real alpha = 2.0_r / 3.0_r;
208
209 const real N = m_viewpoint->template get<EViewpointData::NUM_PHOTONS>();
210 const real M = static_cast<real>(m_photonCache.size());
211 const real newN = N + alpha * M;
212 const real newR = (N + M) != 0.0_r ? R * std::sqrt(newN / (N + M)) : R;
213
214 const BsdfQueryContext bsdfContext(ALL_ELEMENTALS, ETransport::IMPORTANCE, ESidednessPolicy::STRICT);
215
216 math::Spectrum tauM(0);
217 BsdfEvalQuery bsdfEval(bsdfContext);
218 for(const auto& photon : m_photonCache)
219 {
220 const math::Vector3R V = photon.template get<EPhotonData::FROM_DIR>();
221
222 bsdfEval.inputs.set(surfaceHit, L, V);
223 if(!surfaceTracer.doBsdfEvaluation(bsdfEval))
224 {
225 continue;
226 }
227
228 math::Spectrum tau = photon.template get<EPhotonData::THROUGHPUT_RADIANCE>();
229 tau.mulLocal(bsdfEval.outputs.bsdf);
230 tau.mulLocal(lta::importance_BSDF_Ns_corrector(Ns, Ng, L, V));
231
232 tauM.addLocal(tau);
233 }
234 tauM.mulLocal(m_viewpoint->template get<EViewpointData::VIEW_THROUGHPUT>());
235
236 const math::Spectrum tauN = m_viewpoint->template get<EViewpointData::TAU>();
237 const math::Spectrum newTau = (N + M) != 0.0_r ? (tauN + tauM) * (newN / (N + M)) : math::Spectrum(0);
238
239 m_viewpoint->template set<EViewpointData::RADIUS>(newR);
240 m_viewpoint->template set<EViewpointData::NUM_PHOTONS>(newN);
241 m_viewpoint->template set<EViewpointData::TAU>(newTau);
242}

244template<typename Viewpoint, typename Photon>
245inline void TSPPMRadianceEvaluator<Viewpoint, Photon>::impl_onSampleBatchFinished()
246{
247 // evaluate radiance using current iteration's data
248 for(int64 y = m_filmRegion.getMinVertex().y(); y < m_filmRegion.getMaxVertex().y(); ++y)
249 {
250 for(int64 x = m_filmRegion.getMinVertex().x(); x < m_filmRegion.getMaxVertex().x(); ++x)
251 {
252 const std::size_t viewpointIdx =
253 (y - m_film->getEffectiveWindowPx().getMinVertex().y()) * static_cast<std::size_t>(m_film->getEffectiveResPx().x()) +
254 (x - m_film->getEffectiveWindowPx().getMinVertex().x());
255
256 PH_ASSERT_LT(viewpointIdx, m_numViewpoints);
257 const auto& viewpoint = m_viewpoints[viewpointIdx];
258
259 const real radius = viewpoint.template get<EViewpointData::RADIUS>();
260 const real kernelArea = radius * radius * math::constant::pi<real>;
261 const real radianceMultiplier = 1.0_r / (kernelArea * static_cast<real>(m_numPhotonPaths));
262
263 math::Spectrum radiance(viewpoint.template get<EViewpointData::TAU>() * radianceMultiplier);
264 radiance.addLocal(viewpoint.template get<EViewpointData::VIEW_RADIANCE>() / static_cast<real>(m_numSamplesPerPixel));
265 m_film->setPixel(static_cast<float64>(x), static_cast<float64>(y), radiance);
266 }
267 }
