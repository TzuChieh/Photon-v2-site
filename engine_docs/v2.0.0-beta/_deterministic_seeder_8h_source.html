30inline T DeterministicSeeder::nextSeed()
31{
32 static_assert(CHAR_BIT == 8);
33
34 // The goal here is to generate values that will not repeat themselves unless this method is
35 // called a large amount of times (e.g., more than 2^32 times), and the values should look
36 // uncorrelated. MurmurHash3 generates unique hash values for <= 32-bit inputs and is used
37 // here (seed values used here are randomly chosen prime numbers).
38
39 const auto number = nextUInt32Number();
40 if constexpr(std::is_same_v<T, uint32>)
41 {
42 return murmur3_32(number, 1236161);
43 }
44 else if constexpr(std::is_same_v<T, uint64>)
45 {
46 const auto lower32 = uint64(murmur3_32(number, 2237617));
47 const auto upper32 = uint64(murmur3_32(number, 3237557)) << 32;
48 return upper32 | lower32;
49 }
50 else
51 {
52 PH_STATIC_ASSERT_DEPENDENT_FALSE(T,
53 "Unsupported seed type `T`. You can provide your own implementation.");
54 }
55}
