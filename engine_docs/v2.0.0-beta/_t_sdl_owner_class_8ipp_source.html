30template<typename Owner, typename FieldSet>
31inline std::shared_ptr<ISdlResource> TSdlOwnerClass<Owner, FieldSet>::createResource() const
32{
33 if constexpr(!std::is_abstract_v<Owner> && std::is_default_constructible_v<Owner>)
34 {
35 if(!isBlueprint() && allowCreateFromClass())
36 {
37 return std::make_shared<Owner>();
38 }
39 else
40 {
41 return nullptr;
42 }
43 }
44 else
45 {
46 // The resource is abstract and/or not default-constructible, make sure this is intended
47 PH_ASSERT_MSG(isBlueprint() || !allowCreateFromClass(),
48 std::format("Cannot create resource while the SDL definition permits it (is blueprint: {}, "
49 "allow create from class: {})", isBlueprint(), allowCreateFromClass()));
50
51 return nullptr;
52 }
53}

104template<typename Owner, typename FieldSet>
105inline void TSdlOwnerClass<Owner, FieldSet>::call(
106 const std::string_view funcName,
107 ISdlResource* const resource,
108 SdlInputClauses& clauses,
109 const SdlInputContext& ctx) const
110{
111 // Find SDL function by name
112 const SdlFunction* func = nullptr;
113 for(std::size_t funcIdx = 0; funcIdx < m_functions.size(); ++funcIdx)
114 {
115 PH_ASSERT(m_functions[funcIdx]);
116 const SdlFunction& funcI = *(m_functions[funcIdx]);
117 if(funcI.getName() == funcName)
118 {
119 func = &funcI;
120 break;
121 }
122 }
123
124 // If found the function in this class, call it
125 if(func)
126 {
127 func->call(resource, clauses, ctx);
128 }
129 // If not found in this class, call on base class instead
130 else if(getBase())
131 {
132 getBase()->call(funcName, resource, clauses, ctx);
133 }
134 else
135 {
136 throw SdlLoadError(
137 "cannot find SDL function named <" + std::string(funcName) + "> in "
138 "SDL class <" + genPrettyName() + ">");
139 }
140}

142template<typename Owner, typename FieldSet>
143inline void TSdlOwnerClass<Owner, FieldSet>::referencedResources(
144 const ISdlResource* const targetResource,
145 std::vector<const ISdlResource*>& out_resources) const
146{
147 static_assert(std::is_base_of_v<ISdlResource, Owner>,
148 "Owner class must derive from ISdlResource.");
149
150 const Owner& owner = *(castTo<const Owner>(targetResource));
151 for(std::size_t fieldIdx = 0; fieldIdx < m_fields.numFields(); ++fieldIdx)
152 {
153 m_fields[fieldIdx].ownedResources(owner, out_resources);
154 }
155
156 // Find more references in base class
157 if(isDerived())
158 {
159 PH_ASSERT(getBase());
160 getBase()->referencedResources(targetResource, out_resources);
161 }
162}

247template<typename Owner, typename FieldSet>
248inline void TSdlOwnerClass<Owner, FieldSet>::loadFieldsFromSdl(
249 Owner& owner,
250 SdlInputClauses& clauses,
251 const SdlInputContext& ctx) const
252{
253 auto noticeReceiver = [](std::string noticeMsg, EFieldImportance importance)
254 {
255 if(importance == EFieldImportance::Optional || importance == EFieldImportance::NiceToHave)
256 {
257 PH_LOG_STRING(SdlClass, noticeMsg);
258 }
259 else
260 {
261 PH_LOG_WARNING_STRING(SdlClass, noticeMsg);
262 }
263 };
264
265 // If this class is the source class, it should be the most derived class of the current
266 // input process and consume all remaining clauses. Otherwise, redundant clauses are
267 // allowed as subsequent (derived) classes would consume them.
268 //
269 if(ctx.getSrcClass() == this)
270 {
271 field_set_op::load_fields_from_sdl(
272 owner,
273 m_fields,
274 clauses,
275 ctx,
276 std::move(noticeReceiver));
277 }
278 else
279 {
280 field_set_op::load_fields_from_sdl_with_redundant_clauses(
281 owner,
282 m_fields,
283 clauses,
284 ctx,
285 std::move(noticeReceiver));
286 }
287}
