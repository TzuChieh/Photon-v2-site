145template<typename T>
146inline T log2_floor(const T value)
147{
148 if constexpr(std::is_floating_point_v<T>)
149 {
150 PH_ASSERT_GT(value, T(0));
151
152 return static_cast<T>(std::floor(std::log2(value)));
153 }
154 else
155 {
156 constexpr T NUM_BITS = sizeof(T) * CHAR_BIT;
157
158 static_assert(std::is_integral_v<T>);
159
160 PH_ASSERT_GT(value, T(0));
161
162#if PH_COMPILER_IS_MSVC
163
164 unsigned long first1BitFromLeftIndex = std::numeric_limits<unsigned long>::max();
165 if constexpr(sizeof(T) <= sizeof(unsigned long))
166 {
167 const auto isIndexSet = _BitScanReverse(
168 &first1BitFromLeftIndex,
169 static_cast<unsigned long>(value));
170 PH_ASSERT_GT(isIndexSet, 0);
171 }
172 else
173 {
174 static_assert(sizeof(T) <= sizeof(unsigned __int64),
175 "size of T is too large");
176
177 const auto isIndexSet = _BitScanReverse64(
178 &first1BitFromLeftIndex,
179 static_cast<unsigned __int64>(value));
180 PH_ASSERT_GT(isIndexSet, 0);
181 }
182 PH_ASSERT_IN_RANGE_INCLUSIVE(first1BitFromLeftIndex, T(0), NUM_BITS - 1);
183
184 return static_cast<T>(first1BitFromLeftIndex);
185
186#elif PH_COMPILER_IS_CLANG || PH_COMPILER_IS_GCC
187
188 if constexpr(sizeof(T) <= sizeof(unsigned int))
189 {
190 const int numLeftZeros = __builtin_clz(static_cast<unsigned int>(value));
191 return static_cast<T>(sizeof(unsigned int) * CHAR_BIT - 1 - numLeftZeros);
192 }
193 else if constexpr(sizeof(T) <= sizeof(unsigned long))
194 {
195 const int numLeftZeros = __builtin_clzl(static_cast<unsigned long>(value));
196 return static_cast<T>(sizeof(unsigned long) * CHAR_BIT - 1 - numLeftZeros);
197 }
198 else
199 {
200 static_assert(sizeof(T) <= sizeof(unsigned long long),
201 "size of T is too large");
202
203 const int numLeftZeros = __builtin_clzll(static_cast<unsigned long long>(value));
204 return static_cast<T>(sizeof(unsigned long long) * CHAR_BIT - 1 - numLeftZeros);
205 }
206
207#endif
208 }

228template<typename T, std::enable_if_t<std::is_integral_v<T>, int> = 0>
229inline T wrap(T value, const T lowerBound, const T upperBound)
230{
231 PH_ASSERT_GE(upperBound, lowerBound);
232
233 const T rangeSize = upperBound - lowerBound + static_cast<T>(1);
234
235 if(value < lowerBound)
236 {
237 value += rangeSize * ((lowerBound - value) / rangeSize + static_cast<T>(1));
238
239 // Possibly fail if <value> overflow
240 PH_ASSERT_GE(value, lowerBound);
241 }
242
243 return lowerBound + (value - lowerBound) % rangeSize;
244}

349inline float fast_rcp_sqrt(float x)
350{
351 // TODO:
352 // 1. implement double version in the reference paper and templatize iteration step
353 // 2. is more iteration steps better than doing it in double
354 // 3. establish a standard, e.g., fast_<X>() is guaranteed to have max. rel. error < 1%
355
356 PH_ASSERT_GT(x, 0.0f);
357
358 const float halvedInput = 0.5f * x;
359
360 // Gives initial guess for later refinements
361 auto bits = bitwise_cast<std::uint32_t>(x);
362 bits = 0x5F375A86 - (bits >> 1);
363 x = bitwise_cast<float>(bits);
364
365 // Newton's method, each iteration increases accuracy.
366
367 // Iteration 1, max. relative error < 0.175125%
368 x = x * (1.5f - halvedInput * x * x);
369
370 // Iteration 2, disabled since <x> is already good enough
371 //x = x * (1.5f - halvedInput * x * x);
372
373 return x;
374}

389template<typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
390inline T reverse_bits(const T value)
391{
392 using namespace detail;
393
394 static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>,
395 "Unsupported type detected");
396
397 constexpr std::size_t NUM_BITS = sizeof(T) * CHAR_BIT;
398
399 // Note that arbitrary number of bits can be supported by careful handling of bitwise operations
400 static_assert(NUM_BITS % 8 == 0,
401 "Non-multiple-of-8 integer bits are not supported");
402
403 if constexpr(NUM_BITS == 8)
404 {
405 return T(detail::BITS8_REVERSE_TABLE[value]);
406 }
407 else if constexpr(NUM_BITS == 16)
408 {
409 return (T(detail::BITS8_REVERSE_TABLE[value & 0xFF]) << 8) |
410 (T(detail::BITS8_REVERSE_TABLE[(value >> 8) & 0xFF]));
411 }
412 else if constexpr(NUM_BITS == 32)
413 {
414 return (T(detail::BITS8_REVERSE_TABLE[value & 0xFF]) << 24) |
415 (T(detail::BITS8_REVERSE_TABLE[(value >> 8) & 0xFF]) << 16) |
416 (T(detail::BITS8_REVERSE_TABLE[(value >> 16) & 0xFF]) << 8) |
417 (T(detail::BITS8_REVERSE_TABLE[(value >> 24) & 0xFF]));
418 }
419 else
420 {
421 static_assert(NUM_BITS == 64);
422
423 return (T(detail::BITS8_REVERSE_TABLE[value & 0xFF]) << 56) |
424 (T(detail::BITS8_REVERSE_TABLE[(value >> 8) & 0xFF]) << 48) |
425 (T(detail::BITS8_REVERSE_TABLE[(value >> 16) & 0xFF]) << 40) |
426 (T(detail::BITS8_REVERSE_TABLE[(value >> 24) & 0xFF]) << 32) |
427 (T(detail::BITS8_REVERSE_TABLE[(value >> 32) & 0xFF]) << 24) |
428 (T(detail::BITS8_REVERSE_TABLE[(value >> 40) & 0xFF]) << 16) |
429 (T(detail::BITS8_REVERSE_TABLE[(value >> 48) & 0xFF]) << 8) |
430 (T(detail::BITS8_REVERSE_TABLE[(value >> 56) & 0xFF]));
431 }
432}

438template<std::unsigned_integral UIntType, std::integral RangeType>
439inline UIntType set_bits_in_range(const UIntType bits, const RangeType beginBitIdx, const RangeType endBitIdx)
440{
441 // Inclusive as empty range is allowed, e.g., `beginBitIdx` == `endBitIdx`
442 PH_ASSERT_IN_RANGE_INCLUSIVE(beginBitIdx, 0, endBitIdx);
443 PH_ASSERT_IN_RANGE_INCLUSIVE(endBitIdx, beginBitIdx, sizeof_in_bits<UIntType>());
444
445 // Mask for the bits in range. Constructed by producing required number of 1's then shift
446 // by <beginBitIdx>.
447 // Note that if there is integer promotion to signed types, it should be fine--promoted type
448 // must be wider than the original unsigned type, the bit shifts will never reach sign bit.
449 //
450 const auto bitMask = static_cast<UIntType>(((UIntType(1) << (endBitIdx - beginBitIdx)) - 1) << beginBitIdx);
451
452 return bits | bitMask;
453}

488template<typename T, T MIN, T MAX, std::size_t N>
489inline std::array<T, N> evenly_spaced_array()
490{
491 static_assert(MAX > MIN);
492 static_assert(N >= 2);
493
494 constexpr T TOTAL_SIZE = MAX - MIN;
495
496 std::array<T, N> arr;
497 arr[0] = MIN;
498 for(std::size_t i = 1; i < N - 1; ++i)
499 {
500 // Calculate current fraction i/(N-1), and multiply it with the total size
501 // (we multiply total size with i fist, otherwise integral types will always result in 0)
502 arr[i] = MIN + static_cast<T>(i * TOTAL_SIZE / (N - 1));
503
504 PH_ASSERT_IN_RANGE_INCLUSIVE(arr[i], MIN, MAX);
505 }
506 arr[N - 1] = MAX;
507
508 return arr;
509}

544inline uint16 fp32_to_fp16_bits(const float32 value)
545{
546 static_assert(std::numeric_limits<float32>::is_iec559);
547
548 if(std::abs(value) < 0.0000610352f)
549 {
550 return std::signbit(value) ? 0x8000 : 0x0000;
551 }
552
553 uint16 fp16Bits;
554 uint32 fp32Bits;
555
556 fp32Bits = std::bit_cast<uint32>(value);
557
558 // truncate the mantissa and doing (exp - 127 + 15)
559 fp16Bits = static_cast<uint16>((((fp32Bits & 0x7FFFFFFF) >> 13) - (0x38000000 >> 13)));
560 fp16Bits |= ((fp32Bits & 0x80000000) >> 16);// sign
561
562 return fp16Bits;
563}

567inline float32 fp16_bits_to_fp32(const uint16 fp16Bits)
568{
569 static_assert(std::numeric_limits<float32>::is_iec559);
570
571 // 0 if all bits other than the sign bit are 0
572 if((fp16Bits & 0x7FFF) == 0)
573 {
574 return (fp16Bits & 0x8000) == 0 ? 0.0f : -0.0f;
575 }
576
577 uint32 fp32Bits = ((fp16Bits & 0x8000) << 16);
578 fp32Bits |= (((fp16Bits & 0x7FFF) << 13) + 0x38000000);
579
580 return std::bit_cast<float32>(fp32Bits);
581}

633inline void uint64_mul(const uint64 lhs, const uint64 rhs, uint64& out_high64, uint64& out_low64)
634{
635#if defined(__SIZEOF_INT128__)
636 const auto result128 = __uint128_t(lhs) + __uint128_t(rhs);
637 out_high64 = static_cast<uint64>(result128 >> 64);
638 out_low64 = static_cast<uint64>(result128);
639#elif PH_COMPILER_IS_MSVC
640 out_low64 = _umul128(lhs, rhs, &out_high64);
641#else
642 /*
643 Divide each 64-bit number into two 32-bit parts, then multiplying (or adding) any 32-bit
644 pairs cannot overflow if done using 64-bit arithmetic. If we multiply them using base 2^32
645 arithmetic, a total of 4 products must be shifted and added to obtain the final result.
646 Here is a simple illustration of the algorithm:
647
648 lhs = [ a ][ b ]
649 rhs = [ c ][ d ]
650
651 Treat this like performing a regular multiplication on paper. Note that the multiplication
652 is done in base 2^32:
653
654 [ a ][ b ]
655 x [ c ][ d ]
656 -------------------
657 [a*d][b*d]
658 + [a*c][b*c]
659 -------------------
660
661 The overflowing part of each partial product is the carry (32-bit carry). This is a good reference:
662 https://stackoverflow.com/questions/26852435/reasonably-portable-way-to-get-top-64-bits-from-64x64-bit-multiply
663 */
664
665 const uint64 a = lhs >> 32, b = lhs & 0xFFFFFFFF;
666 const uint64 c = rhs >> 32, d = rhs & 0xFFFFFFFF;
667
668 const uint64 ac = a * c;
669 const uint64 bc = b * c;
670 const uint64 ad = a * d;
671 const uint64 bd = b * d;
672
673 // This is the middle part of the above illustration, which is a sum of three 32-bit numbers
674 // (so it is 34-bit at most): two products b*c and a*d, and the carry from b*d
675 const uint64 mid34 = (bd >> 32) + (bc & 0xFFFFFFFF) + (ad & 0xFFFFFFFF);
676
677 out_high64 = ac + (bc >> 32) + (ad >> 32) + (mid34 >> 32);
678 out_low64 = (mid34 << 32) | (bd & 0xFFFFFFFF);
679#endif
680}
