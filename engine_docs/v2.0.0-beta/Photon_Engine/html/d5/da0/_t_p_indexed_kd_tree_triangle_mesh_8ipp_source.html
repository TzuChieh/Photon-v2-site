51template<typename Index>
52inline bool TPIndexedKdTreeTriangleMesh<Index>::isIntersecting(const Ray& ray, HitProbe& probe) const
53{
54 real closestHitT = std::numeric_limits<real>::max();
55 detail::TClosestHitProbeResult<Index> closestHit;
56
57 const bool hasHit = m_kdTree.nearestTraversal(
58 ray.getSegment(),
59 [ray, &closestHitT, &closestHit](
60 const Triangle& triangle,
61 const math::TLineSegment<real>& segment,
62 const std::size_t triangleIndex)
63 -> std::optional<real>
64 {
65 real hitT;
66 math::Vector3R hitBary;
67 if(triangle.isIntersecting(segment, &hitT, &hitBary))
68 {
69 if(hitT < closestHitT)
70 {
71 closestHitT = hitT;
72 closestHit.bary = hitBary;
73 closestHit.faceIndex = static_cast<Index>(triangleIndex);
74 }
75
76 return hitT;
77 }
78 else
79 {
80 return {};
81 }
82 });
83
84 if(hasHit)
85 {
86 probe.pushBaseHit(this, closestHitT);
87 probe.pushCache(closestHit);
88
89 return true;
90 }
91 else
92 {
93 return false;
94 }
95}

97template<typename Index>
98inline void TPIndexedKdTreeTriangleMesh<Index>::calcIntersectionDetail(
99 const Ray& ray,
100 HitProbe& probe,
101 HitDetail* const out_detail) const
102{
103 PH_ASSERT(out_detail);
104 PH_ASSERT(m_triangleBuffer);
105
106 const auto closestHit = probe.popCache<detail::TClosestHitProbeResult<Index>>();
107 PH_ASSERT_MSG(!closestHit.bary.isZero() && closestHit.bary.isFinite(),
108 closestHit.bary.toString());
109
110 const auto& positions = m_triangleBuffer->getPositions(closestHit.faceIndex);
111 const auto& texCoords = m_triangleBuffer->getTexCoords(closestHit.faceIndex);
112 const auto& normals = m_triangleBuffer->getNormals(closestHit.faceIndex);
113
114 const Triangle triangle(positions);
115 const math::Vector3R position = triangle.barycentricToSurface(closestHit.bary);
116 const math::Vector3R faceNormal = triangle.getFaceNormal();
117
118 const bool hasTexCoords = m_triangleBuffer->hasTexCoord();
119 const bool hasShadingNormal = m_triangleBuffer->hasNormal();
120
121 const auto shadingNormal = hasShadingNormal
122 ? Triangle::interpolate(normals, closestHit.bary).normalize()
123 : faceNormal;
124
125 const auto uvw = hasTexCoords
126 ? Triangle::interpolate(texCoords, closestHit.bary)
127 : math::Vector3R(0);
128
129 PH_ASSERT_MSG(position.isFinite() && shadingNormal.isFinite() && uvw.isFinite(), "\n"
130 "position = " + position.toString() + "\n"
131 "shading-normal = " + shadingNormal.toString() + "\n"
132 "uvw = " + uvw.toString() + "\n");
133
134 // TODO: respect primitive channel
135 // (if it's default channel, use vertex uvw; otherwise, use mapper)
136
137 math::Vector3R dPdU(0.0_r), dPdV(0.0_r);
138 math::Vector3R dNdU(0.0_r), dNdV(0.0_r);
139 if(hasTexCoords)
140 {
141 const std::array<math::Vector2R, 3> uvs = {
142 math::Vector2R(texCoords[0].x(), texCoords[0].y()),
143 math::Vector2R(texCoords[1].x(), texCoords[1].y()),
144 math::Vector2R(texCoords[2].x(), texCoords[2].y())};
145
146 if(!Triangle::calcSurfaceParamDerivatives(
147 positions, uvs,
148 &dPdU, &dPdV))
149 {
150 dPdU.set(0); dPdV.set(0);
151 }
152
153 if(hasShadingNormal && !Triangle::calcSurfaceParamDerivatives(
154 normals, uvs,
155 &dNdU, &dNdV))
156 {
157 dNdU.set(0); dNdV.set(0);
158 }
159 }
160
161 PH_ASSERT_MSG(dPdU.isFinite() && dPdV.isFinite() &&
162 dNdU.isFinite() && dNdV.isFinite(), "\n"
163 "dPdU = " + dPdU.toString() + ", dPdV = " + dPdV.toString() + "\n"
164 "dNdU = " + dNdU.toString() + ", dNdV = " + dNdV.toString() + "\n");
165
166 out_detail->setHitIntrinsics(this, uvw, probe.getHitRayT());
167 out_detail->getHitInfo(ECoordSys::LOCAL).setAttributes(
168 position,
169 faceNormal,
170 shadingNormal);
171 out_detail->getHitInfo(ECoordSys::LOCAL).setDerivatives(
172 dPdU, dPdV, dNdU, dNdV);
173 out_detail->getHitInfo(ECoordSys::WORLD) = out_detail->getHitInfo(ECoordSys::LOCAL);
174}
