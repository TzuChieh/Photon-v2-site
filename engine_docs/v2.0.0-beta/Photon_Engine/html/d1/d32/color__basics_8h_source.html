73template<typename ImplType>
74concept CColorValuesInterface = requires (
75 ImplType instance,
76 const ImplType constInstance)
77{
78 // getColorSpace() must be usable as a non-type template argument
79 detail::TColorSpaceDummy<ImplType::getColorSpace()>();
80
81 // Test for the method void setColorValues(const TRawColorValues<T, N>&).
82 // Note: we do not want to pass in the template parameters (for convenience), so we test by calling
83 // the method with an empty braced-init-list.

85

94template<typename ColorValuesType>
95concept CRawColorValues =
96 // We would want to have something like `std::is_std_array`, the following is good enough though
97 // (covers most of the use cases)
98 !CColorValuesInterface<ColorValuesType> &&
99 CSubscriptable<ColorValuesType> &&
100 std::is_aggregate_v<ColorValuesType> &&
101 requires (ColorValuesType colorValues)
102 {
103 colorValues.fill(0);
104 colorValues.begin();
105 colorValues.end();
106 colorValues.cbegin();
107 colorValues.cend();
108 { colorValues.size() } -> std::convertible_to<std::size_t>;
109 };

182template<typename ColorValuesType>
183inline constexpr auto get_any_element_from_color_values(const ColorValuesType& colorValues)
184{
185 if constexpr(CRawColorValues<ColorValuesType>)
186 {
187 return colorValues[0];
188 }
189 else if constexpr(CColorValuesInterface<ColorValuesType>)
190 {
191 return colorValues.getColorValues()[0];
192 }

195 /* Note: According to [expr.type.conv]/2, quote:
196 "if the type is cv void and the initializer is () or {} (after pack expansion, if any),
197 the expression is a prvalue of the specified type that performs no initialization."

199 return void{};

218/*
219For reference whites, set Y value to 1 (the brightest white that a color display supports).
220*/
221template<typename T = ColorValue>
222inline TTristimulusValues<T> CIEXYZ_of_custom_reference_white(const TChromaticityValues<T>& refWhite)
223{
224 // Reference whites should not have any component being <= 0
225 PH_ASSERT_GT(refWhite[0], static_cast<T>(0));
226 PH_ASSERT_GT(refWhite[1], static_cast<T>(0));
227
228 const T x = refWhite[0];
229 const T y = refWhite[1];
230
231 // Y is set to 1
232 const T Y_over_y = static_cast<T>(1) / y;
233 return
234 {
235 Y_over_y * x,
236 1.0,
237 Y_over_y * (static_cast<T>(1) - x - y)
238 };
239}

241template<typename T = ColorValue>
242inline TTristimulusValues<T> CIEXYZ_of(const EReferenceWhite refWhite)
243{
244 const auto chromaticity = chromaticity_of<T>(refWhite);
