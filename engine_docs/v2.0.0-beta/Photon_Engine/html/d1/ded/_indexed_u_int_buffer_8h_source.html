120template<std::integral IntegerType>
121inline void IndexedUIntBuffer::setUInt(const std::size_t index, const IntegerType value)
122{
123 PH_ASSERT(isAllocated());
124
125 const auto uint64Value = lossless_integer_cast<uint64>(value);
126 if(uint64Value > getMaxAllowedValue())
127 {
128 throw std::invalid_argument(std::format(
129 "Integer value {} cannot be stored using {} bits.", value, m_numBitsPerUInt));
130 }
131
132 const std::size_t firstByteIndex = index * m_numBitsPerUInt / CHAR_BIT;
133 const std::size_t firstByteBitOffset = index * m_numBitsPerUInt - firstByteIndex * CHAR_BIT;
134 const std::size_t numStraddledBytes = (firstByteBitOffset + m_numBitsPerUInt + (CHAR_BIT - 1)) / CHAR_BIT;
135
136 PH_ASSERT_LT(firstByteBitOffset, CHAR_BIT);
137 PH_ASSERT_LE(numStraddledBytes, 8 + 1);
138 PH_ASSERT_LE(firstByteIndex + numStraddledBytes, m_byteBufferSize);
139
140 // Potentially read straddled previous & next values' bits
141 uint64 rawBits = 0;
142 std::memcpy(&rawBits, &m_byteBuffer[firstByteIndex], std::min<std::size_t>(numStraddledBytes, 8));
143
144 // Store value between the back of previous value and the head of next value
145 rawBits = math::clear_bits_in_range(rawBits, firstByteBitOffset, firstByteBitOffset + m_numBitsPerUInt);
146 rawBits |= (uint64Value << firstByteBitOffset);
147 std::memcpy(&m_byteBuffer[firstByteIndex], &rawBits, std::min<std::size_t>(numStraddledBytes, 8));
148
149 // Handle situations where the value needs the 9-th byte (straddles next byte), this can happen
150 // since we support any number of bits per index
151 if(numStraddledBytes > 8)
152 {
153 uint8 remainingRawBits;
154 std::memcpy(&remainingRawBits, &m_byteBuffer[firstByteIndex + 8], 1);
155
156 // Store the remaining value before next value
157 remainingRawBits = math::clear_bits_in_range(remainingRawBits, static_cast<std::size_t>(0), firstByteBitOffset + m_numBitsPerUInt - 64);
158 remainingRawBits |= static_cast<uint8>(uint64Value >> (64 - firstByteBitOffset));
159 std::memcpy(&m_byteBuffer[firstByteIndex + 8], &remainingRawBits, 1);
160 }
161}

179 // Resort to copy one by one

186 setUInt(uintIdx, values[valueIdx]);

188 else

190 setUInt(uintIdx, static_cast<uint64>(values[valueIdx]));

196inline uint64 IndexedUIntBuffer::getUInt(const std::size_t index) const
197{
198 PH_ASSERT(isAllocated());
199
200 const std::size_t firstByteIndex = index * m_numBitsPerUInt / CHAR_BIT;
201 const std::size_t firstByteBitOffset = index * m_numBitsPerUInt - firstByteIndex * CHAR_BIT;
202 const std::size_t numStraddledBytes = (firstByteBitOffset + m_numBitsPerUInt + (CHAR_BIT - 1)) / CHAR_BIT;
203
204 PH_ASSERT_LT(firstByteBitOffset, CHAR_BIT);
205 PH_ASSERT_LE(numStraddledBytes, 8 + 1);
206 PH_ASSERT_LE(firstByteIndex + numStraddledBytes, m_byteBufferSize);
207
208 // Read current value's bits (first 8 bytes, at most)
209 uint64 rawBits = 0;
210 std::memcpy(&rawBits, &m_byteBuffer[firstByteIndex], std::min<std::size_t>(numStraddledBytes, 8));
211
212 // Clear previous and next values' bits if any, then get the value
213 const auto bitMask = math::set_bits_in_range<uint64>(0, firstByteBitOffset, firstByteBitOffset + m_numBitsPerUInt);
214 uint64 value = (rawBits & bitMask) >> firstByteBitOffset;
215
216 // Handle situations where the value needs the 9-th byte (straddles next byte), this can happen
217 // since we support any number of bits per index
218 if(numStraddledBytes > 8)
219 {
220 uint8 remainingRawBits;
221 std::memcpy(&remainingRawBits, &m_byteBuffer[firstByteIndex + 8], 1);
222
223 // Extract the remaining value and clear next value's bits if any
224 const auto remainingBitsBitMask = math::set_bits_in_range<uint8>(0, static_cast<std::size_t>(0), firstByteBitOffset + m_numBitsPerUInt - 64);
225 remainingRawBits &= remainingBitsBitMask;
226
227 // Add the remaining bits to the value
228 value |= (static_cast<uint64>(remainingRawBits) << (64 - firstByteBitOffset));
229 }
230
231 return value;
232}
