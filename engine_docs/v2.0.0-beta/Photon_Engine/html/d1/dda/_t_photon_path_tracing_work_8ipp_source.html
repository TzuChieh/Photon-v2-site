52template<CPhoton Photon>
53inline void TPhotonPathTracingWork<Photon>::doWork()
54{
55 PH_PROFILE_SCOPE();
56
57 // FIXME: currently we exit immediately when photon buffer is full; we should trace a full path instead
58
59 Timer timer;
60 timer.start();
61
62 const BsdfQueryContext bsdfContext(ALL_SURFACE_ELEMENTALS, ETransport::Importance, lta::ESidednessPolicy::Strict);
63 const lta::SurfaceTracer surfaceTracer{m_scene};
64
65 const auto raySampleHandle = m_sampleGenerator->declareStageND(2, m_photonBuffer.size());
66 m_sampleGenerator->prepareSampleBatch();// HACK: check if succeeded
67 auto raySamples = m_sampleGenerator->getSamplesND(raySampleHandle);
68
69 m_numPhotonPaths = 0;
70 std::size_t numStoredPhotons = 0;
71 std::size_t photonStatsCounter = 0;
72 while(numStoredPhotons < m_photonBuffer.size())
73 {
74 ++m_numPhotonPaths;
75
76 SampleFlow sampleFlow = raySamples.readSampleAsFlow();
77
78 // TODO: properly sample time
79 EnergyEmissionSampleQuery energyEmission;
80 energyEmission.inputs.set(Time{});
81
82 SurfaceHit surfaceHit;
83 Ray tracingRay;
84 {
85 HitProbe probe;
86 m_scene->emitRay(energyEmission, sampleFlow, probe);
87 if(!energyEmission.outputs)
88 {
89 continue;
90 }
91
92 constexpr SurfaceHitReason reason(ESurfaceHitReason::SampledPosDir);
93 tracingRay = energyEmission.outputs.getEmittedRay();
94 surfaceHit = SurfaceHit(tracingRay, probe, reason);
95 }
96
97 PH_ASSERT_IN_RANGE(tracingRay.getDirection().lengthSquared(), 0.9_r, 1.1_r);
98
99 // Here 0-bounce lighting is never accounted for
100 math::Spectrum throughputRadiance(energyEmission.outputs.getEmittedEnergy());
101 throughputRadiance.divLocal(energyEmission.outputs.getPdfA());
102 throughputRadiance.divLocal(energyEmission.outputs.getPdfW());
103 throughputRadiance.mulLocal(surfaceHit.getShadingNormal().absDot(tracingRay.getDirection()));
104
105 // Start tracing single photon path with at least 1 bounce
106 uint32 photonPathLength = 0;
107 while(!throughputRadiance.isZero())
108 {
109 PH_ASSERT_LT(photonPathLength, m_maxPhotonPathLength);
110
111 if(!surfaceTracer.traceNextSurfaceFrom(
112 surfaceHit, tracingRay, bsdfContext.sidedness, &surfaceHit))
113 {
114 break;
115 }
116
117 ++photonPathLength;
118
119 // TODO: we can also skip storing this photon if the BSDF has little contribution
120 // (e.g., by measuring its integrated value)
121
122 const PrimitiveMetadata* metadata = surfaceHit.getDetail().getPrimitive()->getMetadata();
123 const SurfaceOptics* optics = metadata->getSurface().getOptics();
124
125 math::Spectrum weightedThroughputRadiance;
126 if(lta::RussianRoulette{}.surviveOnLuminance(throughputRadiance, sampleFlow, &weightedThroughputRadiance))
127 {
128 throughputRadiance = weightedThroughputRadiance;
129
130 if(photonPathLength >= m_minPhotonPathLength)
131 {
132 m_photonBuffer[numStoredPhotons++] = makePhoton(
133 surfaceHit, throughputRadiance, tracingRay, photonPathLength);
134 ++photonStatsCounter;
135 }
136
137 if(numStoredPhotons == m_photonBuffer.size() ||
138 photonPathLength == m_maxPhotonPathLength)
139 {
140 break;
141 }
142 }// end if photon survived
143 else
144 {
145 break;
146 }
147
148 BsdfSampleQuery bsdfSample(bsdfContext);
149 Ray sampledRay;
150 bsdfSample.inputs.set(surfaceHit, tracingRay.getDirection().mul(-1));
151 if(!surfaceTracer.doBsdfSample(bsdfSample, sampleFlow, &sampledRay))
152 {
153 break;
154 }
155
156 math::Vector3R V = tracingRay.getDirection().mul(-1);
157 math::Vector3R L = bsdfSample.outputs.getL();
158 math::Vector3R Ng = surfaceHit.getGeometryNormal();
159 math::Vector3R Ns = surfaceHit.getShadingNormal();
160 throughputRadiance.mulLocal(bsdfSample.outputs.getPdfAppliedBsdf());
161 throughputRadiance.mulLocal(lta::tamed_importance_scatter_Ns_corrector(Ns, Ng, L, V));
162 throughputRadiance.mulLocal(Ns.absDot(L));
163
164 tracingRay = sampledRay;
165 }// end single photon path
166
167 if(photonStatsCounter >= 16384)
168 {
169 timer.stop();
170 setElapsedMs(timer.getDeltaMs());
171
172 if(m_statistics)
173 {
174 m_statistics->addNumTracedPhotons(photonStatsCounter);
175 }
176
177 photonStatsCounter = 0;
178 }
179 }// end while photon buffer is not full
180
181 timer.stop();
182 setElapsedMs(timer.getDeltaMs());
183
184 if(m_statistics)
185 {
186 m_statistics->addNumTracedPhotons(photonStatsCounter);
187 }
188}

202template<CPhoton Photon>
203inline Photon TPhotonPathTracingWork<Photon>::makePhoton(
204 const SurfaceHit& surfaceHit,
205 const math::Spectrum& throughputRadiance,
206 const Ray& tracingRay,
207 const std::size_t pathLength)
208{
209 Photon photon;
210 if constexpr(Photon::template has<EPhotonData::Position>())
211 {
212 photon.template set<EPhotonData::Position>(surfaceHit.getPosition());
213 }
214 if constexpr(Photon::template has<EPhotonData::ThroughputRadiance>())
215 {
216 photon.template set<EPhotonData::ThroughputRadiance>(throughputRadiance);
217 }
218 if constexpr(Photon::template has<EPhotonData::FromDir>())
219 {
220 photon.template set<EPhotonData::FromDir>(tracingRay.getDirection().mul(-1));
221 }
222 if constexpr(Photon::template has<EPhotonData::GeometryNormal>())
223 {
224 photon.template set<EPhotonData::GeometryNormal>(surfaceHit.getGeometryNormal());
225 }
226 if constexpr(Photon::template has<EPhotonData::PathLength>())
227 {
228 photon.template set<EPhotonData::PathLength>(pathLength);
229 }
230
231 return photon;
232}
