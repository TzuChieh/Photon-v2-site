50inline void TPhotonMappingWork<Photon>::doWork()
51{
52 PH_PROFILE_SCOPE();
53
54 // FIXME: currently we exit immediately when photon buffer is full; we should trace a full path instead
55
56 Timer timer;
57 timer.start();
58
59 const BsdfQueryContext bsdfContext(ALL_ELEMENTALS, ETransport::IMPORTANCE, ESidednessPolicy::STRICT);
60 const SurfaceTracer surfaceTracer(m_scene);
61
62 const auto raySampleHandle = m_sampleGenerator->declareStageND(2, m_numPhotons);
63 m_sampleGenerator->prepareSampleBatch();// HACK: check if succeeded
64 auto raySamples = m_sampleGenerator->getSamplesND(raySampleHandle);
65
66 std::size_t numStoredPhotons = 0;
67 *m_numPhotonPaths = 0;
68 std::size_t photonCounter = 0;
69 while(numStoredPhotons < m_numPhotons)
70 {
71 ++(*m_numPhotonPaths);
72
73 SampleFlow sampleFlow = raySamples.readSampleAsFlow();
74
75 Ray tracingRay;
76 math::Spectrum emittedRadiance;
77 math::Vector3R emitN;
78 real pdfA;
79 real pdfW;
80 m_scene->emitRay(sampleFlow, &tracingRay, &emittedRadiance, &emitN, &pdfA, &pdfW);
81 if(pdfA * pdfW == 0.0_r)
82 {
83 continue;
84 }
85
86 // here 0-bounce lighting is not accounted for
87
88 math::Spectrum throughputRadiance(emittedRadiance);
89 throughputRadiance.divLocal(pdfA);
90 throughputRadiance.divLocal(pdfW);
91 throughputRadiance.mulLocal(emitN.absDot(tracingRay.getDirection()));
92
93 // start tracing single photon path
94 while(!throughputRadiance.isZero())
95 {
96 SurfaceHit surfaceHit;
97 if(!surfaceTracer.traceNextSurface(tracingRay, bsdfContext.sidedness, &surfaceHit))
98 {
99 break;
100 }
101
102 const PrimitiveMetadata* metadata = surfaceHit.getDetail().getPrimitive()->getMetadata();
103 const SurfaceOptics* optics = metadata->getSurface().getOptics();
104
105 math::Spectrum weightedThroughputRadiance;
106 if(RussianRoulette::surviveOnLuminance(throughputRadiance, sampleFlow, &weightedThroughputRadiance))
107 {
108 throughputRadiance = weightedThroughputRadiance;
109
110 Photon photon;
111
112 if constexpr(Photon::template has<EPhotonData::POSITION>()) {
113 photon.template set<EPhotonData::POSITION>(surfaceHit.getPosition());
114 }
115 if constexpr(Photon::template has<EPhotonData::THROUGHPUT_RADIANCE>()) {
116 photon.template set<EPhotonData::THROUGHPUT_RADIANCE>(throughputRadiance);
117 }
118 if constexpr(Photon::template has<EPhotonData::FROM_DIR>()) {
119 photon.template set<EPhotonData::FROM_DIR>(tracingRay.getDirection().mul(-1));
120 }
121
122 m_photonBuffer[numStoredPhotons++] = photon;
123 ++photonCounter;
124
125 if(numStoredPhotons == m_numPhotons)
126 {
127 break;
128 }
129 }// end if photon survived
130 else
131 {
132 break;
133 }
134
135 BsdfSampleQuery bsdfSample(bsdfContext);
136 Ray sampledRay;
137 bsdfSample.inputs.set(surfaceHit, tracingRay.getDirection().mul(-1));
138 if(!surfaceTracer.doBsdfSample(bsdfSample, sampleFlow, &sampledRay))
139 {
140 break;
141 }
142
143 math::Vector3R V = tracingRay.getDirection().mul(-1);
144 math::Vector3R L = bsdfSample.outputs.L;
145 math::Vector3R Ng = surfaceHit.getGeometryNormal();
146 math::Vector3R Ns = surfaceHit.getShadingNormal();
147 throughputRadiance.mulLocal(bsdfSample.outputs.pdfAppliedBsdf);
148 throughputRadiance.mulLocal(lta::importance_BSDF_Ns_corrector(Ns, Ng, L, V));
149 throughputRadiance.mulLocal(Ns.absDot(L));
150
151 tracingRay = sampledRay;
152 }// end single photon path
153
154 if(photonCounter >= 16384)
155 {
156 timer.stop();
157 setElapsedMs(timer.getDeltaMs());
158
159 if(m_statistics)
160 {
161 m_statistics->asyncAddNumTracedPhotons(photonCounter);
162 }
163
164 photonCounter = 0;
165 }
166 }// end while photon buffer is not full
167
168 timer.stop();
169 setElapsedMs(timer.getDeltaMs());
170
171 if(m_statistics)
172 {
173 m_statistics->asyncAddNumTracedPhotons(photonCounter);
174 }
175}
