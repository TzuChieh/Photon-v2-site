102template<typename Target, typename Source>
103inline Target bitwise_cast(const Source& source)
104{
105 static_assert(std::is_trivially_copyable_v<Source>);
106 static_assert(std::is_trivially_copyable_v<Target>);
107
108 static_assert(sizeof(Source) == sizeof(Target),
109 "Source and Target must have the same size");
110
111 if constexpr(std::is_same_v<Source, Target>)
112 {
113 return source;
114 }
115 else
116 {
117#if __cpp_lib_bit_cast
118 return std::bit_cast<Target>(source);
119#else
120 Target target;
121 std::memcpy(&target, &source, sizeof(Source));
122 return target;
123#endif
124 }
125}

244template<std::integral DstType, std::integral SrcType>
245inline DstType lossless_integer_cast(const SrcType src)
246{
247 using SrcLimits = std::numeric_limits<SrcType>;
248 using DstLimits = std::numeric_limits<DstType>;
249
250 // Note that the use of `std::cmp_<X>` functions are important as the comparisons
251 // may be signed <-> unsigned comparisons, which may cause signed limits to overflow
252
253 // TODO: we may need to cast src to some integer first to support char and bool types (they are not supported by cmp functions)
254
255 constexpr bool mayHavePositiveOverflow = std::cmp_greater(SrcLimits::max(), DstLimits::max());
256 constexpr bool mayHaveNegativeOverflow = std::cmp_less(SrcLimits::lowest(), DstLimits::lowest());
257
258 if constexpr(mayHavePositiveOverflow)
259 {
260 if(std::cmp_greater(src, DstLimits::max()))
261 {
262 throw_formatted<OverflowException>("cast results in positive overflow: {} exceeds the limit {}",
263 src, DstLimits::max());
264 }
265 }
266
267 if constexpr(mayHaveNegativeOverflow)
268 {
269 if(std::cmp_less(src, DstLimits::lowest()))
270 {
271 throw_formatted<OverflowException>("cast results in negative overflow: {} exceeds the limit {}",
272 src, DstLimits::lowest());
273 }
274 }
275
276 // All possible integer overflow scenarios are checked so it is safe to cast now
277 return static_cast<DstType>(src);
278}

280template<std::floating_point DstType, std::floating_point SrcType>
281inline DstType lossless_float_cast(const SrcType src)
282{
283 // Nothing to do if both types are the same
284 if constexpr(std::is_same_v<SrcType, DstType>)
285 {
286 return src;
287 }
288 // If we are converting to a wider floating-point type, generally it will be lossless
289 else if constexpr(sizeof(DstType) > sizeof(SrcType))
290 {
291 // We need both types to be IEEE-754
292 static_assert(std::numeric_limits<SrcType>::is_iec559);
293 static_assert(std::numeric_limits<DstType>::is_iec559);
294
295 return static_cast<DstType>(src);
296 }
297 // Otherwise, cast to `DstType` then back to `SrcType` and see if there is any difference
298 else
299 {
300 const auto dst = static_cast<DstType>(src);
301 const auto dstBackToSrc = static_cast<SrcType>(dst);
302 if(src != dstBackToSrc)
303 {
304 throw_formatted<NumericException>("cast results in numeric precision loss: {} -> {}",
305 src, dstBackToSrc);
306 }
307
308 return dst;
309 }
310}

317template<CNumber DstType, CNumber SrcType>
318inline DstType lossless_cast(const SrcType src)
319{
320 // Integer -> Integer
321 if constexpr(std::is_integral_v<SrcType> && std::is_integral_v<DstType>)
322 {
323 return lossless_integer_cast<DstType>(src);
324 }
325 // Integer -> Floating-point
326 else if constexpr(std::is_integral_v<SrcType> && std::is_floating_point_v<DstType>)
327 {
328 // TODO
329 PH_ASSERT_UNREACHABLE_SECTION();
330 return 0;
331 }
332 // Floating-point -> Integer
333 else if constexpr(std::is_floating_point_v<SrcType> && std::is_integral_v<DstType>)
334 {
335 // TODO
336 PH_ASSERT_UNREACHABLE_SECTION();
337 return 0;
338 }
339 // Floating-point -> Floating-point
340 else
341 {
342 static_assert(std::is_floating_point_v<SrcType> && std::is_floating_point_v<DstType>);
343
344 return lossless_float_cast<DstType>(src);
345 }
346}
