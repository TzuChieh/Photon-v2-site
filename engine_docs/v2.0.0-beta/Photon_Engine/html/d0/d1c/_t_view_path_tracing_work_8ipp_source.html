56template<CViewPathHandler Handler>
57inline void TViewPathTracingWork<Handler>::doWork()
58{
59 PH_PROFILE_SCOPE();
60 PH_ASSERT(m_handler);
61 PH_ASSERT_GT(m_sampleRes.product(), 0);
62
63 const auto rasterSampleHandle = m_sampleGenerator->declareStageND(
64 2,
65 m_sampleRes.product(),
66 {m_sampleRes.x(), m_sampleRes.y()});
67
68 const auto raySampleHandle = m_sampleGenerator->declareStageND(2, m_sampleRes.product());
69
70 while(m_sampleGenerator->prepareSampleBatch())
71 {
72 const auto rasterSamples = m_sampleGenerator->getSamplesND(rasterSampleHandle);
73 auto raySamples = m_sampleGenerator->getSamplesND(raySampleHandle);
74 for(std::size_t i = 0; i < rasterSamples.numSamples(); ++i)
75 {
76 // OPT: could be tiled; this has bad cache behavior (sample may jump around too far)
77
78 const auto rasterSample = math::sample_cast<float64>(rasterSamples.template get<2>(i));
79 const math::Vector2D rasterCoord = m_rasterSampleWindowPx.sampleToSurface(rasterSample);
80 const math::Vector2S sampleIndex(math::Vector2D(
81 math::clamp(m_sampleRes.x() * rasterSample[0], 0.0, m_sampleRes.x() - 1.0),
82 math::clamp(m_sampleRes.y() * rasterSample[1], 0.0, m_sampleRes.y() - 1.0)));
83
84 PH_ASSERT_LT(sampleIndex.x(), m_sampleRes.x());
85 PH_ASSERT_LT(sampleIndex.y(), m_sampleRes.y());
86
87 Ray tracingRay;
88 const auto quantityWeight = m_receiver->receiveRay(rasterCoord, &tracingRay);
89 tracingRay.reverse();
90
91 math::Spectrum pathThroughput(quantityWeight);
92 if(!m_handler->onReceiverSampleStart(rasterCoord, sampleIndex, pathThroughput))
93 {
94 m_handler->onReceiverSampleEnd();
95 continue;
96 }
97
98 SampleFlow sampleFlow = raySamples.readSampleAsFlow();
99
100 traceViewPath(
101 SurfaceHit{},
102 tracingRay,
103 pathThroughput,
104 0,
105 sampleFlow);
106
107 m_handler->onReceiverSampleEnd();
108 }// end for single sample
109
110 m_handler->onSampleBatchFinished();
111 }// end while single sample batch
112}

114template<CViewPathHandler Handler>
115inline void TViewPathTracingWork<Handler>::traceViewPath(
116 SurfaceHit prevHit,
117 Ray tracingRay,
118 math::Spectrum pathThroughput,
119 std::size_t pathLength,
120 SampleFlow& sampleFlow)
121{
122 const lta::SurfaceTracer surfaceTracer{m_scene};
123
124 while(true)
125 {
126 SurfaceHit X;
127 if(pathLength == 0)
128 {
129 if(!surfaceTracer.traceNextSurface(tracingRay, BsdfQueryContext{}.sidedness, &X))
130 {
131 break;
132 }
133 }
134 else
135 {
136 if(!surfaceTracer.traceNextSurfaceFrom(
137 prevHit, tracingRay, BsdfQueryContext{}.sidedness, &X))
138 {
139 break;
140 }
141 }
142
143 ++pathLength;
144 const ViewPathTracingPolicy& policy = m_handler->onPathHitSurface(pathLength, X, pathThroughput);
145 if(policy.isKilled())
146 {
147 break;
148 }
149
150 PH_ASSERT_IN_RANGE(tracingRay.getDirection().lengthSquared(), 0.9_r, 1.1_r);
151
152 const math::Vector3R V = -tracingRay.getDirection();
153 const math::Vector3R N = X.getShadingNormal();
154
155 if(policy.getSampleMode() == EViewPathSampleMode::SinglePath)
156 {
157 BsdfSampleQuery bsdfSample(BsdfQueryContext(policy.getTargetElemental(), ETransport::Radiance, lta::ESidednessPolicy::Strict));
158 bsdfSample.inputs.set(X, V);
159
160 Ray sampledRay;
161 if(!surfaceTracer.doBsdfSample(bsdfSample, sampleFlow, &sampledRay))
162 {
163 break;
164 }
165
166 pathThroughput.mulLocal(bsdfSample.outputs.getPdfAppliedBsdf());
167 pathThroughput.mulLocal(N.absDot(bsdfSample.outputs.getL()));
168
169 if(policy.useRussianRoulette())
170 {
171 math::Spectrum weightedThroughput;
172 if(lta::RussianRoulette{}.surviveOnLuminance(pathThroughput, sampleFlow, &weightedThroughput))
173 {
174 pathThroughput = weightedThroughput;
175 }
176 else
177 {
178 break;
179 }
180 }
181
182 tracingRay = sampledRay;
183 }
184 else
185 {
186 traceElementallyBranchedPath(policy, X, V, N, pathThroughput, pathLength, sampleFlow);
187 break;
188 }
189
190 prevHit = X;
191 }// end while true
192}

194template<CViewPathHandler Handler>
195inline void TViewPathTracingWork<Handler>::traceElementallyBranchedPath(
196 const ViewPathTracingPolicy& policy,
197 const SurfaceHit& X,
198 const math::Vector3R& V,
199 const math::Vector3R& N,
200 const math::Spectrum& pathThroughput,
201 const std::size_t pathLength,
202 SampleFlow& sampleFlow)
203{
204 PH_ASSERT(policy.getSampleMode() == EViewPathSampleMode::ElementalBranch);
205
206 const lta::SurfaceTracer surfaceTracer{m_scene};
207
208 const PrimitiveMetadata* metadata = X.getDetail().getPrimitive()->getMetadata();
209 const SurfaceOptics* surfaceOptics = metadata->getSurface().getOptics();
210
211 const SurfacePhenomena targetPhenomena = policy.getTargetPhenomena();
212 for(SurfaceElemental i = 0; i < surfaceOptics->numElementals(); ++i)
213 {
214 if(targetPhenomena.hasNone({surfaceOptics->getPhenomenonOf(i)}))
215 {
216 continue;
217 }
218
219 BsdfSampleQuery sample(BsdfQueryContext(i, ETransport::Radiance, lta::ESidednessPolicy::Strict));
220 sample.inputs.set(X, V);
221
222 Ray sampledRay;
223 if(!surfaceTracer.doBsdfSample(sample, sampleFlow, &sampledRay))
224 {
225 continue;
226 }
227
228 math::Spectrum elementalPathThroughput(pathThroughput);
229 elementalPathThroughput.mulLocal(sample.outputs.getPdfAppliedBsdf());
230 elementalPathThroughput.mulLocal(N.absDot(sampledRay.getDirection()));
231
232 if(policy.useRussianRoulette())
233 {
234 math::Spectrum weightedThroughput;
235 if(lta::RussianRoulette{}.surviveOnLuminance(elementalPathThroughput, sampleFlow, &weightedThroughput))
236 {
237 elementalPathThroughput = weightedThroughput;
238 }
239 else
240 {
241 continue;
242 }
243 }
244
245 traceViewPath(
246 X,
247 sampledRay,
248 elementalPathThroughput,
249 pathLength,
250 sampleFlow);
251 }// end for each phenomenon
252}
