94template<typename ViewPathHandler>
95inline void TViewPathTracingWork<ViewPathHandler>::traceViewPath(
96 Ray tracingRay,
97 math::Spectrum pathThroughput,
98 std::size_t pathLength,
99 SampleFlow& sampleFlow)
100{
101 const SurfaceTracer surfaceTracer(m_scene);
102 while(true)
103 {
104 SurfaceHit surfaceHit;
105 if(!surfaceTracer.traceNextSurface(tracingRay, BsdfQueryContext().sidedness, &surfaceHit))
106 {
107 break;
108 }
109
110 ++pathLength;
111 const ViewPathTracingPolicy& policy = m_handler->onPathHitSurface(pathLength, surfaceHit, pathThroughput);
112 if(policy.isKilled())
113 {
114 break;
115 }
116
117 // FIXME: reversed then assigned again later seems to be dangerous, state is unclear
118 tracingRay.reverse();
119 const math::Vector3R V = tracingRay.getDirection();
120 const math::Vector3R N = surfaceHit.getShadingNormal();
121
122 if(policy.getSampleMode() == EViewPathSampleMode::SINGLE_PATH)
123 {
124 BsdfSampleQuery bsdfSample(BsdfQueryContext(policy.getTargetElemental(), ETransport::RADIANCE, ESidednessPolicy::STRICT));
125 bsdfSample.inputs.set(surfaceHit, V);
126 Ray sampledRay;
127 if(!surfaceTracer.doBsdfSample(bsdfSample, sampleFlow, &sampledRay))
128 {
129 break;
130 }
131
132 pathThroughput.mulLocal(bsdfSample.outputs.pdfAppliedBsdf);
133 pathThroughput.mulLocal(N.absDot(bsdfSample.outputs.L));
134
135 if(policy.useRussianRoulette())
136 {
137 math::Spectrum weightedThroughput;
138 if(RussianRoulette::surviveOnLuminance(pathThroughput, sampleFlow, &weightedThroughput))
139 {
140 pathThroughput = weightedThroughput;
141 }
142 else
143 {
144 break;
145 }
146 }
147
148 tracingRay = sampledRay;
149 }
150 else
151 {
152 traceElementallyBranchedPath(policy, V, N, surfaceHit, pathThroughput, pathLength, sampleFlow);
153 break;
154 }
155 }// end while true
156}
