146template<CViewpoint Viewpoint, CPhoton Photon>
147inline auto TSPPMRadianceEvaluator<Viewpoint, Photon>::impl_onPathHitSurface(
148 const std::size_t pathLength,
149 const SurfaceHit& surfaceHit,
150 const math::Spectrum& pathThroughput) -> ViewPathTracingPolicy
151{
152 const SurfaceOptics* optics = surfaceHit.getSurfaceOptics();
153 if(!optics)
154 {
155 return ViewPathTracingPolicy().kill();
156 }
157
158 if constexpr(Viewpoint::template has<EViewpointData::ViewRadiance>())
159 {
160 const auto unaccountedEnergy = estimate_certainly_lost_energy(
161 pathLength,
162 surfaceHit,
163 pathThroughput,
164 m_photonMap->getInfo(),
165 m_scene);
166 addViewRadiance(*m_viewpoint, unaccountedEnergy);
167 }
168
169 if(optics->getAllPhenomena().hasNone(DELTA_SURFACE_PHENOMENA) &&
170 optics->getAllPhenomena().hasAny(ESurfacePhenomenon::DiffuseReflection))
171 {
172 // For path length = N, we can construct light transport path lengths with photon map,
173 // all at once, for the range [N_min, N_max] =
174 // [`N + m_photonMap->minPathLength`, `N + m_photonMap->maxPathLength`].
175
176 if constexpr(Viewpoint::template has<EViewpointData::SurfaceHit>())
177 {
178 m_viewpoint->template set<EViewpointData::SurfaceHit>(surfaceHit);
179 }
180 if constexpr(Viewpoint::template has<EViewpointData::ViewThroughput>())
181 {
182 m_viewpoint->template set<EViewpointData::ViewThroughput>(pathThroughput);
183 }
184 if constexpr(Viewpoint::template has<EViewpointData::ViewDir>())
185 {
186 m_viewpoint->template set<EViewpointData::ViewDir>(surfaceHit.getIncidentRay().getDirection().mul(-1));
187 }
188
189 m_foundTargetHitPoint = true;
190
191 if constexpr(Viewpoint::template has<EViewpointData::ViewRadiance>())
192 {
193 const auto unaccountedEnergy = estimate_lost_energy_for_merging(
194 pathLength,
195 surfaceHit,
196 pathThroughput,
197 m_photonMap->getInfo(),
198 m_scene);
199 addViewRadiance(*m_viewpoint, unaccountedEnergy);
200 }
201
202 return ViewPathTracingPolicy().kill();
203 }
204 else
205 {
206 if constexpr(Viewpoint::template has<EViewpointData::ViewRadiance>())
207 {
208 const auto unaccountedEnergy = estimate_lost_energy_for_extending(
209 pathLength,
210 surfaceHit,
211 pathThroughput,
212 m_photonMap->getInfo(),
213 m_scene);
214 addViewRadiance(*m_viewpoint, unaccountedEnergy);
215 }
216
217 return ViewPathTracingPolicy().
218 traceSinglePathFor(ALL_SURFACE_ELEMENTALS).
219 useRussianRoulette(true);
220 }
221}

223template<CViewpoint Viewpoint, CPhoton Photon>
224inline void TSPPMRadianceEvaluator<Viewpoint, Photon>::impl_onReceiverSampleEnd()
225{
226 if(!m_foundTargetHitPoint)
227 {
228 return;
229 }
230
231 const lta::SurfaceTracer surfaceTracer{m_scene};
232
233 const SurfaceHit& surfaceHit = m_viewpoint->template get<EViewpointData::SurfaceHit>();
234 const math::Vector3R L = m_viewpoint->template get<EViewpointData::ViewDir>();
235 const math::Vector3R Ng = surfaceHit.getGeometryNormal();
236 const math::Vector3R Ns = surfaceHit.getShadingNormal();
237 const real R = m_viewpoint->template get<EViewpointData::Radius>();
238
239 m_photonCache.clear();
240 m_photonMap->find(surfaceHit.getPosition(), R, m_photonCache);
241
242 // FIXME: as a parameter
243 const real alpha = 2.0_r / 3.0_r;
244
245 const real N = m_viewpoint->template get<EViewpointData::NumPhotons>();
246 const real M = static_cast<real>(m_photonCache.size());
247 const real newN = N + alpha * M;
248 const real newR = (N + M) != 0.0_r ? R * std::sqrt(newN / (N + M)) : R;
249
250 const BsdfQueryContext bsdfContext(ALL_SURFACE_ELEMENTALS, ETransport::Importance, lta::ESidednessPolicy::Strict);
251
252 math::Spectrum tauM(0);
253 BsdfEvalQuery bsdfEval(bsdfContext);
254 for(const auto& photon : m_photonCache)
255 {
256 const math::Vector3R V = photon.template get<EPhotonData::FromDir>();
257 if(!accept_photon_by_surface_topology(photon, Ng, Ns, L, V, bsdfContext.sidedness))
258 {
259 continue;
260 }
261
262 bsdfEval.inputs.set(surfaceHit, L, V);
263 if(!surfaceTracer.doBsdfEvaluation(bsdfEval))
264 {
265 continue;
266 }
267
268 math::Spectrum tau = photon.template get<EPhotonData::ThroughputRadiance>();
269 tau.mulLocal(bsdfEval.outputs.getBsdf());
270 tau.mulLocal(lta::tamed_importance_BSDF_Ns_corrector(Ns, Ng, V));
271
272 tauM.addLocal(tau);
273 }
274 tauM.mulLocal(m_viewpoint->template get<EViewpointData::ViewThroughput>());
275
276 const math::Spectrum tauN = m_viewpoint->template get<EViewpointData::Tau>();
277 const math::Spectrum newTau = (N + M) != 0.0_r ? (tauN + tauM) * (newN / (N + M)) : math::Spectrum(0);
278
279 m_viewpoint->template set<EViewpointData::Radius>(newR);
280 m_viewpoint->template set<EViewpointData::NumPhotons>(newN);
281 m_viewpoint->template set<EViewpointData::Tau>(newTau);
282}

284template<CViewpoint Viewpoint, CPhoton Photon>
285inline void TSPPMRadianceEvaluator<Viewpoint, Photon>::impl_onSampleBatchFinished()
286{
287 PH_ASSERT_GT(m_statisticsRes.product(), 0);
288
289 const math::TAABB2D<float64> statisticsRegion(m_statisticsRegion);
290 const auto rcpStatisticsRes = math::Vector2D(m_statisticsRes).rcp();
291
292 // Evaluate radiance using current iteration's data. Each SPPM statistics gathered is for
293 // estimating average radiance for a small sub-pixel region. We treat each estimated value
294 // as if it is centered on the sub-pixel region it represents.
295 for(int64 y = 0; y < m_statisticsRes.y(); ++y)
296 {
297 for(int64 x = 0; x < m_statisticsRes.x(); ++x)
298 {
299 const auto& viewpoint = m_viewpoints[getViewpointIndex(x, y)];
300 const auto& statisticsRasterCoord = statisticsRegion.xy01ToSurface(
301 (math::Vector2D(math::TVector2<int64>{x, y}) + 0.5) * rcpStatisticsRes);
302
303 // For most accurate result, we should average the viewpoints' radiance by ourselves and
304 // use `setPixel()` to store the value directly. Using `addSample()` will only be correct
305 // if a box filter is used. Nevertheless, the worst result is a slightly overblurred image.
306 // If higher number of viewpoints/regions per pixel were used, it will still approach a
307 // per-sample filtered PT result in the limit (infinite viewpoints per pixel) except on
308 // edges as we are not storing out-of-raster-bound statistics.
309 m_film->addSample(
310 statisticsRasterCoord.x(),
311 statisticsRasterCoord.y(),
312 estimateRadiance(viewpoint));
313 }
314 }

317template<CViewpoint Viewpoint, CPhoton Photon>
318inline math::Spectrum TSPPMRadianceEvaluator<Viewpoint, Photon>::estimateRadiance(
319 const Viewpoint& viewpoint) const
320{
321 const real radius = viewpoint.template get<EViewpointData::Radius>();
322 const real kernelArea = radius * radius * math::constant::pi<real>;
323 const real radianceMultiplier = m_rcpTotalPhotonPaths / kernelArea;
324 const auto tau = viewpoint.template get<EViewpointData::Tau>();
325 const auto viewRadiance = viewpoint.template get<EViewpointData::ViewRadiance>();
326
327 math::Spectrum radiance(tau * radianceMultiplier);
328 radiance += viewRadiance * m_rcpNumViewRadianceSamples;
329 return radiance;
330}

332template<CViewpoint Viewpoint, CPhoton Photon>
333inline std::size_t TSPPMRadianceEvaluator<Viewpoint, Photon>::getViewpointIndex(
334 const int64 sampleX, const int64 sampleY) const
335{
336 PH_ASSERT_IN_RANGE(sampleX, 0, m_statisticsRes.x());
337 PH_ASSERT_IN_RANGE(sampleY, 0, m_statisticsRes.y());
338
339 const std::size_t viewpointIdx = sampleY * m_statisticsRes.x() + sampleX;
340 PH_ASSERT_LT(viewpointIdx, m_viewpoints.size());
341
342 return viewpointIdx;
343}
