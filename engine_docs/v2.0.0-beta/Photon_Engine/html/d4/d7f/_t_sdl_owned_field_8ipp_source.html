62template<typename Owner>
63inline void TSdlOwnedField<Owner>::fromSdl(
64 Owner& owner,
65 const SdlInputClause& clause,
66 const SdlInputContext& ctx) const
67{
68 try
69 {
70 loadFromSdl(owner, clause, ctx);
71 }
72 catch(const SdlException& e)
73 {
74 if(isFallbackEnabled())
75 {
76 ownedValueToDefault(owner);
77
78 // Always log for recovered failed loading attempt
79 // (as the user provided a SDL value string for the field,
80 // a successful parse was expected)
81 //
82 PH_LOG_WARNING(SdlField, "load error from {} -> {}; value defaults to {}",
83 sdl::gen_pretty_name(ctx.getSrcClass(), this),
84 e.whatStr(),
85 valueToString(owner));
86 }
87 else
88 {
89 // Let caller handle the error if fallback is disabled
90 //
91 throw SdlLoadError(
92 "load error from " + sdl::gen_pretty_name(ctx.getSrcClass(), this) +
93 " -> " + e.whatStr() + "; " +
94 "value left uninitialized");
95 }
96 }
