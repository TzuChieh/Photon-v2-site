51template<typename T>
52inline std::shared_ptr<T> ISdlReferenceGroup::getTyped(std::string_view resourceName) const
53{
54 static_assert(std::is_base_of_v<ISdlResource, T>,
55 "T is not a SDL resource.");
56
57 std::shared_ptr<ISdlResource> rawResource = get(resourceName);
58 if(!rawResource)
59 {
60 return nullptr;
61 }
62
63 std::shared_ptr<T> castedResource = std::dynamic_pointer_cast<T>(std::move(rawResource));
64 if(!castedResource)
65 {
66 // The cast can fail if a wrong type for the resource is specified
67 // (name is correct, but with a wrong type).
68
69 throw_formatted<SdlLoadError>(
70 "resource type is not the requested type (category: {}, name: {})",
71 sdl::category_to_string(sdl::category_of<T>()), resourceName);
72 }
73
74 return castedResource;
75}
