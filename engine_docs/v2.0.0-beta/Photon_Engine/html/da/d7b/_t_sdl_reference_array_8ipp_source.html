76template<typename T, typename Owner>
77inline SdlNativeData TSdlReferenceArray<T, Owner>::ownedNativeData(Owner& owner) const
78{
79 std::vector<std::shared_ptr<T>>* const refVec = &(owner.*m_valuePtr);
80
81 SdlNativeData data;
82 if(refVec)
83 {
84 std::shared_ptr<T>* const refVecData = refVec->data();
85
86 // Read-only for ordinary access to avoid accidental object slicing and other polymorphic
87 // assignment issues. User should use direct accessor for assignment.
88 data = SdlNativeData(
89 [refVecData](std::size_t elementIdx) -> SdlGetterVariant
90 {
91 T* const originalDataPtr = refVecData[elementIdx].get();
92 return originalDataPtr
93 ? SdlNativeData::permissiveElementGetter(originalDataPtr)
94 : SdlConstInstance{};
95 });
96
97 data.numElements = refVec->size();
98 data.setDirectAccessor(AnyNonConstPtr(refVec));
99 }
100 data.elementContainer = ESdlDataFormat::SharedPtrVector;
101 data.elementType = sdl::resource_type_of<T>();
102 return data;
103}

138template<typename T, typename Owner>
139inline void TSdlReferenceArray<T, Owner>::saveToSdl(
140 const Owner& owner,
141 SdlOutputClause& out_clause,
142 const SdlOutputContext& ctx) const
143{
144 const std::vector<std::shared_ptr<T>>& referenceVector = getValueVec(owner);
145
146 // Basically generates a list of reference names
147 try
148 {
149 out_clause.value = '{';
150 for(const std::shared_ptr<T>& resource : referenceVector)
151 {
152 const auto& resourceName = ctx.getDependencyResolver().getResourceName(resource.get());
153 if(resourceName.empty())
154 {
155 throw SdlSaveError(
156 "resource name is not tracked by the dependency resolver");
157 }
158
159 const bool hasWhitespaces = string_utils::has_any_of(resourceName, string_utils::get_whitespaces());
160 out_clause.value += hasWhitespaces ? "\"" : "";
161 out_clause.value += resourceName;
162 out_clause.value += hasWhitespaces ? "\"" : "";
163 }
164 out_clause.value += '}';
165 }
166 catch(const SdlException& e)
167 {
168 throw SdlSaveError(
169 "unable to save reference array " +
170 valueToString(owner) + " -> " + e.whatStr());
171 }
172}

174template<typename T, typename Owner>
175template<typename ResourceType>
176inline std::vector<std::shared_ptr<T>> TSdlReferenceArray<T, Owner>::loadReferenceArray(
177 const SdlInputClause& clause,
178 const SdlInputContext& ctx)
179{
180 static const Tokenizer tokenizer({' ', '\t', '\n', '\r'}, {{'"', '"'}});
181
182 try
183 {
184 std::vector<std::string> referenceTokens;
185 tokenizer.tokenize(clause.value, referenceTokens);
186
187 std::vector<std::shared_ptr<T>> referenceVector(referenceTokens.size());
188 for(std::size_t i = 0; i < referenceVector.size(); ++i)
189 {
190 referenceVector[i] = loadReference(referenceTokens[i], ctx);
191 }
192
193 return referenceVector;
194 }
195 catch(const SdlException& e)
196 {
197 throw SdlLoadError("on parsing reference array -> " + e.whatStr());
198 }
199}

201template<typename T, typename Owner>
202template<typename ResourceType>
203inline std::shared_ptr<ResourceType> TSdlReferenceArray<T, Owner>::loadReference(
204 std::string_view referenceName,
205 const SdlInputContext& ctx)
206{
207 if(referenceName.empty())
208 {
209 throw SdlLoadError(
210 "reference name cannot be empty");
211 }
212
213 if(!ctx.getSrcReferences())
214 {
215 throw_formatted<SdlLoadError>(
216 "no target reference group specified");
217 }
218
219 // TODO: allow type mismatch?
220 // TODO: we may support some simple syntax such as wildcards or empty ref etc.
221 auto resource = ctx.getSrcReferences()->getTyped<ResourceType>(referenceName);
222 if(!resource)
223 {
224 throw_formatted<SdlLoadError>(
225 "cannot find resource referenced by <{}>",
226 referenceName);
227 }
228
229 return resource;
230}
