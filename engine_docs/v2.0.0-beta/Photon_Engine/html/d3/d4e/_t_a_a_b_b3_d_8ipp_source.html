242template<typename T>
243inline bool TAABB3D<T>::isIntersectingVolumeKajiyaKay(
244 const TLineSegment<T>& segment,
245 T* const out_nearHitT,
246 T* const out_farHitT) const
247{
248 PH_ASSERT(out_nearHitT);
249 PH_ASSERT(out_farHitT);
250
251 // The starting ray interval (tMin, tMax) will be incrementally intersect
252 // against each ray-slab hitting interval (t1, t2) and be updated with the
253 // resulting interval.
254 //
255 // Note that the following implementation is NaN-aware
256 // (tMin & tMax will never have NaNs)
257
258 PH_ASSERT(!std::isnan(segment.getMinT()) && !std::isnan(segment.getMaxT()));
259
260 T tMin = segment.getMinT();
261 T tMax = segment.getMaxT();
262
263 // find ray-slab hitting interval in x-axis then intersect with (tMin, tMax)
264
265 T rcpDir = T(1) / segment.getDirection().x();
266 T t1 = (m_minVertex.x() - segment.getOrigin().x()) * rcpDir;
267 T t2 = (m_maxVertex.x() - segment.getOrigin().x()) * rcpDir;
268
269 if(t1 < t2)
270 {
271 tMin = t1 > tMin ? t1 : tMin;
272 tMax = t2 < tMax ? t2 : tMax;
273 }
274 else
275 {
276 tMin = t2 > tMin ? t2 : tMin;
277 tMax = t1 < tMax ? t1 : tMax;
278 }
279
280 if(tMin > tMax)
281 {
282 return false;
283 }
284
285 // find ray-slab hitting interval in y-axis then intersect with (tMin, tMax)
286
287 rcpDir = T(1) / segment.getDirection().y();
288 t1 = (m_minVertex.y() - segment.getOrigin().y()) * rcpDir;
289 t2 = (m_maxVertex.y() - segment.getOrigin().y()) * rcpDir;
290
291 if(t1 < t2)
292 {
293 tMin = t1 > tMin ? t1 : tMin;
294 tMax = t2 < tMax ? t2 : tMax;
295 }
296 else
297 {
298 tMin = t2 > tMin ? t2 : tMin;
299 tMax = t1 < tMax ? t1 : tMax;
300 }
301
302 if(tMin > tMax)
303 {
304 return false;
305 }
306
307 // find ray-slab hitting interval in z-axis then intersect with (tMin, tMax)
308
309 rcpDir = T(1) / segment.getDirection().z();
310 t1 = (m_minVertex.z() - segment.getOrigin().z()) * rcpDir;
311 t2 = (m_maxVertex.z() - segment.getOrigin().z()) * rcpDir;
312
313 if(t1 < t2)
314 {
315 tMin = t1 > tMin ? t1 : tMin;
316 tMax = t2 < tMax ? t2 : tMax;
317 }
318 else
319 {
320 tMin = t2 > tMin ? t2 : tMin;
321 tMax = t1 < tMax ? t1 : tMax;
322 }
323
324 if(tMin > tMax)
325 {
326 return false;
327 }
328
329 *out_nearHitT = tMin;
330 *out_farHitT = tMax;
331 return true;
332}
