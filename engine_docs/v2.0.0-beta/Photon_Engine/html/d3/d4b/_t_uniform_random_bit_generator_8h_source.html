66template<typename Derived, typename Bits>
67template<typename TargetBits>
68inline TargetBits TUniformRandomBitGenerator<Derived, Bits>::generate()
69{
70 static_assert(CHAR_BIT == 8);
71
72 // Safer for bit operations (custom types are explicitly allowed).
73 static_assert(std::is_unsigned_v<TargetBits> || std::is_class_v<TargetBits>);
74
75 if constexpr(std::is_unsigned_v<TargetBits>)
76 {
77 constexpr auto bitsSize = sizeof(Bits);
78 constexpr auto targetBitsSize = sizeof(TargetBits);
79
80 if constexpr(targetBitsSize == bitsSize)
81 {
82 return bitwise_cast<TargetBits>(generate());
83 }
84 else if constexpr(targetBitsSize == 8 && bitsSize == 4)
85 {
86 // Generate 8-byte bits by combining two 4-byte bits
87 const uint64 lower4B = bitwise_cast<uint32>(generate());
88 const uint64 higher4B = bitwise_cast<uint32>(generate()) << 32;
89 return bitwise_cast<TargetBits>(higher4B | lower4B);
90
91 // Note: combining 2 `uint32` like this may not yield as good result as using 2 separate
92 // generators (with different state and sequence).
93 }
94 else if constexpr(targetBitsSize == 4 && bitsSize == 8)
95 {
96 // Just to make sure the cast will truncate `uint64`
97 static_assert(static_cast<uint32>(uint64(0xABCD'ABCD'0000'0000) >> 32) == 0xABCD'ABCD);
98
99 // Generate 4-byte bits by using the higher 4-byte in 8-byte bits
100 const auto full8B = bitwise_cast<uint64>(generate());
101 return bitwise_cast<TargetBits>(static_cast<uint32>(full8B >> 32));
102
103 // Note: generally higher bits may be of better quality
104 }
105 }
106 else
107 {
108 PH_STATIC_ASSERT_DEPENDENT_FALSE(Derived,
109 "No existing implementation can do `Bits` -> `TargetBits`.");
110 }
111}

114template<typename TargetSample, typename SourceBits>
115inline TargetSample TUniformRandomBitGenerator<Derived, Bits>::generateSample()
116{
117 static_assert(CHAR_BIT == 8);
118
119 // Safer for bit operations (custom types are explicitly allowed).
120 static_assert(std::is_unsigned_v<SourceBits> || std::is_class_v<SourceBits>);
121
122 // Samples must be within [0, 1], makes no sense for using integral types
123 // (custom types are explicitly allowed).
124 static_assert(std::is_floating_point_v<TargetSample> || std::is_class_v<TargetSample>);
125
126 if constexpr(std::is_same_v<SourceBits, uint32>)
127 {
128 const TargetSample sample = generate<SourceBits>() * TargetSample(0x1p-32);
129 PH_ASSERT_IN_RANGE_INCLUSIVE(sample, TargetSample(0.0), TargetSample(1.0));
130 return sample;
131 }
132 else if constexpr(std::is_same_v<SourceBits, uint64>)
133 {
134 const TargetSample sample = generate<SourceBits>() * TargetSample(0x1p-64);
135 PH_ASSERT_IN_RANGE_INCLUSIVE(sample, TargetSample(0.0), TargetSample(1.0));
136 return sample;
137 }
138 else
139 {
140 PH_STATIC_ASSERT_DEPENDENT_FALSE(Derived,
141 "No existing implementation can do `SourceBits` -> `TargetSample`.");
142 }
143}
