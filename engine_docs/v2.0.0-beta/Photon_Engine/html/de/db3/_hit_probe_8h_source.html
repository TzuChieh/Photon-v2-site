122inline void HitProbe::pushCache(const T& data)
123{
124 static_assert(std::is_trivially_copyable_v<T>,
125 "target type is not cacheable");
126 static_assert(sizeof(T) <= sizeof(m_cache),
127 "not enough cache to store target type, consider increasing config.PH_HIT_PROBE_CACHE_BYTES");
128
129 PH_ASSERT_MSG(m_cacheHead + sizeof(T) <= sizeof(m_cache),
130 "ran out of cache, consider increasing config.PH_HIT_PROBE_CACHE_BYTES \n"
131 "m_cacheHead = " + std::to_string(m_cacheHead) + "\n" +
132 "sizeof(T) = " + std::to_string(sizeof(T)) + "\n" +
133 "sizeof(m_cache) = " + std::to_string(sizeof(m_cache)));
134
135 std::memcpy(m_cache + m_cacheHead, &data, sizeof(T));
136 m_cacheHead += sizeof(T);
137}

139template<typename T>
140inline T HitProbe::popCache()
141{
142 static_assert(std::is_trivially_copyable_v<T>);
143 static_assert(sizeof(T) <= sizeof(m_cache));
144
145 PH_ASSERT_IN_RANGE_INCLUSIVE(m_cacheHead, sizeof(T), sizeof(m_cache));
146 m_cacheHead -= sizeof(T);
147
148 T data;
149 std::memcpy(&data, m_cache + m_cacheHead, sizeof(T));
150 return data;
151}
