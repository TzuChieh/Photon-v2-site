107inline auto VPMRadianceEvaluator::impl_onPathHitSurface(
108 const std::size_t pathLength,
109 const SurfaceHit& surfaceHit,
110 const math::Spectrum& pathThroughput) -> ViewPathTracingPolicy
111{
112 // TODO: remove hardcoded max path length
113 constexpr std::size_t MAX_PATH_LENGTH = 5;
114
115 PH_ASSERT_GE(pathLength, 1);
116
117 const BsdfQueryContext bsdfContext(ALL_ELEMENTALS, ETransport::IMPORTANCE, ESidednessPolicy::STRICT);
118 const PrimitiveMetadata* const metadata = surfaceHit.getDetail().getPrimitive()->getMetadata();
119 const SurfaceOptics* const surfaceOptics = metadata->getSurface().getOptics();
120
121 if(metadata->getSurface().getEmitter())
122 {
123 math::Spectrum viewPathRadiance;
124 metadata->getSurface().getEmitter()->evalEmittedRadiance(surfaceHit, &viewPathRadiance);
125 m_sampledRadiance.addLocal(pathThroughput * viewPathRadiance);
126 }
127
128 // FIXME: properly handle delta optics (mixed case)
129
130 if(pathLength < MAX_PATH_LENGTH && surfaceOptics->getAllPhenomena().hasAny({
131 ESurfacePhenomenon::DELTA_REFLECTION,
132 ESurfacePhenomenon::DELTA_TRANSMISSION}))
133 {
134 return ViewPathTracingPolicy().
135 traceBranchedPathFor(SurfacePhenomena({
136 ESurfacePhenomenon::DELTA_REFLECTION,
137 ESurfacePhenomenon::DELTA_TRANSMISSION})).
138 useRussianRoulette(false);
139 }
140
141 PH_ASSERT_LE(pathLength, MAX_PATH_LENGTH);
142
143 m_photonCache.clear();
144 m_photonMap->findWithinRange(surfaceHit.getPosition(), m_kernelRadius, m_photonCache);
145
146 const SurfaceTracer surfaceTracer(m_scene);
147
148 const math::Vector3R L = surfaceHit.getIncidentRay().getDirection().mul(-1);
149 const math::Vector3R Ns = surfaceHit.getShadingNormal();
150 const math::Vector3R Ng = surfaceHit.getGeometryNormal();
151
152 BsdfEvalQuery bsdfEval(bsdfContext);
153 math::Spectrum radiance(0);
154 for(const auto& photon : m_photonCache)
155 {
156 const math::Vector3R V = photon.get<EPhotonData::FROM_DIR>();
157
158 bsdfEval.inputs.set(surfaceHit, L, V);
159 if(!surfaceTracer.doBsdfEvaluation(bsdfEval))
160 {
161 continue;
162 }
163
164 math::Spectrum throughput(pathThroughput);
165 throughput.mulLocal(bsdfEval.outputs.bsdf);
166 throughput.mulLocal(lta::importance_BSDF_Ns_corrector(Ns, Ng, L, V));
167
168 radiance.addLocal(throughput * photon.get<EPhotonData::THROUGHPUT_RADIANCE>());
169 }
170
171 // OPT: cache
172 const real kernelArea = m_kernelRadius * m_kernelRadius * math::constant::pi<real>;
173 const real radianceMultiplier = 1.0_r / (kernelArea * static_cast<real>(m_numPhotonPaths));
174
175 radiance.mulLocal(radianceMultiplier);
176
177 m_sampledRadiance.addLocal(radiance);
178
179 return ViewPathTracingPolicy().kill();
180}
