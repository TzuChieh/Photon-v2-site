181inline std::string_view cut_head(const std::string_view srcStr, const std::string_view candidates)
182{
183 const auto nonCutPos = srcStr.find_first_not_of(candidates);
184
185 auto cutStr = srcStr;
186
187 // remove_prefix(): behavior is undefined for inputPos > size(), avoid that
188 // with the ternary operator
189 cutStr.remove_prefix(
190 nonCutPos != std::string_view::npos ? nonCutPos : srcStr.size());
191
192 return cutStr;
193}
194
204inline std::string_view cut_tail(const std::string_view srcStr, const std::string_view candidates)
205{
206 const auto nonCutPos = srcStr.find_last_not_of(candidates);
207
208 auto cutStr = srcStr;
209
210 // remove_suffix(): behavior is undefined for inputPos > size(), avoid that
211 // with the ternary operator;
212 // also, if <nonCutPos> is not npos, <srcStr> will not be empty
213 cutStr.remove_suffix(
214 nonCutPos != std::string_view::npos ? srcStr.size() - 1 - nonCutPos : srcStr.size());
215
216 return cutStr;
217}

228inline std::string_view cut_ends(const std::string_view srcStr, const std::string_view candidates)
229{
230 return cut_head(cut_tail(srcStr, candidates), candidates);
231}

273inline std::string_view next_token(
274 std::string_view srcStr,
275 std::string_view* const out_remainingStr = nullptr,
276 const std::string_view tokenSeparators = get_whitespaces<>())
277{
278 srcStr = cut_head(srcStr, tokenSeparators);
279
280 const auto separatorPos = srcStr.find_first_of(tokenSeparators);
281 if(separatorPos != std::string_view::npos)
282 {
283 const auto nextToken = srcStr.substr(0, separatorPos);
284 if(out_remainingStr)
285 {
286 // `separatorPos + 1` as we do not want to include the separator
287 *out_remainingStr = srcStr.substr(separatorPos + 1);
288 }
289
290 return nextToken;
291 }
292 else
293 {
294 return srcStr;
295 }

385inline void throw_from_std_errc_if_has_error(const std::errc errorCode)
386{
387 // According to several sources, 0, or zero-initialized std::errc,
388 // indicates no error.
389 //
390 // [1] see the example for std::from_chars
391 // https://en.cppreference.com/w/cpp/utility/from_chars
392 // [2] https://stackoverflow.com/a/63567008
393 //
394 constexpr std::errc NO_ERROR_VALUE = std::errc();
395
396 switch(errorCode)
397 {
398 case NO_ERROR_VALUE:
399 return;
400
401 case std::errc::invalid_argument:
402 throw InvalidArgumentException(
403 "input cannot be interpreted as a numeric value");
404
405 case std::errc::result_out_of_range:
406 throw OverflowException(
407 "result will overflow the arithmetic type");
408
409 case std::errc::value_too_large:
410 throw OutOfRangeException(
411 "result cannot fit in the output buffer");
412
413 default:
414 throw RuntimeException(
415 "unknown error: std::errc = " + std::to_string(
416 static_cast<std::underlying_type_t<std::errc>>(errorCode)));
417 }
418}

426inline T parse_float(const std::string_view floatStr)
427{
428 static_assert(std::is_floating_point_v<T>,
429 "parse_float() accepts only floating point type.");
430
431 // `std::from_chars()` do not ignore leading whitespaces, we need to do it manually
432 const std::string_view floatStrNoLeadingWS = trim_head(floatStr);
433
434 T value;
435 const std::from_chars_result result = std::from_chars(
436 floatStrNoLeadingWS.data(),
437 floatStrNoLeadingWS.data() + floatStrNoLeadingWS.size(),
438 value);
439
440 detail_from_to_char::throw_from_std_errc_if_has_error(result.ec);
441
442 return value;
443}

450template<typename T>
451inline T parse_int(std::string_view intStr)
452{
453 // TODO: option to handle base prefix (e.g., 0x)
454
455 static_assert(std::is_integral_v<T>,
456 "parse_int() accepts only integer type.");
457
458 // `std::from_chars()` do not ignore leading whitespaces, we need to do it manually
459 intStr = trim_head(intStr);
460
461 int base = 10;
462 if(intStr.starts_with("0x"))
463 {
464 base = 16;

466 // Remove "0x" as `std::from_chars()` do not recognize base prefix
467 intStr.remove_prefix(2);
468 }
469
470 T value;
471 const std::from_chars_result result = std::from_chars(
472 intStr.data(),
473 intStr.data() + intStr.size(),
474 value,
475 base);
476
477 detail_from_to_char::throw_from_std_errc_if_has_error(result.ec);
478
479 return value;

541template<std::integral T>
542inline std::size_t stringify_int_alphabetic(
543 T value,
544 char* const out_buffer,
545 const std::size_t bufferSize,
546 const int base)
547{
548 PH_ASSERT(out_buffer);
549 PH_ASSERT_IN_RANGE_INCLUSIVE(base, 2, 62);
550
551 std::size_t numCharsWritten = 0;
552
553 // Write sign
554 if constexpr(std::is_signed_v<T>)
555 {
556 if(value < 0)
557 {
558 if(bufferSize == 0)
559 {
560 throw OutOfRangeException(
561 "result cannot fit in the output buffer: 0 buffer size, cannot hold the negative sign");
562 }
563
564 out_buffer[0] = '-';
565 ++numCharsWritten;
566
567 value = -value;
568 }
569 }
570
571 // Use a temporary buffer, enough to hold base 2 output
572 std::array<unsigned char, sizeof(T) * CHAR_BIT> tmpBuffer;
573 auto tmpBufferEnd = tmpBuffer.end();
574
575 PH_ASSERT_GE(value, 0);
576 do
577 {
578 *(--tmpBufferEnd) = table::BASE62_DIGITS[value % base];
579 value /= base;
580 } while(value > 0);
581
582 auto numDigits = tmpBuffer.end() - tmpBufferEnd;
583 if(numCharsWritten + numDigits > bufferSize)
584 {
585 throw_formatted<OutOfRangeException>(
586 "result cannot fit in the output buffer: need={}, given={}",
587 numCharsWritten + numDigits, bufferSize);
588 }
589 else
590 {
591 std::copy(tmpBufferEnd, tmpBuffer.end(), out_buffer + numCharsWritten);
592 numCharsWritten += numDigits;
593 }
594
595 return numCharsWritten;
596}

608inline std::size_t stringify_int(

613{

623 out_buffer,
624 out_buffer + bufferSize,
625 value,
