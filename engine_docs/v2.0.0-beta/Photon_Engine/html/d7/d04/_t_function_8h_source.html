28template<auto Func, typename R, typename... Args>
29concept CFreeFunctionForm =
30 std::is_function_v<std::remove_pointer_t<decltype(Func)>> &&
31 std::is_invocable_r_v<R, decltype(Func), Args...>;
32
33template<auto Func, typename Class, typename R, typename... Args>
34concept CConstCallableMethodForm =
35 std::is_member_function_pointer_v<decltype(Func)> &&
36 std::is_invocable_r_v<R, decltype(Func), const Class*, Args...>;
37
38template<auto Func, typename Class, typename R, typename... Args>
39concept CNonConstCallableMethodForm =
40 std::is_member_function_pointer_v<decltype(Func)> &&
41 std::is_invocable_r_v<R, decltype(Func), Class*, Args...>;
42
43template<typename Func, typename R, typename... Args>
44concept CEmptyFunctorForm =
45 std::is_empty_v<std::decay_t<Func>> &&// so we do not need to store its states
46 std::is_default_constructible_v<std::decay_t<Func>> &&// we construct it on every call
47 std::is_invocable_r_v<R, std::decay_t<Func>, Args...>;// we call from the newly constructed value, no matter the constness
48
49template<typename Func, typename R, typename... Args>
50concept CNonEmptyTrivialFunctorForm =
51 !std::is_empty_v<std::decay_t<Func>> &&// to disambiguate from the empty form
52 !std::is_function_v<std::remove_pointer_t<std::decay_t<Func>>> &&// to disambiguate from the free function form
53 std::is_constructible_v<std::decay_t<Func>, Func> &&// we placement new using existing instance
54 std::is_trivially_copyable_v<std::decay_t<Func>> &&// so copying to a byte buffer is legal
55 std::is_trivially_destructible_v<std::decay_t<Func>> &&// we are neither storing dtor nor calling it
56 std::is_invocable_r_v<R, const std::decay_t<Func>&, Args...>;// must be const as we store its states and `operator ()` is `const`

82 // Aligning to the pointer size should be sufficient in most cases. Currently we do not align the
83 // buffer to `std::max_align_t` or anything greater to save space.
84 constexpr static std::size_t BUFFER_ALIGNMENT = alignof(void*);
85
86 constexpr static std::size_t BUFFER_SIZE = MIN_SIZE_HINT > sizeof(UnifiedCaller) + sizeof(void*)
87 ? MIN_SIZE_HINT - sizeof(UnifiedCaller)
88 : sizeof(void*);

164 template<typename Func>
165 inline TFunction(Func&& func)
166 requires (!std::is_same_v<std::decay_t<Func>, TFunction>)// avoid ambiguity during copy init
167 : TFunction()
168 {
169 if constexpr(TIsEmptyFunctor<Func>::value)
170 {
171 set<Func>();
172 }
173 else
174 {
175 static_assert(TIsNonEmptyTrivialFunctor<Func>::value,
176 "Cannot direct-init TFunction. Possible cause of errors: (1) sizeof functor exceeds current "
177 "limit, reduce functor size or increase the limit; (2) Invalid/mismatched functor signature; "
178 "(3) The direct-init ctor only works for functors. For other function types, please use setters.");

180 set<Func>(std::forward<Func>(func));
181 }

307 inline static R constCallableMethodCaller(const TFunction* const self, Args... args)
308 requires TIsConstCallableMethod<Func, Class>::value
309 {
310 const auto* const instancePtr = static_cast<const Class*>(self->m_data.u_constInstance);
311 return (instancePtr->*Func)(std::forward<Args>(args)...);
312 }

316 requires TIsNonConstCallableMethod<Func, Class>::value
317 {
318 auto* const instancePtr = static_cast<Class*>(self->m_data.u_nonConstInstance);
319
320 return (instancePtr->*Func)(std::forward<Args>(args)...);
321 }

323 template<typename Func>
324 inline static R emptyFunctorCaller(const TFunction* /* unused */, Args... args)
325 requires TIsEmptyFunctor<Func>::value
326 {
327 // Under the assumption that a stateless functor should be cheap to create (and without any
328 // side effects), we construct a new `Func` on every call to it
329 return Func{}(std::forward<Args>(args)...);
330 }
331
332 template<typename Func>
333 inline static R nonEmptyTrivialFunctorCaller(const TFunction* const self, Args... args)
334 {
335 // We do not obtain the pointer to `Func` via placement new (or `std::construct_at`).
336 // Instead, we cast it from raw buffer and laundering it is required by the standard
337 const auto& func = *std::launder(reinterpret_cast<const Func*>(self->m_data.u_buffer));
338
339 return func(std::forward<Args>(args)...);
340
341 // FIXME: currently this is UB for non implicit lifetime types--we need to init an instance
342 // and copy into it for types with std::is_implicit_lifetime_v == false, which requires C++23
343 }

345 [[noreturn]]
346 inline static R invalidFunctionCaller(const TFunction* /* unused */, Args... args)
347 {
348 throw UninitializedObjectException("Invalid function call: function is not set");
349 }

351private:
352 struct EmptyStruct
353 {};
354
355 union Data
356 {
357 // Intentionally provided so that default init of the union is a no-op.
358 EmptyStruct u_emptyStruct;
359
360 // Pointer to const class instance. May be empty except for methods.
361 const void* u_constInstance;
362
363 // Pointer to non-const class instance. May be empty except for methods.
364 void* u_nonConstInstance;
365
366 // Buffer for non-empty functors.
367 alignas(BUFFER_ALIGNMENT) std::byte u_buffer[BUFFER_SIZE];
368 };
369
370 // Ensure we are not wasting memory. Adjust buffer alignment if failed.
371 static_assert(alignof(Data) == BUFFER_ALIGNMENT);
372
373 // Member variables: smallest possible size of `TFunction` is two pointers--one for `UnifiedCaller`
374 // and another one in `Data`
375
376 Data m_data = EmptyStruct{};
377
378 // Wrapper function with unified signature for calling the actual function.
379 UnifiedCaller m_caller = &invalidFunctionCaller;
380};
