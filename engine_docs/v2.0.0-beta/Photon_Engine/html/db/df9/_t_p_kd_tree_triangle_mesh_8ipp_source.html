29template<typename Index>
30inline bool TPKdTreeTriangleMesh<Index>::isIntersecting(const Ray& ray, HitProbe& probe) const
31{
32 real closestHitT = std::numeric_limits<real>::max();
33 math::Vector3R closestHitBary;
34 std::size_t closestHitFaceIndex;
35
36 const bool hasHit = m_tree.nearestTraversal(
37 ray.getSegment(),
38 [ray, &closestHitT, &closestHitBary, &closestHitFaceIndex](
39 const TriFace& triFace,
40 const math::TLineSegment<real>& segment)
41 -> std::optional<real>
42 {
43 const Triangle triangle(triFace.getVertices());
44 const Ray raySegment(segment, ray.getTime());
45
46 real hitT;
47 math::Vector3R hitBary;
48 if(triangle.isIntersecting(raySegment, &hitT, &hitBary))
49 {
50 if(hitT < closestHitT)
51 {
52 closestHitT = hitT;
53 closestHitBary = hitBary;
54 closestHitFaceIndex = triFace.getIndex();
55 }
56
57 return hitT;
58 }
59 else
60 {
61 return {};
62 }
63 });
64
65 if(hasHit)
66 {
67 probe.pushBaseHit(this, closestHitT);
68 probe.pushCache(closestHitBary);
69 probe.pushCache(closestHitFaceIndex);
70
71 return true;
72 }
73 else
74 {
75 return false;
76 }
77}

79template<typename Index>
80inline void TPKdTreeTriangleMesh<Index>::calcIntersectionDetail(
81 const Ray& ray,
82 HitProbe& probe,
83 HitDetail* const out_detail) const
84{
85 PH_ASSERT(out_detail);
86 PH_ASSERT(m_triMesh);
87
88 const auto faceIndex = probe.popCache<std::size_t>();
89 const auto bary = probe.popCache<math::Vector3R>();
90 PH_ASSERT_MSG(bary.isNotZero() && bary.isFinite(),
91 bary.toString());
92
93 const TriFace triFace = m_triMesh->getFace(faceIndex);
94 const Triangle triangle(triFace.getVertices());
95
96 const math::Vector3R position = triangle.barycentricToSurface(bary);
97 const math::Vector3R faceNormal = triangle.getFaceNormal();
98
99 const auto shadingNormal = triFace.hasShadingNormal() ?
100 Triangle::interpolate(triFace.getShadingNormals(), bary) : faceNormal;
101
102 const auto uvw = triFace.hasTexCoords() ?
103 Triangle::interpolate(triFace.getTexCoords(), bary) : math::Vector3R(0);
104
105 PH_ASSERT_MSG(position.isFinite() && shadingNormal.isFinite() && uvw.isFinite(), "\n"
106 "position = " + position.toString() + "\n"
107 "shading-normal = " + shadingNormal.toString() + "\n"
108 "uvw = " + uvw.toString() + "\n");
109
110 // TODO: respect primitive channel
111 // (if it's default channel, use vertex uvw; otherwise, use mapper)
112
113 math::Vector3R dPdU(0.0_r), dPdV(0.0_r);
114 math::Vector3R dNdU(0.0_r), dNdV(0.0_r);
115 if(triFace.hasTexCoords())
116 {
117 const std::array<math::Vector2R, 3> uvs = {
118 {triFace.getTexCoords()[0].x, triFace.getTexCoords()[0].y},
119 {triFace.getTexCoords()[1].x, triFace.getTexCoords()[1].y},
120 {triFace.getTexCoords()[2].x, triFace.getTexCoords()[2].y}};
121
122 if(!Triangle::calcSurfaceParamDerivatives(
123 triFace.getVertices(), uvs,
124 &dPdU, &dPdV))
125 {
126 dPdU.set(0); dPdV.set(0);
127 }
128
129 if(triFace.hasShadingNormals() && !Triangle::calcSurfaceParamDerivatives(
130 triFace.getShadingNormals(), uvs,
131 &dNdU, &dNdV))
132 {
133 dNdU.set(0); dNdV.set(0);
134 }
135 }
136
137 PH_ASSERT_MSG(dPdU.isFinite() && dPdV.isFinite() &&
138 dNdU.isFinite() && dNdV.isFinite(), "\n"
139 "dPdU = " + dPdU.toString() + ", dPdV = " + dPdV.toString() + "\n"
140 "dNdU = " + dNdU.toString() + ", dNdV = " + dNdV.toString() + "\n");
141
142 out_detail->setHitIntrinsics(this, uvw, probe.getHitRayT());
143 out_detail->getHitInfo(ECoordSys::LOCAL).setAttributes(
144 position,
145 faceNormal,
146 shadingNormal);
147 out_detail->getHitInfo(ECoordSys::LOCAL).setDerivatives(
148 dPdU, dPdV, dNdU, dNdV);
149 out_detail->getHitInfo(ECoordSys::WORLD) = out_detail->getHitInfo(ECoordSys::LOCAL);
150}
