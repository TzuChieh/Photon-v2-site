70inline bool SurfaceTracer::traceNextSurface(

74{
75 PH_ASSERT(m_scene);
76 PH_ASSERT(out_X);
77
78 HitProbe probe;

81 return false;
82 }
83
84 *out_X = SurfaceHit(ray, probe);
85 sidedness.adjustForSidednessAgreement(*out_X);

87 return sidedness.isSidednessAgreed(*out_X, ray.getDirection());
