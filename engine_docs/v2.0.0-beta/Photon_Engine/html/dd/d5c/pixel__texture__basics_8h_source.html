58inline float64 uv_to_st_scalar(const float64 inputUV, const EWrapMode wrapMode)
59{
60 float64 outputST = 0.0;
61 switch(wrapMode)
62 {
63 case EWrapMode::Repeat:
64 {
65 const float64 st = math::fractional_part(inputUV);
66 outputST = st >= 0.0 ? st : st + 1.0;
67 break;
68 }
69
70 case EWrapMode::ClampToEdge:
71 {
72 outputST = math::clamp(inputUV, 0.0, 1.0);
73 break;
74 }
75
76 case EWrapMode::FlippedClampToEdge:
77 {
78 const float64 flippedST = inputUV * -1.0 + 1.0;
79 outputST = math::clamp(flippedST, 0.0, 1.0);
80 break;
81 }
82
83 default:
84 PH_ASSERT_UNREACHABLE_SECTION();
85 }
86
87 PH_ASSERT_IN_RANGE_INCLUSIVE(outputST, 0.0, 1.0);
88 return outputST;
89}
