150template<typename T, std::integral SIndex>
151inline bool TStableIndexDenseVector<T, SIndex>::remove(const SIndex stableIndex)
152{
153 if(!isStableIndexValid(stableIndex))
154 {
155 return false;
156 }
157
158 PH_ASSERT_LT(stableIndex, m_stableIndexToObject.size());
159 const OIndex objIndex = m_stableIndexToObject[stableIndex].first;
160 const OIndex lastObjIndex = static_cast<OIndex>(size() - 1);
161
162 // Swap the target object with the last object; also copy the last object's stable
163 // index to new location.
164 PH_ASSERT_LT(objIndex, m_objects.size());
165 PH_ASSERT_LT(lastObjIndex, m_objects.size());
166 PH_ASSERT_LT(objIndex, m_objectToStableIndex.size());
167 PH_ASSERT_LT(lastObjIndex, m_objectToStableIndex.size());
168 std::swap(m_objects[objIndex], m_objects[lastObjIndex]);
169 m_objectToStableIndex[objIndex] = m_objectToStableIndex[lastObjIndex];
170
171 // Update target object's validity.
172 m_stableIndexToObject[stableIndex].second = false;
173
174 // Update swapped object's stable index mapping.
175 PH_ASSERT_LT(m_objectToStableIndex[objIndex], m_stableIndexToObject.size());
176 m_stableIndexToObject[m_objectToStableIndex[objIndex]].first = objIndex;
177
178 // Add freed stable index for later use.
179 m_freeIndices.push_back(stableIndex);
180
181 m_objects.pop_back();
182 m_objectToStableIndex.pop_back();
183
184 return true;
185}

274template<typename T, std::integral SIndex>
275inline auto TStableIndexDenseVector<T, SIndex>::dispatchStableIndex()
276 -> SIndex
277{
278 SIndex stableIndex = INVALID_STABLE_INDEX;
279 if(m_freeIndices.empty())
280 {
281 stableIndex = static_cast<SIndex>(m_objects.size());
282 m_stableIndexToObject.push_back(std::make_pair(stableIndex, true));
283 }
284 else
285 {
286 stableIndex = m_freeIndices.back();
287 m_freeIndices.pop_back();
288
289 PH_ASSERT_LT(stableIndex, m_stableIndexToObject.size());
290
291 m_stableIndexToObject[stableIndex].first = static_cast<OIndex>(m_objects.size());
292 m_stableIndexToObject[stableIndex].second = true;
293 }
294
295 m_objectToStableIndex.push_back(stableIndex);
296
297 return stableIndex;
298}
