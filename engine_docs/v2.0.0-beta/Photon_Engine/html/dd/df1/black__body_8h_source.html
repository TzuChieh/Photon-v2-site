22template<typename T>
23inline T black_body_spectral_radiance_at(const T temperatureK, const T wavelengthNM)
24{
25 PH_ASSERT_GE(temperatureK, 0);
26 PH_ASSERT_GT(wavelengthNM, 0);
27
28 if(temperatureK == 0)
29 {
30 return 0;
31 }
32
33 // Using double for calculation as the values can extend a wide range
34
35 using namespace constant;
36
37 const double nume = 2.0 * h_Planck<double> * c_light<double> * c_light<double>;
38
39 const double lambda = wavelengthNM * 1e-9;// convert nm to m
40 const double lambda5 = (lambda * lambda) * (lambda * lambda) * lambda;
41 const double exp = (h_Planck<double> * c_light<double>) / (lambda * k_Boltzmann<double> * temperatureK);
42 const double deno = lambda5 * (std::exp(exp) - 1.0);
43
44 return static_cast<T>(nume / deno);
45}

57std::vector<T> black_body_spectral_radiance_curve(
