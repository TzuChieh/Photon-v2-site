109template<typename U>
110StaticAffineTransform StaticAffineTransform::makeParentedForward(const std::vector<TDecomposedTransform<U>>& fromRootToLocal)
111{
112 TMatrix4<U> parentedMatrix(TMatrix4<U>::IDENTITY());
113 TMatrix4<U> invParentedMatrix(TMatrix4<U>::IDENTITY());
114 for(const auto& transform : fromRootToLocal)
115 {
116 TMatrix4<U> matrix;
117 TMatrix4<U> invMatrix;
118 transform.genTransformMatrix(&matrix);
119 transform.genInverseTransformMatrix(&invMatrix);
120
121 parentedMatrix = parentedMatrix.mul(matrix);
122 invParentedMatrix = invMatrix.mul(invParentedMatrix);
123 }
124
125 return StaticAffineTransform(Matrix4R(parentedMatrix), Matrix4R(invParentedMatrix));
126}
