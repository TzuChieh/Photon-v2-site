10/*
11 Options for dealing with situations where a vector is within the hemisphere
12 defined by geometry normal but outside the one defined by shading normal or
13 vice versa.
14*/
15enum class ESidednessPolicy
16{
17 // Perform any calculations without caring sidedness agreement.
18 DO_NOT_CARE,

20 // A vector must lies in hemispheres defined by geometry normal and shading

67inline bool SidednessAgreement::isSidednessAgreed(
68 const SurfaceHit& X,
69 const math::Vector3R& vec) const
70{
71 switch(m_policy)
72 {
73 case ESidednessPolicy::DO_NOT_CARE:
74 case ESidednessPolicy::TRUST_GEOMETRY:
75 case ESidednessPolicy::TRUST_SHADING:
76 {
77 // no agreement issue with single input vector with these policies
78 return true;
79 break;
80 }
81
82 case ESidednessPolicy::STRICT:
83 {
84 const math::Vector3R& Ng = X.getGeometryNormal();
85 const math::Vector3R& Ns = X.getShadingNormal();
86
87 return Ng.dot(vec) * Ns.dot(vec) > 0.0_r;
88 break;
89 }
90
91 default:
92 PH_ASSERT_UNREACHABLE_SECTION();
93 return false;
94 break;
95 }
96}

98inline bool SidednessAgreement::isSameHemisphere(

106 case ESidednessPolicy::DO_NOT_CARE:

113
114 case ESidednessPolicy::TRUST_SHADING:
115 {
116 const math::Vector3R& Ns = X.getShadingNormal();
117
118 return Ns.dot(vecA) * Ns.dot(vecB) > 0.0_r;
119 break;
120 }
121
122 case ESidednessPolicy::STRICT:
123 {
124 const math::Vector3R& N = X.getGeometryNormal();
125
126 return isSidednessAgreed(X, vecA) && // Both vectors need to be strictly
127 isSidednessAgreed(X, vecB) && // agreed on sidedness.
128 vecA.dot(N) * vecB.dot(N) > 0.0_r;// Then testing hemisphere with either normal
129 // (the other normal would yield same sign)
130 break;
131 }
132
133 default:
134 PH_ASSERT_UNREACHABLE_SECTION();
