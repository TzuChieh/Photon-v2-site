120template<typename T>
121inline void HitProbe::pushCache(const T& data)
122{
123 static_assert(std::is_trivially_copyable_v<T>,
124 "target type is not cacheable");
125 static_assert(sizeof(T) <= sizeof(m_cache),
126 "not enough cache to store target type, consider increasing config.PH_HIT_PROBE_CACHE_BYTES");
127
128 PH_ASSERT_MSG(m_cacheHead + sizeof(T) <= sizeof(m_cache),
129 "ran out of cache, consider increasing config.PH_HIT_PROBE_CACHE_BYTES \n"
130 "m_cacheHead = " + std::to_string(m_cacheHead) + "\n" +
131 "sizeof(T) = " + std::to_string(sizeof(T)) + "\n" +
132 "sizeof(m_cache) = " + std::to_string(sizeof(m_cache)));
133
134 std::memcpy(m_cache + m_cacheHead, &data, sizeof(T));
135 m_cacheHead += sizeof(T);
136}

138template<typename T>
139inline T HitProbe::popCache()
140{
141 static_assert(std::is_trivially_copyable_v<T>);
142 static_assert(sizeof(T) <= sizeof(m_cache));
143
144 PH_ASSERT_IN_RANGE_INCLUSIVE(m_cacheHead, sizeof(T), sizeof(m_cache));
145 m_cacheHead -= sizeof(T);
146
147 T data;
148 std::memcpy(&data, m_cache + m_cacheHead, sizeof(T));
149 return data;
150}
