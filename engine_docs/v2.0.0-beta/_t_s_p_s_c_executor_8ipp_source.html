70template<typename Work>
71inline void TSPSCExecutor<Work>::waitAllWorks()
72{
73 PH_ASSERT(isProducerThread());
74 PH_ASSERT(m_thread.hasStarted());
75
76 // As the work queue is FIFO when observed from any single thread, to wait all works added
77 // by this thread to finish, we add a new work and wait for it to finish (could be slower
78 // because we have one extra work to process, but generally good enough)
79
80 std::atomic_flag isFinished;
81
82 CustomCallable waitWork;
83 waitWork.callable =
84 [&isFinished]()
85 {
86 // Memory effects on consumer thread should be made visible
87 isFinished.test_and_set(std::memory_order_release);
88 isFinished.notify_one();
89 };
90 m_workloadQueue.enqueue(std::move(waitWork));
91
92 // Wait for works added by this thread to finish; memory effects on consumer thread should be
93 // made visible
94 isFinished.wait(false, std::memory_order_acquire);
95}

156template<typename Work>
157inline void TSPSCExecutor<Work>::asyncProcessWorks()
158{
159 PH_ASSERT(isConsumerThread());
160 PH_ASSERT(m_thread.hasStarted());
161 PH_ASSERT(m_workProcessor);
162
163 Workload currentWorkload;
164 while(!m_isTerminationRequested.test(std::memory_order_relaxed))
165 {
166 m_workloadQueue.waitDequeue(&currentWorkload);
167
168 if(std::holds_alternative<Work>(currentWorkload))
169 {
170 m_workProcessor(std::get<Work>(currentWorkload));
171 }
172 else if(std::holds_alternative<CustomCallable>(currentWorkload))
173 {
174 std::get<CustomCallable>(currentWorkload).callable();
175 }
176 else
177 {
178 PH_ASSERT(std::holds_alternative<std::monostate>(currentWorkload));
179 PH_ASSERT_UNREACHABLE_SECTION();
180 }
181 }
182}

198template<typename Work>
199inline void TSPSCExecutor<Work>::terminate()
200{
201 // Nothing to terminate if the executor was not even started. Basically querying this state is safe as
202 // it is either called on producer thread or on dtor calling thread, and the contract is that the
203 // programmer should ensure a proper ordering of `start()` -> `requestTermination()` -> `dtor()`.
204 if(!m_thread.hasStarted())
205 {
206 return;
207 }
208
209 // Just try to break the worker loop as soon as possible (and as non-intrusive as possible)
210 m_isTerminationRequested.test_and_set(std::memory_order_relaxed);
211
212 // If the consumer is waiting or is going to wait, adding another work that sets the flag is
213 // required (so the consumer can unwait). The flag will be set on the consumer thread and is
214 // guaranteed to break the loop there as testing the flag is sequentially after setting
215 // the flag.
216 CustomCallable terminateWork;
217 terminateWork.callable =
218 [this]()
219 {
220 m_isTerminationRequested.test_and_set(std::memory_order_relaxed);
221 };
222 m_workloadQueue.enqueue(std::move(terminateWork));
223}
