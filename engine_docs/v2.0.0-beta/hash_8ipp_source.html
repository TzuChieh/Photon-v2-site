122template<typename T>
123inline uint32 murmur3_32(const T* const data, const std::size_t dataSize, const uint32 seed)
124{
125 /*
126 References:
127 [1] Wiki: https://en.wikipedia.org/wiki/MurmurHash (`murmur3_32()`)
128 [2] aappleby's smhasher: https://github.com/aappleby/smhasher/ (`MurmurHash3_x86_32()`)
129 */
130
131 static_assert(CHAR_BIT == 8);
132 static_assert(std::is_trivially_copyable_v<T>,
133 "`T` should be trivially copyable to be able to interpret it as bytes.");
134
135 constexpr uint32 c1 = 0xCC9E2D51UL;
136 constexpr uint32 c2 = 0x1B873593UL;
137 constexpr int r1 = 15;
138 constexpr int r2 = 13;
139 constexpr uint32 m = 5;
140 constexpr uint32 n = 0xE6546B64UL;
141
142 auto const bytes = reinterpret_cast<const uint8*>(data);
143 const std::size_t numBytes = dataSize * sizeof(T);
144 const std::size_t numBlocks = numBytes / 4;
145
146 uint32 h1 = seed;
147
148 // Body
149
150 // Read in blocks of 4 bytes
151 for(uint32 bi = 0; bi < numBlocks; ++bi)
152 {
153 const auto byteIndex = bi * 4;
154 PH_ASSERT_LT(byteIndex, numBytes);
155
156 uint32 block32;
157 std::copy_n(bytes + byteIndex, 4, reinterpret_cast<uint8*>(&block32));
158
159 // To remove a source of differing results across endiannesses, perform byte swap on
160 // big-endian CPUs (a swap here has no effects on hash properties though)
161 if constexpr(std::endian::native == std::endian::big)
162 {
163 block32 = std::byteswap(block32);
164 }
165
166 uint32 k1 = block32;
167
168 k1 *= c1;
169 k1 = std::rotl(k1, r1);
170 k1 *= c2;
171
172 h1 ^= k1;
173 h1 = std::rotl(h1, r2);
174 h1 = h1 * m + n;
175 }
176
177 // Tail
178
179 auto const tailBytes = bytes + numBlocks * 4;
180
181 uint32 k1 = 0;
182
183 switch(numBytes & 3)
184 {
185 case 3: k1 ^= tailBytes[2] << 16UL;
186 [[fallthrough]];
187 case 2: k1 ^= tailBytes[1] << 8UL;
188 [[fallthrough]];
189 case 1: k1 ^= tailBytes[0];
190 k1 *= c1; k1 = std::rotl(k1, r1); k1 *= c2; h1 ^= k1;
191 };
192
193 // Finalization
194
195 h1 ^= static_cast<uint32>(dataSize);
196 h1 = murmur3_bit_mix_32(h1);
197 return h1;
198}
