21/*
22 Regions are recursively refined and dispatched based on an error metric
23 calculated from two frames. A region will not be dispatched again if its
24 error is below a certain threshold. The implementation roughly follows
25 the paper written by Dammertz et al, with some modifications.
26
27 Reference:
28
29 "A Hierarchical Automatic Stopping Condition for Monte Carlo Global
30 Illumination", Holger Dammertz, Johannes Hanika, Alexander Keller,
31 Hendrik Lensch; Full Papers Proceedings of the WSCG 2010, p. 159-164.
32*/

165template<>
166inline void DammertzDispatcher::TAnalyzer<DammertzDispatcher::ERefineMode::MIDPOINT>::analyzeFinishedRegion(
167 const Region& finishedRegion,
168 const HdrRgbFrame& allEffortFrame,
169 const HdrRgbFrame& halfEffortFrame)
170{
171 using namespace math;
172
173 PH_ASSERT_GE(finishedRegion.getMinVertex().x(), 0);
174 PH_ASSERT_GE(finishedRegion.getMinVertex().y(), 0);
175 PH_ASSERT_LE(finishedRegion.getWidth(), allEffortFrame.widthPx());
176 PH_ASSERT_LE(finishedRegion.getHeight(), allEffortFrame.heightPx());
177 PH_ASSERT_LE(finishedRegion.getWidth(), halfEffortFrame.widthPx());
178 PH_ASSERT_LE(finishedRegion.getHeight(), halfEffortFrame.heightPx());
179 const TAABB2D<uint32> frameRegion(finishedRegion);
180
181 real regionError = 0;
182 for(uint32 y = frameRegion.getMinVertex().y(); y < frameRegion.getMaxVertex().y(); ++y)
183 {
184 for(uint32 x = frameRegion.getMinVertex().x(); x < frameRegion.getMaxVertex().x(); ++x)
185 {
186 HdrRgbFrame::Pixel I, A;
187 allEffortFrame.getPixel(x, y, &I);
188 halfEffortFrame.getPixel(x, y, &A);
189
190 const real numerator = I.sub(A).abs().sum();
191 const real sumOfI = I.sum();
192 const real rcpDenominator = sumOfI > 0 ? math::fast_rcp_sqrt(sumOfI) : 0;
193
194 regionError += numerator * rcpDenominator;
195 }
196 }
197 regionError /= frameRegion.getArea();
198 regionError *= fast_sqrt(frameRegion.getArea() * m_rcpNumRegionPixels);
199 PH_ASSERT_MSG(std::isfinite(regionError), std::to_string(regionError));
200
201 if(regionError >= m_splitThreshold)
202 {
203 // error is large, added for more effort
204 m_nextRegions.first = finishedRegion;
205 m_nextRegions.second = Region({0, 0});
206 }
207 else if(regionError >= m_terminateThreshold)
208 {
209 if(finishedRegion.getArea() >= MIN_REGION_AREA)
210 {
211 // error is small, splitted and added for more effort
212 const auto maxDimension = finishedRegion.getExtents().maxDimension();
213 const int64 midPoint = (finishedRegion.getMinVertex()[maxDimension] + finishedRegion.getMaxVertex()[maxDimension]) / 2;
214
215 m_nextRegions = finishedRegion.getSplitted(maxDimension, midPoint);
216 }
217 else
218 {
219 m_nextRegions.first = finishedRegion;
220 m_nextRegions.second = Region({0, 0});
221 }
222 }
223 else
224 {
225 // error is very small, no further effort needed
226 m_nextRegions.first = Region({0, 0});
227 m_nextRegions.second = Region({0, 0});
228 }
229}

231template<>
232inline void DammertzDispatcher::TAnalyzer<DammertzDispatcher::ERefineMode::MIN_ERROR_DIFFERENCE>::analyzeFinishedRegion(
233 const Region& finishedRegion,
234 const HdrRgbFrame& allEffortFrame,
235 const HdrRgbFrame& halfEffortFrame)
236{
237 using namespace math;
238
239 PH_ASSERT_GE(finishedRegion.getMinVertex().x(), 0);
240 PH_ASSERT_GE(finishedRegion.getMinVertex().y(), 0);
241 PH_ASSERT_LE(finishedRegion.getWidth(), allEffortFrame.widthPx());
242 PH_ASSERT_LE(finishedRegion.getHeight(), allEffortFrame.heightPx());
243 PH_ASSERT_LE(finishedRegion.getWidth(), halfEffortFrame.widthPx());
244 PH_ASSERT_LE(finishedRegion.getHeight(), halfEffortFrame.heightPx());
245 const TAABB2D<uint32> frameRegion(finishedRegion);
246
247 const auto regionExtents = frameRegion.getExtents();
248 const auto maxDimension = regionExtents.maxDimension();
249
250 m_accumulatedEps.resize(regionExtents[maxDimension]);
251 std::fill(m_accumulatedEps.begin(), m_accumulatedEps.end(), 0.0_r);
252
253 real summedEp = 0;
254 for(uint32 y = frameRegion.getMinVertex().y(); y < frameRegion.getMaxVertex().y(); ++y)
255 {
256 real summedRowEp = 0;
257 for(uint32 x = frameRegion.getMinVertex().x(); x < frameRegion.getMaxVertex().x(); ++x)
258 {
259 HdrRgbFrame::Pixel I, A;
260 allEffortFrame.getPixel(x, y, &I);
261 halfEffortFrame.getPixel(x, y, &A);
262
263 const real numerator = I.sub(A).abs().sum();
264 const real sumOfI = I.sum();
265 const real rcpDenominator = sumOfI > 0 ? fast_rcp_sqrt(sumOfI) : 0;
266
267 PH_ASSERT_GE(numerator * rcpDenominator, 0);
268 summedRowEp += numerator * rcpDenominator;
269
270 if(maxDimension == constant::X_AXIS)
271 {
272 m_accumulatedEps[x - frameRegion.getMinVertex().x()] += summedRowEp;
273 }
274 }
275 summedEp += summedRowEp;
276
277 if(maxDimension == constant::Y_AXIS)
278 {
279 m_accumulatedEps[y - frameRegion.getMinVertex().y()] = summedEp;
280 }
281 }
282
283 real regionError = summedEp;
284 regionError /= frameRegion.getArea();
285 regionError *= fast_sqrt(frameRegion.getArea() * m_rcpNumRegionPixels);
286 PH_ASSERT_MSG(regionError > 0 && std::isfinite(regionError), std::to_string(regionError));
287
288 if(regionError >= m_splitThreshold)
289 {
290 // error is large, added for more effort
291 m_nextRegions.first = finishedRegion;
292 m_nextRegions.second = Region({0, 0});
293 }
294 else if(regionError >= m_terminateThreshold)
295 {
296 if(finishedRegion.getArea() >= MIN_REGION_AREA)
297 {
298 // Split on the point that minimizes the difference of error
299 // across two splitted regions. To find the point, we squared the
300 // error metric (to avoid sqrt) and stripped away some constants
301 // which do not affect the result.
302
303 const real totalEps = m_accumulatedEps.back();
304
305 int64 bestPosPx = 0;
306 real minErrorDiff = totalEps * fast_rcp_sqrt(static_cast<real>(m_accumulatedEps.size()));
307 for(std::size_t i = 0; i < m_accumulatedEps.size(); ++i)
308 {
309 const real summedEp0 = m_accumulatedEps[i];
310 const real summedEp1 = totalEps - summedEp0;
311 PH_ASSERT_GE(summedEp0, 0);
312 PH_ASSERT_GE(summedEp1, 0);
313
314 const real error0 = summedEp0 * fast_rcp_sqrt(static_cast<real>(i + 1));
315 const real error1 = summedEp1 * (i != m_accumulatedEps.size() - 1 ?
316 fast_rcp_sqrt(static_cast<real>(m_accumulatedEps.size() - i - 1)) : 0);
317 const real errorDiff = std::abs(error0 - error1);
318
319 if(errorDiff < minErrorDiff)
320 {
321 minErrorDiff = errorDiff;
322 bestPosPx = static_cast<int64>(i + 1);
323 }
324 }
325
326 m_nextRegions = finishedRegion.getSplitted(
327 maxDimension,
328 finishedRegion.getMinVertex()[maxDimension] + bestPosPx);
329 }
330 else
331 {
332 m_nextRegions.first = finishedRegion;
333 m_nextRegions.second = Region({0, 0});
334 }
335 }
336 else
337 {
338 // error is very small, no further effort needed
339 m_nextRegions.first = Region({0, 0});
340 m_nextRegions.second = Region({0, 0});
341 }
342}
