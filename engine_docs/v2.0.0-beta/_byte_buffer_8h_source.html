103template<typename T>
104inline void ByteBuffer::write(const T* const data, const std::size_t numElements)
105{
106 static_assert(std::is_trivially_copyable_v<T>,
107 "writing the type into ByteBuffer is not allowed");
108
109 PH_ASSERT(data);
110 PH_ASSERT_LE(m_writeHead, m_buffer.size());
111
112 const std::size_t numDataBytes = sizeof(T) * numElements;
113 const std::size_t numAvailableBytes = m_buffer.size() - m_writeHead;
114
115 // Existing space is enough for writing
116 if(numAvailableBytes >= numDataBytes)
117 {
118 std::memcpy(m_buffer.data() + m_writeHead, data, numDataBytes);
119 }
120 else
121 {
122 // So existing space is not enough
123
124 const std::size_t numShortedBytes = numDataBytes - numAvailableBytes;
125
126 // First copy into what we have left
127 std::memcpy(m_buffer.data() + m_writeHead, data, numAvailableBytes);
128
129 // Then copy remaining bytes into the buffer (and let the buffer grow by itself)
130 m_buffer.insert(
131 m_buffer.end(),
132 reinterpret_cast<const std::byte*>(data) + numAvailableBytes,
133 reinterpret_cast<const std::byte*>(data) + numDataBytes);
134 }
135
136 m_writeHead += numDataBytes;
137
138 PH_ASSERT_LE(m_writeHead, m_buffer.size());
139}
