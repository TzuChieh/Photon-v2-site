124inline int TableFGD::calcIndex(const int iCosWi, const int iAlpha, const int iIorN, const int iIorK) const
125{
126 // make sure the indices stay in the limits
127 PH_ASSERT(0 <= iCosWi && iCosWi < m_numCosWi);
128 PH_ASSERT(0 <= iAlpha && iAlpha < m_numAlpha);
129 PH_ASSERT(0 <= iIorN && iIorN < m_numIorN);
130 PH_ASSERT(0 <= iIorK && iIorK < m_numIorK);
131
132 return iIorK + m_numIorK * (iIorN + m_numIorN * (iAlpha + m_numAlpha * iCosWi));
133}
