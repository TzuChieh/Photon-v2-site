15inline TOrthonormalBasis3<T> TOrthonormalBasis3<T>::makeFromUnitY(const TVector3<T>& unitYAxis)
16{
17 PH_ASSERT_MSG(unitYAxis.length() > T(0.9) && unitYAxis.isFinite(), unitYAxis.toString());
18
19 // choose an axis deviate enough to specified y-axis to perform cross product in order to avoid some
20 // numeric errors
21 TVector3<T> unitXAxis;
22 if(std::abs(unitYAxis.y()) < constant::rcp_sqrt_2<real>)
23 {
24 unitXAxis.set({-unitYAxis.z(), T(0), unitYAxis.x()});// yAxis cross (0, 1, 0)
25 unitXAxis.mulLocal(T(1) / std::sqrt(unitXAxis.x() * unitXAxis.x() + unitXAxis.z() * unitXAxis.z()));
26 }
27 else
28 {
29 unitXAxis.set({unitYAxis.y(), -unitYAxis.x(), T(0)});// yAxis cross (0, 0, 1)
30 unitXAxis.mulLocal(T(1) / std::sqrt(unitXAxis.x() * unitXAxis.x() + unitXAxis.y() * unitXAxis.y()));
31 }
32
33 const TVector3<T> unitZAxis = unitXAxis.cross(unitYAxis);
34
35 PH_ASSERT_MSG(unitXAxis.isFinite() && unitZAxis.isFinite(), "\n"
36 "unit-x-axis = " + unitXAxis.toString() + "\n"
37 "unit-z-axis = " + unitZAxis.toString() + "\n");
38
39 return TOrthonormalBasis3(unitXAxis, unitYAxis, unitZAxis);

94inline T TOrthonormalBasis3<T>::cosPhi(const TVector3<T>& unitVec) const
95{

127template<typename T>
128inline T TOrthonormalBasis3<T>::tanPhi(const TVector3<T>& unitVec) const
129{
130 const T zProjection = unitVec.dot(m_zAxis);
131 const T xProjection = unitVec.dot(m_xAxis);
132
133 if(zProjection != T(0))
134 {
135 return math::clamp(
136 xProjection / zProjection,
137 std::numeric_limits<T>::lowest(),
138 std::numeric_limits<T>::max());
139 }
140 else
141 {
142 switch(math::sign(xProjection))
143 {
144 case -1: return std::numeric_limits<T>::lowest();
145 case 0: return T(0);
146 case 1: return std::numeric_limits<T>::max();
147 }
148
149 PH_ASSERT_UNREACHABLE_SECTION();
150 return T(0);
