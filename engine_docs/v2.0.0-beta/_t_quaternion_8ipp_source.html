44template<typename T>
45inline TQuaternion<T>::TQuaternion(const TMatrix4<T>& rotationMatrix) :
46 TQuaternion()
47{
48 const TMatrix4<T>& m = rotationMatrix;
49
50 const T trace = m.m[0][0] + m.m[1][1] + m.m[2][2];
51 if(trace > 0)
52 {
53 const T s = static_cast<T>(0.5) / std::sqrt(trace + 1);
54 x = (m.m[2][1] - m.m[1][2]) * s;
55 y = (m.m[0][2] - m.m[2][0]) * s;
56 z = (m.m[1][0] - m.m[0][1]) * s;
57 w = static_cast<T>(0.25) / s;
58 }
59 else
60 {
61 if(m.m[0][0] > m.m[1][1] && m.m[0][0] > m.m[2][2])
62 {
63 const T s = static_cast<T>(0.5) / std::sqrt(1 + m.m[0][0] - m.m[1][1] - m.m[2][2]);
64 x = static_cast<T>(0.25) / s;
65 y = (m.m[0][1] + m.m[1][0]) * s;
66 z = (m.m[0][2] + m.m[2][0]) * s;
67 w = (m.m[2][1] - m.m[1][2]) * s;
68 }
69 else if(m.m[1][1] > m.m[2][2])
70 {
71 const T s = static_cast<T>(0.5) / std::sqrt(1 + m.m[1][1] - m.m[0][0] - m.m[2][2]);
72 x = (m.m[0][1] + m.m[1][0]) * s;
73 y = static_cast<T>(0.25) / s;
74 z = (m.m[1][2] + m.m[2][1]) * s;
75 w = (m.m[0][2] - m.m[2][0]) * s;
76 }
77 else
78 {
79 const T s = static_cast<T>(0.5) / std::sqrt(1 + m.m[2][2] - m.m[0][0] - m.m[1][1]);
80 x = (m.m[0][2] + m.m[2][0]) * s;
81 y = (m.m[1][2] + m.m[2][1]) * s;
82 z = static_cast<T>(0.25) / s;
83 w = (m.m[1][0] - m.m[0][1]) * s;
84 }
85 }
86
87 normalizeLocal();
88}

121template<typename T>
122inline void TQuaternion<T>::toRotationMatrix(TMatrix4<T>* const out_result) const
123{
124 out_result->m[0][0] = 1 - 2*(y*y + z*z);
125 out_result->m[0][1] = 2*(x*y - z*w);
126 out_result->m[0][2] = 2*(y*w + x*z);
127 out_result->m[0][3] = 0;
128
129 out_result->m[1][0] = 2*(x*y + z*w);
130 out_result->m[1][1] = 1 - 2*(x*x + z*z);
131 out_result->m[1][2] = 2*(y*z - x*w);
132 out_result->m[1][3] = 0;
133
134 out_result->m[2][0] = 2*(x*z - y*w);
135 out_result->m[2][1] = 2*(y*z + x*w);
136 out_result->m[2][2] = 1 - 2*(x*x + y*y);
137 out_result->m[2][3] = 0;
138
139 out_result->m[3][0] = 0;
140 out_result->m[3][1] = 0;
141 out_result->m[3][2] = 0;
142 out_result->m[3][3] = 1;
143}
