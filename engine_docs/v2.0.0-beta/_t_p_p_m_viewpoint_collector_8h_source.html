75inline bool TPPMViewpointCollector<Viewpoint>::impl_onReceiverSampleStart(
76 const math::Vector2D& rasterCoord,
77 const math::Spectrum& pathThroughput)
78{
79 if(pathThroughput.isZero())

81 // TODO: should we add a 0-contribution viewpoint?

84

95 m_viewpoint.template set<EViewpointData::TAU>(math::Spectrum(0));
96 }
97
98 m_receiverSampleViewpoints = 0;

100 return true;

103template<typename Viewpoint>
104inline auto TPPMViewpointCollector<Viewpoint>::impl_onPathHitSurface(
105 const std::size_t pathLength,
106 const SurfaceHit& surfaceHit,
107 const math::Spectrum& pathThroughput) -> ViewPathTracingPolicy
108{
109 const PrimitiveMetadata* const metadata = surfaceHit.getDetail().getPrimitive()->getMetadata();
110 const SurfaceOptics* const optics = metadata->getSurface().getOptics();
111
112 // check if there's any non-delta elemental
113 SurfacePhenomena phenomena = optics->getAllPhenomena();
114 phenomena.turnOff({
115 ESurfacePhenomenon::DELTA_REFLECTION,
116 ESurfacePhenomenon::DELTA_TRANSMISSION});
117 if(!phenomena.isEmpty())
118 {
119 // It is okay to add a viewpoint without specifying which surface
120 // elemental is used--since it is impossible for delta elementals
121 // to have non-zero contribution in any BSDF evaluation process, so
122 // we will not double-count any path throughput.
123 addViewpoint(
124 surfaceHit,
125 surfaceHit.getIncidentRay().getDirection().mul(-1),
126 pathThroughput);
127 }
128
129 if(pathLength < m_maxViewpointDepth &&
130 optics->getAllPhenomena().hasAny({
131 ESurfacePhenomenon::DELTA_REFLECTION,
132 ESurfacePhenomenon::DELTA_TRANSMISSION}))
133 {
134 return ViewPathTracingPolicy().
135 traceBranchedPathFor(SurfacePhenomena({
136 ESurfacePhenomenon::DELTA_REFLECTION,
137 ESurfacePhenomenon::DELTA_TRANSMISSION})).
138 useRussianRoulette(false);
139 }
140 else
141 {
142 PH_ASSERT_LE(pathLength, m_maxViewpointDepth);
143
144 return ViewPathTracingPolicy().kill();
145 }
146}

190template<typename Viewpoint>
191void TPPMViewpointCollector<Viewpoint>::addViewpoint(
192 const SurfaceHit& surfaceHit,
193 const math::Vector3R& viewDir,
194 const math::Spectrum& pathThroughput)
195{
196 if constexpr(Viewpoint::template has<EViewpointData::VIEW_RADIANCE>())
197 {
198 const PrimitiveMetadata* const metadata = surfaceHit.getDetail().getPrimitive()->getMetadata();
199 PH_ASSERT(metadata);
200
201 math::Spectrum viewRadiance(0);
202 if(metadata->getSurface().getEmitter())
203 {
204 metadata->getSurface().getEmitter()->evalEmittedRadiance(surfaceHit, &viewRadiance);
205 }
206 m_viewpoint.template set<EViewpointData::VIEW_RADIANCE>(viewRadiance);
207 }
208
209 if constexpr(Viewpoint::template has<EViewpointData::SURFACE_HIT>()) {
210 m_viewpoint.template set<EViewpointData::SURFACE_HIT>(surfaceHit);
211 }
212 if constexpr(Viewpoint::template has<EViewpointData::VIEW_THROUGHPUT>()) {
213 m_viewpoint.template set<EViewpointData::VIEW_THROUGHPUT>(pathThroughput);
214 }
215 if constexpr(Viewpoint::template has<EViewpointData::VIEW_DIR>()) {
216 m_viewpoint.template set<EViewpointData::VIEW_DIR>(viewDir);
217 }
218
219 m_viewpoints.push_back(m_viewpoint);
220 ++m_receiverSampleViewpoints;
221}
