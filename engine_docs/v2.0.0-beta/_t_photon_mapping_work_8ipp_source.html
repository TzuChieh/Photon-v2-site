47template<typename Photon>
48inline void TPhotonMappingWork<Photon>::doWork()
49{
50 // FIXME: currently we exit immediately when photon buffer is full; we should trace a full path instead
51
52 Timer timer;
53 timer.start();
54
55 const BsdfQueryContext bsdfContext(ALL_ELEMENTALS, ETransport::IMPORTANCE, ESidednessPolicy::STRICT);
56 const SurfaceTracer surfaceTracer(m_scene);
57
58 const auto raySampleHandle = m_sampleGenerator->declareStageND(2, m_numPhotons);
59 m_sampleGenerator->prepareSampleBatch();// HACK: check if succeeded
60 auto raySamples = m_sampleGenerator->getSamplesND(raySampleHandle);
61
62 std::size_t numStoredPhotons = 0;
63 *m_numPhotonPaths = 0;
64 std::size_t photonCounter = 0;
65 while(numStoredPhotons < m_numPhotons)
66 {
67 ++(*m_numPhotonPaths);
68
69 SampleFlow sampleFlow = raySamples.readSampleAsFlow();
70
71 Ray tracingRay;
72 math::Spectrum emittedRadiance;
73 math::Vector3R emitN;
74 real pdfA;
75 real pdfW;
76 m_scene->emitRay(sampleFlow, &tracingRay, &emittedRadiance, &emitN, &pdfA, &pdfW);
77 if(pdfA * pdfW == 0.0_r)
78 {
79 continue;
80 }
81
82 // here 0-bounce lighting is not accounted for
83
84 math::Spectrum throughputRadiance(emittedRadiance);
85 throughputRadiance.divLocal(pdfA);
86 throughputRadiance.divLocal(pdfW);
87 throughputRadiance.mulLocal(emitN.absDot(tracingRay.getDirection()));
88
89 // start tracing single photon path
90 while(!throughputRadiance.isZero())
91 {
92 SurfaceHit surfaceHit;
93 if(!surfaceTracer.traceNextSurface(tracingRay, bsdfContext.sidedness, &surfaceHit))
94 {
95 break;
96 }
97
98 const PrimitiveMetadata* metadata = surfaceHit.getDetail().getPrimitive()->getMetadata();
99 const SurfaceOptics* optics = metadata->getSurface().getOptics();
100
101 math::Spectrum weightedThroughputRadiance;
102 if(RussianRoulette::surviveOnLuminance(throughputRadiance, sampleFlow, &weightedThroughputRadiance))
103 {
104 throughputRadiance = weightedThroughputRadiance;
105
106 Photon photon;
107
108 if constexpr(Photon::template has<EPhotonData::POSITION>()) {
109 photon.template set<EPhotonData::POSITION>(surfaceHit.getPosition());
110 }
111 if constexpr(Photon::template has<EPhotonData::THROUGHPUT_RADIANCE>()) {
112 photon.template set<EPhotonData::THROUGHPUT_RADIANCE>(throughputRadiance);
113 }
114 if constexpr(Photon::template has<EPhotonData::FROM_DIR>()) {
115 photon.template set<EPhotonData::FROM_DIR>(tracingRay.getDirection().mul(-1));
116 }
117
118 m_photonBuffer[numStoredPhotons++] = photon;
119 ++photonCounter;
120
121 if(numStoredPhotons == m_numPhotons)
122 {
123 break;
124 }
125 }// end if photon survived
126 else
127 {
128 break;
129 }
130
131 BsdfSampleQuery bsdfSample(bsdfContext);
132 Ray sampledRay;
133 bsdfSample.inputs.set(surfaceHit, tracingRay.getDirection().mul(-1));
134 if(!surfaceTracer.doBsdfSample(bsdfSample, sampleFlow, &sampledRay))
135 {
136 break;
137 }
138
139 math::Vector3R V = tracingRay.getDirection().mul(-1);
140 math::Vector3R L = bsdfSample.outputs.L;
141 math::Vector3R Ng = surfaceHit.getGeometryNormal();
142 math::Vector3R Ns = surfaceHit.getShadingNormal();
143 throughputRadiance.mulLocal(bsdfSample.outputs.pdfAppliedBsdf);
144 throughputRadiance.mulLocal(lta::importance_BSDF_Ns_corrector(Ns, Ng, L, V));
145 throughputRadiance.mulLocal(Ns.absDot(L));
146
147 tracingRay = sampledRay;
148 }// end single photon path
149
150 if(photonCounter >= 16384)
151 {
152 timer.stop();
153 setElapsedMs(timer.getDeltaMs());
154
155 if(m_statistics)
156 {
157 m_statistics->asyncAddNumTracedPhotons(photonCounter);
158 }
159
160 photonCounter = 0;
161 }
162 }// end while photon buffer is not full
163
164 timer.stop();
165 setElapsedMs(timer.getDeltaMs());
166
167 if(m_statistics)
168 {
169 m_statistics->asyncAddNumTracedPhotons(photonCounter);
170 }
171}
