104template<typename U>
105inline auto StaticRigidTransform::getScaleFreeTransforms(const std::vector<TDecomposedTransform<U>>& transforms)
106 -> std::vector<TDecomposedTransform<U>>
107{
108 std::vector<TDecomposedTransform<U>> scaleFreeTransforms;
109 for(const auto& transform : transforms)
110 {
111 // FIXME: better scale tolerance value, not hardcoded like this
112 if(!transform.hasScaleEffect(0.000001_r))
113 {
114 scaleFreeTransforms.push_back(transform);
115 }
116 else
117 {
118 PH_LOG_WARNING(StaticRigidTransform,
119 "scale effect detected, which is {}, ignoring", transform.getScale().toString());
120
121 scaleFreeTransforms.push_back(TDecomposedTransform<U>(transform).setScale(1));
122 }
123 }
124
125 PH_ASSERT(scaleFreeTransforms.size() == transforms.size());
126
127 return scaleFreeTransforms;
