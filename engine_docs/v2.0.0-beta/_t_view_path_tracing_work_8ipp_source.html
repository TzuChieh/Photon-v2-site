42template<typename ViewPathHandler>
43inline void TViewPathTracingWork<ViewPathHandler>::doWork()
44{
45 PH_ASSERT(m_handler);
46
47 const auto rasterSampleHandle = m_sampleGenerator->declareStageND(
48 2,
49 m_rasterRegion.getArea(),
50 {static_cast<std::size_t>(m_rasterRegion.getWidth()), static_cast<std::size_t>(m_rasterRegion.getHeight())});
51
52 const auto raySampleHandle = m_sampleGenerator->declareStageND(2, m_rasterRegion.getArea());
53
54 const math::TAABB2D<float64> rRegion(m_rasterRegion);
55
56 while(m_sampleGenerator->prepareSampleBatch())
57 {
58 auto rasterSamples = m_sampleGenerator->getSamplesND(rasterSampleHandle);
59 auto raySamples = m_sampleGenerator->getSamplesND(raySampleHandle);
60 for(std::size_t i = 0; i < rasterSamples.numSamples(); ++i)
61 {
62 // TODO: use TArithmeticArray directly
63
64 const auto rasterCoord = rRegion.sampleToSurface(math::sample_cast<float64>(rasterSamples.template get<2>(i)));
65 SampleFlow sampleFlow = raySamples.readSampleAsFlow();
66
67 Ray tracingRay;
68 const auto quantityWeight = m_receiver->receiveRay(rasterCoord, &tracingRay);
69 tracingRay.reverse();
70
71 math::Spectrum pathThroughput(quantityWeight);
72 if(!m_handler->onReceiverSampleStart(rasterCoord, pathThroughput))
73 {
74 m_handler->onReceiverSampleEnd();
75 continue;
76 }
77
78 traceViewPath(
79 tracingRay,
80 pathThroughput,
81 0,
82 sampleFlow);
83
84 m_handler->onReceiverSampleEnd();
85 }// end for single sample
86
87 m_handler->onSampleBatchFinished();
88 }// end while single sample batch

91template<typename ViewPathHandler>
92inline void TViewPathTracingWork<ViewPathHandler>::traceViewPath(
93 Ray tracingRay,
94 math::Spectrum pathThroughput,
95 std::size_t pathLength,
96 SampleFlow& sampleFlow)
97{
98 const SurfaceTracer surfaceTracer(m_scene);
99 while(true)
100 {
101 SurfaceHit surfaceHit;
102 if(!surfaceTracer.traceNextSurface(tracingRay, BsdfQueryContext().sidedness, &surfaceHit))
103 {
104 break;
105 }
106
107 ++pathLength;
108 const ViewPathTracingPolicy& policy = m_handler->onPathHitSurface(pathLength, surfaceHit, pathThroughput);
109 if(policy.isKilled())
110 {
111 break;
112 }
113
114 // FIXME: reversed then assigned again later seems to be dangerous, state is unclear
115 tracingRay.reverse();
116 const math::Vector3R V = tracingRay.getDirection();
117 const math::Vector3R N = surfaceHit.getShadingNormal();
118
119 if(policy.getSampleMode() == EViewPathSampleMode::SINGLE_PATH)
120 {
121 BsdfSampleQuery bsdfSample(BsdfQueryContext(policy.getTargetElemental(), ETransport::RADIANCE, ESidednessPolicy::STRICT));
122 bsdfSample.inputs.set(surfaceHit, V);
123 Ray sampledRay;
124 if(!surfaceTracer.doBsdfSample(bsdfSample, sampleFlow, &sampledRay))
125 {
126 break;
127 }
128
129 pathThroughput.mulLocal(bsdfSample.outputs.pdfAppliedBsdf);
130 pathThroughput.mulLocal(N.absDot(bsdfSample.outputs.L));
131
132 if(policy.useRussianRoulette())
133 {
134 math::Spectrum weightedThroughput;
135 if(RussianRoulette::surviveOnLuminance(pathThroughput, sampleFlow, &weightedThroughput))
136 {
137 pathThroughput = weightedThroughput;
138 }
139 else
140 {
141 break;
142 }
143 }
144
145 tracingRay = sampledRay;
146 }
147 else
148 {
149 traceElementallyBranchedPath(policy, V, N, surfaceHit, pathThroughput, pathLength, sampleFlow);
150 break;
151 }
152 }// end while true
153}

155template<typename ViewPathHandler>
156inline void TViewPathTracingWork<ViewPathHandler>::traceElementallyBranchedPath(
157 const ViewPathTracingPolicy& policy,

159 const math::Vector3R& N,
160 const SurfaceHit& surfaceHit,
161 const math::Spectrum& pathThroughput,
162 const std::size_t pathLength,
163 SampleFlow& sampleFlow)
164{
165 PH_ASSERT(policy.getSampleMode() == EViewPathSampleMode::ELEMENTAL_BRANCH);
166
167 const SurfaceTracer surfaceTracer(m_scene);
168
169 const PrimitiveMetadata* metadata = surfaceHit.getDetail().getPrimitive()->getMetadata();
170 const SurfaceOptics* surfaceOptics = metadata->getSurface().getOptics();
171
172 const SurfacePhenomena targetPhenomena = policy.getTargetPhenomena();
173 for(SurfaceElemental i = 0; i < surfaceOptics->numElementals(); ++i)
174 {
175 if(targetPhenomena.hasNone({surfaceOptics->getPhenomenonOf(i)}))
176 {
177 continue;
178 }
179
180 BsdfSampleQuery sample(BsdfQueryContext(i, ETransport::RADIANCE, ESidednessPolicy::STRICT));
181 sample.inputs.set(surfaceHit, V);
182
183 Ray sampledRay;
184 if(!surfaceTracer.doBsdfSample(sample, sampleFlow, &sampledRay))
185 {
186 continue;
187 }
188
189 math::Spectrum elementalPathThroughput(pathThroughput);
190 elementalPathThroughput.mulLocal(sample.outputs.pdfAppliedBsdf);
191 elementalPathThroughput.mulLocal(N.absDot(sampledRay.getDirection()));
192
193 if(policy.useRussianRoulette())
194 {
195 math::Spectrum weightedThroughput;
196 if(RussianRoulette::surviveOnLuminance(elementalPathThroughput, sampleFlow, &weightedThroughput))
197 {
198 elementalPathThroughput = weightedThroughput;
199 }
200 else
201 {
202 continue;
203 }
204 }
205
206 traceViewPath(
207 sampledRay,
208 elementalPathThroughput,
209 pathLength,
210 sampleFlow);
211 }// end for each phenomenon
