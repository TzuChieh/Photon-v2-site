46template<typename T>
47inline bool TSphere<T>::isIntersectingNaive(
48 const TLineSegment<T>& segment,
49 real* const out_hitT) const
50{
51 PH_ASSERT_GE(m_radius, T(0));
52 PH_ASSERT(out_hitT);
53
54 // Construct a ray/line then use its parametric form to test against the sphere and solve for t
55 //
56 // ray origin: o
57 // ray direction: d
58 // sphere center: c
59 // sphere radius: r
60 // intersection point: p
61 // vector dot: *
62 // ray equation: o + td (t is a scalar variable)
63 //
64 // To find the intersection point, the length of vector (td - oc) must equals r.
65 // This is equivalent to (td - oc)*(td - oc) = r^2. After reformatting, we have
66 //
67 // t^2(d*d) - 2t(d*oc) + (oc*oc) - r^2 = 0 --- (1)
68 //
69 // Solving equation (1) for t will yield the intersection point (o + td).
70
71 // FIXME: T may be more precise than float64
72
73 const Vector3D rayO(segment.getOrigin());
74 const Vector3D rayD(segment.getDirection());
75
76 // Vector from ray origin (o) to sphere center (c)
77 const Vector3D oc = Vector3D(0).sub(rayO);
78
79 // a in equation (1)
80 const float64 a = rayD.dot(rayD);
81
82 // b in equation (1) (-2 is cancelled while solving t)
83 const float64 b = rayD.dot(oc);
84
85 // c in equation (1)
86 const float64 c = oc.dot(oc) - static_cast<float64>(m_radius) * m_radius;
87
88 float64 D = b * b - a * c;
89 if(D < 0.0)
90 {
91 return false;
92 }
93 else
94 {
95 D = std::sqrt(D);
96
97 const float64 rcpA = 1.0 / a;
98
99 // t = (b +- D) / a
100 // t1 must be <= t2 (rcpA & D always > 0)
101 const float64 t1 = (b - D) * rcpA;
102 const float64 t2 = (b + D) * rcpA;
103
104 PH_ASSERT_MSG(t1 <= t2, "\n"
105 "t1 = " + std::to_string(t1) + "\n"
106 "t2 = " + std::to_string(t2) + "\n"
107 "(a, b, c) = (" + std::to_string(a) + ", " + std::to_string(b) + ", " + std::to_string(c) + ")\n"
108 "ray-origin = " + rayO.toString() + "\n"
109 "ray-direction = " + rayD.toString());
110
111 // We now know that the ray/line intersects the sphere, but it may not be the case for the line
112 // segment. We test all cases with t1 <= t2 being kept in mind:
113
114 float64 tClosest;
115
116 // t1 is smaller than t2, we test t1 first
117 if(segment.getMinT() <= t1 && t1 <= segment.getMaxT())
118 {
119 tClosest = t1;
120 }
121 // test t2
122 else if(segment.getMinT() <= t2 && t2 <= segment.getMaxT())
123 {
124 tClosest = t2;
125 }
126 // no intersection found in [t_min, t_max] (NaN aware)
127 else
128 {
129 return false;
130 }
131
132 PH_ASSERT_IN_RANGE_INCLUSIVE(tClosest, segment.getMinT(), segment.getMaxT());
133
134 *out_hitT = static_cast<real>(tClosest);
135 return true;
136 }
137}

145template<typename T>
146inline bool TSphere<T>::isIntersectingHearnBaker(
147 const TLineSegment<T>& segment,
148 real* const out_hitT) const
149{
150 PH_ASSERT_GE(m_radius, T(0));
151 PH_ASSERT(out_hitT);
152
153 // We use the same notation as `isIntersectingNaive()` for the geometries
154
155 const Vector3D rayO(segment.getOrigin());
156 const Vector3D rayD(segment.getDirection());
157 const Vector3D unitRayD = rayD.normalize();
158
159 // Vector from sphere center (c) to ray origin (o)
160 // (vector f in Ray Tracing Gems, note this is the negated version of oc in `isIntersectingNaive()`)
161 const Vector3D co = rayO;// rayO - Vector3D(0);
162
163 const float64 r2 = static_cast<float64>(m_radius) * m_radius;
164
165 // a in equation (1) in `isIntersectingNaive()`
166 // (same as in RT Gems)
167 const float64 a = rayD.dot(rayD);
168
169 // b term in RT Gems eq. 8 without the negative sign (handled in q later to save one negation)
170 const float64 b = co.dot(rayD);
171
172 // vector fd in RT Gems eq. 5 (curvy_L^2)
173 const Vector3D fd = co - unitRayD * co.dot(unitRayD);
174
175 // Basically the discriminant term in RT Gems eq. 5 with 4s and 2s eliminated
176 const float64 D = a * (r2 - fd.dot(fd));
177 if(D < 0.0)
178 {
179 return false;
180 }
181 else
182 {
183 // c in equation (1) in `isIntersectingNaive()`
184 // (same as in RT Gems)
185 const float64 c = co.dot(co) - r2;
186
187 const float64 sqrtD = std::sqrt(D);
188
189 // q term in RT Gems eq. 10
190 const float64 q = (b >= 0.0) ? -sqrtD - b : sqrtD - b;
191
192 // t0 & t1 term in RT Gems eq. 10 (here we let t0 <= t1)
193 // (unlike `isIntersectingNaive()`, we cannot determine which one is smaller due to its formulation)
194 float64 t0 = c / q;
195 float64 t1 = q / a;
196 if(t0 > t1)
197 {
198 std::swap(t0, t1);
199 }
200
201 PH_ASSERT_MSG(t0 <= t1, "\n"
202 "t0 = " + std::to_string(t0) + "\n"
203 "t1 = " + std::to_string(t1) + "\n"
204 "(a, b, c) = (" + std::to_string(a) + ", " + std::to_string(b) + ", " + std::to_string(c) + ")\n"
205 "ray-origin = " + rayO.toString() + "\n"
206 "ray-direction = " + rayD.toString());
207
208 // We now know that the ray/line intersects the sphere, but it may not be the case for the line
209 // segment. We test all cases with t0 <= t1 being kept in mind:
210
211 float64 tClosest;
212
213 // t0 is smaller than t1, we test t0 first
214 if(segment.getMinT() <= t0 && t0 <= segment.getMaxT())
215 {
216 tClosest = t0;
217 }
218 // test t1
219 else if(segment.getMinT() <= t1 && t1 <= segment.getMaxT())
220 {
221 tClosest = t1;
222 }
223 // no intersection found in [t_min, t_max] (NaN aware)
224 else
225 {
226 return false;
227 }
228
229 PH_ASSERT_IN_RANGE_INCLUSIVE(tClosest, segment.getMinT(), segment.getMaxT());
230
231 *out_hitT = static_cast<real>(tClosest);
232 return true;
233 }

255template<typename T>
256inline bool TSphere<T>::mayOverlapVolume(const TAABB3D<T>& volume) const
257{
258 // Intersection test for solid box and hollow sphere.
259 // Reference: Jim Arvo's algorithm in Graphics Gems 2
260
261 const T radius2 = math::squared(m_radius);
262
263 // These variables are gonna store minimum and maximum squared distances
264 // from the sphere's center to the AABB volume.
265 T minDist2 = T(0);
266 T maxDist2 = T(0);
267
268 T a, b;
269
270 a = math::squared(volume.getMinVertex().x());
271 b = math::squared(volume.getMaxVertex().x());
272 maxDist2 += std::max(a, b);
273 if (T(0) < volume.getMinVertex().x()) minDist2 += a;
274 else if(T(0) > volume.getMaxVertex().x()) minDist2 += b;
275
276 a = math::squared(volume.getMinVertex().y());
277 b = math::squared(volume.getMaxVertex().y());
278 maxDist2 += std::max(a, b);
279 if (T(0) < volume.getMinVertex().y()) minDist2 += a;
280 else if(T(0) > volume.getMaxVertex().y()) minDist2 += b;
281
282 a = math::squared(volume.getMinVertex().z());
283 b = math::squared(volume.getMaxVertex().z());
284 maxDist2 += std::max(a, b);
285 if (T(0) < volume.getMinVertex().z()) minDist2 += a;
286 else if(T(0) > volume.getMaxVertex().z()) minDist2 += b;
287
288 return minDist2 <= radius2 && radius2 <= maxDist2;
289}

291template<typename T>
292inline TVector3<T> TSphere<T>::sampleToSurfaceArchimedes(const std::array<T, 2>& sample) const
293{
294 PH_ASSERT_IN_RANGE_INCLUSIVE(sample[0], static_cast<T>(0), static_cast<T>(1));
295 PH_ASSERT_IN_RANGE_INCLUSIVE(sample[1], static_cast<T>(0), static_cast<T>(1));
296 PH_ASSERT_GE(m_radius, static_cast<T>(0));
297
298 const T y = static_cast<T>(2) * (sample[0] - static_cast<T>(0.5));
299 const T phi = constant::two_pi<T> * sample[1];
300 const T r = std::sqrt(std::max(static_cast<T>(1) - y * y, static_cast<T>(0)));
301
302 const auto localUnitPos = TVector3<T>(
303 r * std::sin(phi),
304 y,
305 r * std::cos(phi));
306
307 return localUnitPos * m_radius;
308}

357template<typename T>
358inline TVector2<T> TSphere<T>::surfaceToPhiTheta(const TVector3<T>& surface) const
359{
360 using namespace math::constant;
361
362 PH_ASSERT_GT(m_radius, 0);
363
364 const math::TVector3<T>& unitDir = surface.div(m_radius);
365
366 const T cosTheta = math::clamp(unitDir.y(), static_cast<T>(-1), static_cast<T>(1));
367
368 const T theta = std::acos(cosTheta); // [ 0, pi]
369 const T phiRaw = std::atan2(unitDir.x(), unitDir.z()); // [-pi, pi]
370 const T phi = phiRaw >= static_cast<T>(0) ? phiRaw : two_pi<T> + phiRaw;// [ 0, 2*pi]
371
372 return {phi, theta};
373}

391template<typename T>
392template<typename SurfaceToUv>
393inline std::pair<TVector3<T>, TVector3<T>> TSphere<T>::surfaceDerivativesWrtUv(
394 const TVector3<T>& surface,
395 SurfaceToUv surfaceToUv,
396 T hInRadians) const
397{
398 static_assert(std::is_invocable_r_v<TVector2<T>, SurfaceToUv, TVector3<T>>,
399 "A surface to UV mapper must accept Vector3 position and return Vector2 "
400 "UV or any type that is convertible to the mentioned ones");
401
402 PH_ASSERT_GT(hInRadians, 0);
403
404 const math::TVector3<T>& normal = surface.div(m_radius);
405 hInRadians = std::min(hInRadians, math::to_radians<T>(45));
406
407 // Calculate displacement vectors on hit normal's tangent plane
408 // (with small angle approximation)
409
410 const T delta = m_radius * std::tan(hInRadians);
411
412 const auto& hitBasis = math::TOrthonormalBasis3<T>::makeFromUnitY(normal);
413 const math::TVector3<T>& dx = hitBasis.getXAxis().mul(delta);
414 const math::TVector3<T>& dz = hitBasis.getZAxis().mul(delta);
415
416 // Compute partial derivatives with 2nd-order approximation
417
418 // Find delta positions on the sphere from displacement vectors
419 const math::TVector3<T>& negX = surface.sub(dx).normalize().mul(m_radius);
420 const math::TVector3<T>& posX = surface.add(dx).normalize().mul(m_radius);
421 const math::TVector3<T>& negZ = surface.sub(dz).normalize().mul(m_radius);
422 const math::TVector3<T>& posZ = surface.add(dz).normalize().mul(m_radius);
423
424 // Find delta uvw vectors
425 const math::TVector2<T>& negXuv = surfaceToUv(negX);
426 const math::TVector2<T>& posXuv = surfaceToUv(posX);
427 const math::TVector2<T>& negZuv = surfaceToUv(negZ);
428 const math::TVector2<T>& posZuv = surfaceToUv(posZ);
429
430 const math::TMatrix2<T> uvwDiff(
431 posXuv.x() - negXuv.x(), posXuv.y() - negXuv.y(),
432 posZuv.x() - negZuv.x(), posZuv.y() - negZuv.y());
433 const auto xDiff = posX - negX;
434 const auto zDiff = posZ - negZ;
435 const std::array<std::array<T, 2>, 3> bs = {
436 xDiff.x(), zDiff.x(),
437 xDiff.y(), zDiff.y(),
438 xDiff.z(), zDiff.z() };
439
440 // Calculate positional partial derivatives
441 math::TVector3<T> dPdU, dPdV;
442 std::array<std::array<T, 2>, 3> xs;
443 if(uvwDiff.solve(bs, &xs))
444 {
445 dPdU.x() = xs[0][0]; dPdV.x() = xs[0][1];
446 dPdU.y() = xs[1][0]; dPdV.y() = xs[1][1];
447 dPdU.z() = xs[2][0]; dPdV.z() = xs[2][1];
448 }
449 else
450 {
451 dPdU = hitBasis.getZAxis();
452 dPdV = hitBasis.getXAxis();
453 }
454
455 return {dPdU, dPdV};
456}
