16inline std::shared_ptr<ISdlResource> TSdl<void>::makeResource(const SdlClass* clazz)
17{
18 if(!clazz)

45template<CSdlResource T>
46inline std::shared_ptr<T> TSdl<T>::makeResource()
47{
48 static_assert(CHasSdlClassDefinition<T>,
49 "No SDL class definition found. Did you call PH_DEFINE_SDL_CLASS() in the body of type T?");
50
51 const SdlClass* clazz = T::getSdlClass();
52 PH_ASSERT(clazz);
53
54 std::shared_ptr<ISdlResource> resource = TSdl<>::makeResource(clazz);
55 if(!resource)
56 {
57 return nullptr;
58 }
59
60 // Obtain typed resource. This dynamic cast also guard against the case where
61 // `T` might not actually have SDL class defined locally but inherited (i.e., the resource
62 // created was in fact not of type `T`).
63 std::shared_ptr<T> typedResource = std::dynamic_pointer_cast<T>(std::move(resource));
64 if(!typedResource)
65 {
66 PH_DEFAULT_LOG_WARNING(
67 "default resource creation failed, {} may not have SDL class defined",
68 clazz->genPrettyName());
69 }
70 return typedResource;
71}

74template<typename... DeducedArgs>
75inline T TSdl<T>::make(DeducedArgs&&... args)
76{
77 static_assert(std::is_constructible_v<T, DeducedArgs...>,
78 "SDL class type T is not constructible using the specified arguments.");
79
80 T instance(std::forward<DeducedArgs>(args)...);
81 if constexpr(CHasSdlClassDefinition<T>)
82 {
83 const SdlClass* clazz = T::getSdlClass();
84 PH_ASSERT(clazz);
85
86 clazz->initDefaultResource(instance);
87 }
88 else if(CHasSdlStructDefinition<T>)
89 {
90 static_assert(CSdlStructSupportsInitToDefault<T>,
91 "SDL struct definition of T does not support initializing to default values");
92
93 const auto* ztruct = T::getSdlStruct();
94 PH_ASSERT(ztruct);
95
96 ztruct->initDefaultStruct(instance);
97 }
98 else
99 {
100 static_assert(CHasSdlClassDefinition<T> || CHasSdlStructDefinition<T>,
101 "No SDL class/struct definition found. Did you call "
102 "PH_DEFINE_SDL_CLASS()/PH_DEFINE_SDL_STRUCT() in the body of type T?");
103 }
104
105 return instance;
106}
