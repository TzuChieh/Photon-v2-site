149 {
150 PH_ASSERT(!m_isBetweenFrameBeginAndEnd);
151
152 // The worker thread must not be already initiated. If this assertion fails, it is likely
153 // that `startWorker()` is called more than once.
154 PH_ASSERT(!hasWorkerStarted());

219 inline void endFrame()
220 {
221 PH_ASSERT(isParentThread());
222 PH_ASSERT(hasWorkerStarted());
223
224 onEndFrame();
225
226 // Clear work queue memory (arena) on worker thread after all works are added
227 addClearArenaWork();
228
229 // Works can no longer be added
230
231#if PH_DEBUG
232 m_isBetweenFrameBeginAndEnd = false;
233#endif

238 // unfortunately we have to wait here otherwise we may cancel some work
239 m_thread.waitAllWorks();
240
241 // Request termination as early as possible since there are still some cleanups
242 // to do on worker thread
243 m_thread.requestTermination();
244 }

256 template<typename Func>
257 inline void addWork(Func&& workFunc)
258 {
259 if constexpr(Work::template TIsStorableFunctor<Func>::value)
260 {
261 addWork(Work(std::forward<Func>(workFunc)));
262 }
263 else
264 {
265 // If `Func` cannot be stored in the internal buffer of `TFunction`, then
266 // we allocate its space in the arena and call it through a wrapper
267

269 PH_ASSERT(funcPtr);
270
271 addWork(Work(
272 [funcPtr](Args... args) -> R
273 {
274 return (*funcPtr)(std::forward<Args>(args)...);
275 }));
276 }
