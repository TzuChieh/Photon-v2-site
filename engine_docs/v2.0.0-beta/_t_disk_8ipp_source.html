36template<typename T>
37inline TVector2<T> TDisk<T>::sampleToSurface2D(const std::array<T, 2>& sample) const
38{
39 PH_ASSERT_IN_RANGE_INCLUSIVE(sample[0], static_cast<T>(0), static_cast<T>(1));
40 PH_ASSERT_IN_RANGE_INCLUSIVE(sample[1], static_cast<T>(0), static_cast<T>(1));
41 PH_ASSERT_GE(m_radius, static_cast<T>(0));
42
43 const T r = std::sqrt(sample[0]);
44 const T phi = constant::two_pi<real> * sample[1];
45
46 const auto localUnitPos = TVector2<T>(
47 r * std::cos(phi),
48 r * std::sin(phi));
49
50 return localUnitPos.mul(m_radius);
51}
