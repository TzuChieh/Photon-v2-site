95 // FIXME: vec cast

98 // Get tristimulus color according to pixel buffer layout. Alpha is ignored for this texture.
99
100 PH_ASSERT_LE(pixel_texture::num_pixel_elements(m_colorLayout), getPixelBuffer()->numPixelElements());
101
102 math::TTristimulusValues<float64> color{};
103 switch(m_colorLayout)
104 {
105 case pixel_texture::EPixelLayout::R:
106 color[0] = sampledPixel[0];
107 break;
108
109 case pixel_texture::EPixelLayout::G:
110 color[1] = sampledPixel[0];
111 break;
112
113 case pixel_texture::EPixelLayout::B:
114 color[2] = sampledPixel[0];
115 break;
116
117 // FIXME: monochromatic may not mean same value on all components -> HSV?
118 case pixel_texture::EPixelLayout::Monochromatic:
119 color.fill(sampledPixel[0]);
120 break;
121
122 case pixel_texture::EPixelLayout::RG:
123 color[0] = sampledPixel[0];
124 color[1] = sampledPixel[1];
125 break;
126
127 case pixel_texture::EPixelLayout::RGB:
128 case pixel_texture::EPixelLayout::RGBA:
129 color[0] = sampledPixel[0];

132 break;
133
134 case pixel_texture::EPixelLayout::BGR:
135 case pixel_texture::EPixelLayout::BGRA:
136 color[0] = sampledPixel[2];

138 color[2] = sampledPixel[0];

140

142 color[0] = sampledPixel[1];

144 color[2] = sampledPixel[3];

146
147 case pixel_texture::EPixelLayout::ABGR:
148 color[0] = sampledPixel[3];
