78inline TMatrix4<T>& TMatrix4<T>::initRotation(const TQuaternion<T>& rot)
79{
80 m[0][0] = 1 - 2*(rot.y*rot.y + rot.z*rot.z);
81 m[0][1] = 2*(rot.x*rot.y - rot.z*rot.w);
82 m[0][2] = 2*(rot.y*rot.w + rot.x*rot.z);
83 m[0][3] = 0;
84
85 m[1][0] = 2*(rot.x*rot.y + rot.z*rot.w);
86 m[1][1] = 1 - 2*(rot.x*rot.x + rot.z*rot.z);
87 m[1][2] = 2*(rot.y*rot.z - rot.x*rot.w);
88 m[1][3] = 0;
89
90 m[2][0] = 2*(rot.x*rot.z - rot.y*rot.w);
91 m[2][1] = 2*(rot.y*rot.z + rot.x*rot.w);
92 m[2][2] = 1 - 2*(rot.x*rot.x + rot.y*rot.y);
93 m[2][3] = 0;
94
95 m[3][0] = 0;
96 m[3][1] = 0;
97 m[3][2] = 0;
98 m[3][3] = 1;
99
100 return *this;

133 const T aspectRatio,
134 const T zNear,
135 const T zFar)
136{

138 const T zRange = zNear - zFar;
