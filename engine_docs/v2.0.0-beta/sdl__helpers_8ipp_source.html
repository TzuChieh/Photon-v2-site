119template<typename Element>
120inline math::TVector2<Element> load_vector2(const std::string& sdlVector2Str)
121{
122 static const Tokenizer tokenizer({' ', '\t', '\n', '\r'}, {});
123
124 try
125 {
126 std::vector<std::string> tokens;
127 tokenizer.tokenize(sdlVector2Str, tokens);
128
129 // 2 input values form a vec2 exactly
130 if(tokens.size() == 2)
131 {
132 return math::TVector2<Element>(
133 load_number<Element>(tokens[0]),
134 load_number<Element>(tokens[1]));
135 }
136 // 1 input value results in vec2 filled with the same value
137 else if(tokens.size() == 1)
138 {
139 return math::TVector2<Element>(
140 load_number<Element>(tokens[0]));
141 }
142 else
143 {
144 throw SdlLoadError(
145 "invalid Vector2 representation "
146 "(number of values = " + std::to_string(tokens.size()) + ")");
147 }
148 }
149 catch(const SdlException& e)
150 {
151 throw SdlLoadError("on parsing Vector2 -> " + e.whatStr());
152 }
153}

155template<typename NumberType>
156inline std::vector<NumberType> load_number_array(const std::string& sdlNumberArrayStr)
157{
158 static const Tokenizer tokenizer({' ', '\t', '\n', '\r'}, {});
159
160 try
161 {
162 std::vector<std::string> arrayTokens;
163 tokenizer.tokenize(sdlNumberArrayStr, arrayTokens);
164
165 std::vector<NumberType> numberArray(arrayTokens.size());
166 for(std::size_t i = 0; i < numberArray.size(); ++i)
167 {
168 numberArray[i] = load_number<NumberType>(arrayTokens[i]);
169 }
170
171 return numberArray;
172 }
173 catch(const SdlException& e)
174 {
175 throw SdlLoadError("on parsing number array -> " + e.whatStr());
176 }

419template<typename DstType, typename SrcType>
420inline DstType* cast_to(SrcType* const srcResource)
421{
422 // `SrcType` and `DstType` are both possibly cv-qualified
423
424 static_assert(std::is_base_of_v<ISdlResource, SrcType>,
425 "Source resource must derive from ISdlResource.");
426
427 static_assert(std::is_base_of_v<ISdlResource, DstType>,
428 "Casted-to type must derive from ISdlResource.");
429
430 if(srcResource == nullptr)
431 {
432 throw SdlException("source resource is empty");
433 }
434
435 DstType* const dstResource = dynamic_cast<DstType*>(srcResource);
436 if(dstResource == nullptr)
437 {
438 throw_formatted<SdlException>(
439 "type cast error: source resource cannot be casted to the specified type (resource ID: {})",
440 srcResource->getId());
441 }
442
443 return dstResource;
444}
